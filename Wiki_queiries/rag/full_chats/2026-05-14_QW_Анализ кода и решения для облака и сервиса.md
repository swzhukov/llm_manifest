def safe_int(value: Any, default: int = 0) -> int:
    try: return int(float(value))
    except: return default

def ensure_audio_stream(file_path: Path) -> bool:
    try:
        res = subprocess.run(["ffprobe", "-v", "error", "-select_streams", "a:0",
                              "-show_entries", "stream=codec_type", "-of", "csv=p=0", str(file_path)],
                             capture_output=True, text=True, timeout=15)
        return "audio" in res.stdout.strip()
    except: return False

def convert_to_wav(input_path: Path, output_path: Path) -> bool:
    try:
        subprocess.run(["ffmpeg", "-y", "-i", str(input_path), "-ac", "1", "-ar", "16000",
                        "-sample_fmt", "s16", str(output_path)],
                       check=True, capture_output=True, timeout=120)
        return output_path.exists()
    except: return False

def run_newton_transcribe(wav_path: Path, output_txt: Path, engine: str, speakers: Optional[int] = None) -> Optional[str]:
    try:
        cmd = ["newton", "transcribe", str(wav_path), "-o", str(output_txt), "-e", engine]
        if engine == "diarize" and speakers is not None: cmd.extend(["-n", str(speakers)])
        subprocess.run(cmd, check=True, capture_output=True, text=True, timeout=600)
        if output_txt.exists(): return output_txt.read_text(encoding="utf-8").strip()
    except: pass
    return None

def transcribe_media(file_path: Path, cache: Dict[str, str], lock: threading.Lock) -> Optional[str]:
    cache_key = f"{file_path.resolve()}_{NEWTON_ENGINE}_{ENABLE_DIARIZATION}_{DIARIZE_SPEAKERS}"
    with lock:
        if CACHE_MEDIA_PROCESSING and cache_key in cache: return cache[cache_key]
    if not ensure_audio_stream(file_path):
        result = "[нет аудиодорожки]"
        with lock:
            if CACHE_MEDIA_PROCESSING: cache[cache_key] = result
        return result
    with tempfile.NamedTemporaryFile(suffix=".wav", delete=False) as tmp_wav:
        wav_path = Path(tmp_wav.name)
    if not convert_to_wav(file_path, wav_path): return None
    engine = "diarize" if (ENABLE_DIARIZATION and NEWTON_ENGINE in ("v3", "diarize")) else NEWTON_ENGINE
    with tempfile.NamedTemporaryFile(suffix=".txt", delete=False) as tmp_txt:
        txt_path = Path(tmp_txt.name)
    try:
        text = run_newton_transcribe(wav_path, txt_path, engine, DIARIZE_SPEAKERS)
        if text is None: text = run_newton_transcribe(wav_path, txt_path, "v3", None)
    finally:
        wav_path.unlink(missing_ok=True); txt_path.unlink(missing_ok=True)
    if text is not None:
        with lock:
            if CACHE_MEDIA_PROCESSING: cache[cache_key] = text
    return text

def ocr_image(file_path: Path, cache: Dict[str, str], lock: threading.Lock) -> Optional[str]:
    cache_key = f"ocr_{file_path.resolve()}_{OCR_LANG}"
    with lock:
        if CACHE_MEDIA_PROCESSING and cache_key in cache: return cache[cache_key]
    try:
        img = Image.open(file_path)
        text = pytesseract.image_to_string(img, lang=OCR_LANG).strip()
        result = text if text else None
    except: result = None
    with lock:
        if CACHE_MEDIA_PROCESSING: cache[cache_key] = result or ""
    return result

def extract_text_from_pptx(pptx_path: Path) -> Optional[str]:
    try:
        prs = Presentation(str(pptx_path))
        parts = []
        for i, slide in enumerate(prs.slides, 1):
            slide_text = []
            def extract_shape(shape):
                if shape.has_text_frame:
                    for para in shape.text_frame.paragraphs:
                        if para.text.strip(): slide_text.append(para.text.strip())
                if shape.has_table:
                    rows = ["| " + " | ".join(cell.text for cell in row.cells) + " |" for row in shape.table.rows]
                    if rows: slide_text.append("\n".join(rows))
                if shape.shape_type == MSO_SHAPE_TYPE.GROUP:
                    for child in shape.shapes: extract_shape(child)
            for shape in slide.shapes: extract_shape(shape)
            if slide.has_notes_slide and slide.notes_slide.notes_text_frame:
                notes = slide.notes_slide.notes_text_frame.text.strip()
                if notes: slide_text.append(f"[Заметки]: {notes}")
            if slide_text: parts.append(f"### Слайд {i}\n" + "\n".join(slide_text))
        return "\n\n".join(parts) if parts else None
    except: return None

def extract_text_from_docx(docx_path: Path) -> Optional[str]:
    try:
        doc = Document(str(docx_path))
        parts = [p.text for p in doc.paragraphs if p.text.strip()]
        for table in doc.tables:
            rows = ["| " + " | ".join(cell.text for cell in row.cells) + " |" for row in table.rows]
            if rows: parts.append("\n".join(rows))
        return "\n\n".join(parts) if parts else None
    except: return None

def extract_text_from_pdf(pdf_path: Path) -> Optional[str]:
    try:
        reader = PdfReader(str(pdf_path))
        text = "\n".join(p.extract_text() or "" for p in reader.pages).strip()
        if text: return text
    except: pass
    try:
        with pdfplumber.open(str(pdf_path)) as pdf:
            text = "\n".join(p.extract_text() or "" for p in pdf.pages).strip()
            if text: return text
    except: pass
    if ENABLE_OCR:
        with tempfile.TemporaryDirectory() as tmpdir:
            try:
                subprocess.run(["pdftoppm", "-png", "-r", "200", str(pdf_path), f"{tmpdir}/page"],
                               check=False, timeout=120)
                pages = []
                for img_file in sorted(Path(tmpdir).glob("*.png")):
                    img = Image.open(img_file)
                    t = pytesseract.image_to_string(img, lang=OCR_LANG).strip()
                    if t: pages.append(t)
                return "\n".join(pages) if pages else None
            except: return None
    return None

def extract_text_from_xlsx(xlsx_path: Path) -> Optional[str]:
    try:
        wb = load_workbook(str(xlsx_path), read_only=True, data_only=True)
        parts = []
        for name in wb.sheetnames:
            ws = wb[name]
            rows = list(ws.iter_rows(max_row=100, values_only=True))
            if not rows: continue
            header = rows[0]
            parts.append(f"### Лист: {name}")
            parts.append("| " + " | ".join(str(c) if c is not None else "" for c in header) + " |")
            parts.append("| " + " | ".join(["---"] * len(header)) + " |")
            for row in rows[1:]:
                parts.append("| " + " | ".join(str(c) if c is not None else "" for c in row) + " |")
            parts.append("")
        return "\n".join(parts) if parts else None
    except: return None

def extract_text_from_txt(file_path: Path) -> Optional[str]:
    for enc in ["utf-8", "cp1251", "latin-1"]:
        try: return file_path.read_text(encoding=enc).strip()
        except: continue
    return None

def find_media_file(base_dir: Path, file_path_str: str) -> Optional[Path]:
    decoded = unquote(file_path_str).lstrip("/").replace("\\", "/")
    candidate = base_dir / decoded
    if candidate.exists(): return candidate
    name = Path(decoded).name
    for sub in ["photos", "voice_messages", "video_files", "videos", "files",
                "stickers", "animations", "video_messages", "documents"]:
        cand = base_dir / sub / name
        if cand.exists(): return cand
    # рекурсивно с ограничением
    try:
        for depth, found in enumerate(base_dir.rglob(name)):
            if found.is_file(): return found
            if depth > 5000: break
    except: pass
    return None

class TelegramExportProcessor:
    AUDIO_VIDEO_EXTENSIONS = {".mp3", ".wav", ".ogg", ".m4a", ".flac", ".opus",
                              ".mp4", ".webm", ".mkv", ".avi", ".mov"}
    IMAGE_EXTENSIONS = {".jpg", ".jpeg", ".png", ".webp", ".bmp", ".gif", ".tiff", ".tif"}
    TRANSCRIBE_MEDIA_TYPES = {"audio", "voice_message", "video", "animation",
                              "video_file", "audio_file", "voice", "round_video"}
    TEXT_DOC_EXTENSIONS = {".pptx", ".ppt", ".docx", ".doc", ".pdf", ".xlsx", ".xls",
                           ".txt", ".md", ".csv", ".json", ".xml", ".html", ".log",
                           ".rtf", ".odt", ".ods", ".odp"}

    def __init__(self, export_root: Path):
        self.export_root = export_root
        self.lock = threading.Lock()
        self.media_cache: Dict[str, str] = {}
        if CACHE_MEDIA_PROCESSING:
            cache_file = export_root / "media_cache.json"
            if cache_file.exists():
                try:
                    with self.lock: self.media_cache = json.loads(cache_file.read_text("utf-8"))
                except: pass
        if EXPECTED_MEDIA_FILE:
            if not find_media_file(self.export_root, EXPECTED_MEDIA_FILE):
                raise FileNotFoundError(f"Обязательный файл {EXPECTED_MEDIA_FILE} не найден!")

    def save_cache(self):
        if CACHE_MEDIA_PROCESSING and self.media_cache:
            cache_file = self.export_root / "media_cache.json"
            try:
                with self.lock: cache_file.write_text(json.dumps(self.media_cache, ensure_ascii=False, indent=2), "utf-8")
            except: pass

    def extract_message_text(self, msg: dict) -> str:
        text = msg.get("text", "")
        if isinstance(text, str): return text
        if isinstance(text, list):
            return "".join(item if isinstance(item, str) else item.get("text", "") for item in text)
        return str(text) if text else ""

    def get_media_info(self, msg: dict) -> Optional[Tuple[str, str]]:
        # Новый формат
        media_type = msg.get("media_type")
        file_path = msg.get("file")
        if media_type and file_path: return (media_type, file_path)

        # Старый формат: ключи-строки или объекты
        for key in ["photo", "video", "document", "audio", "voice_message",
                    "sticker", "animation", "video_file", "audio_file"]:
            if key in msg:
                val = msg[key]
                # строка — путь
                if isinstance(val, str): return (key, val)
                # объект
                if isinstance(val, dict) and "file" in val: return (key, val["file"])

        # Только file без media_type
        if file_path: return ("document", file_path)

        return None

    def classify_media_action(self, media_type: str, file_path: str) -> str:
        ext = Path(file_path).suffix.lower()
        # Принудительная транскрибация для аудио/видео расширений и явных типов
        if media_type in self.TRANSCRIBE_MEDIA_TYPES or ext in self.AUDIO_VIDEO_EXTENSIONS:
            return "transcribe"
        # Изображения (стикеры, фото, а также любые файлы с графическими расширениями)
        if media_type in ("photo", "sticker") or ext in self.IMAGE_EXTENSIONS:
            return "ocr"
        # Документы
        if media_type in ("document", "file") or ext in self.TEXT_DOC_EXTENSIONS:
            return "document"
        # Неизвестное расширение — попробуем OCR, если похоже на картинку, иначе документ
        if ext in self.IMAGE_EXTENSIONS:
            return "ocr"
        return "document"  # на всякий случай попробуем прочитать как документ

    def process_media(self, media_type: str, relative_path: str) -> Optional[str]:
        file_path = find_media_file(self.export_root, relative_path)
        if file_path is None: return "(файл отсутствует)"

        action = self.classify_media_action(media_type, relative_path)
        ext = file_path.suffix.lower()

        if VERBOSE_MEDIA_LOG:
            size_mb = file_path.stat().st_size / (1024*1024)
            print(f"[МЕДИА] тип={media_type} | путь={relative_path} | найдено={file_path} | "
                  f"размер={size_mb:.2f} MB | расширение={ext} | действие={action}")

        if action == "transcribe":
            return transcribe_media(file_path, self.media_cache, self.lock)
        elif action == "ocr":
            return ocr_image(file_path, self.media_cache, self.lock)
        elif action == "document":
            if ext in {".pptx", ".ppt"}: return extract_text_from_pptx(file_path)
            elif ext in {".docx", ".doc"}: return extract_text_from_docx(file_path) or "(неподдерживаемый формат .doc)"
            elif ext == ".pdf": return extract_text_from_pdf(file_path)
            elif ext in {".xlsx", ".xls"}: return extract_text_from_xlsx(file_path)
            elif ext in {".txt", ".md", ".csv", ".json", ".xml", ".html", ".log", ".rtf", ".odt", ".ods", ".odp"}:
                return extract_text_from_txt(file_path)
            else: return extract_text_from_txt(file_path) or "(не удалось извлечь текст)"
        return "(медиа не требует обработки)"

    def process_all_messages(self) -> List[dict]:
        json_files = list(self.export_root.rglob("result.json"))
        if not json_files: raise FileNotFoundError("result.json не найден")
        result_json = json_files[0]

        if DEBUG_MODE:
            print(f"Найден result.json: {result_json} ({result_json.stat().st_size/1024/1024:.2f} MB)")

        with open(result_json, "r", encoding="utf-8") as f: data = json.load(f)
        messages = data.get("messages", [])
        if not messages: raise ValueError("Нет поля 'messages'")
        chat_messages = [m for m in messages if m.get("type") == "message"]
        print(f"Всего сообщений в чате: {len(chat_messages)}")

        media_tasks = {}
        records = []
        for idx, msg in enumerate(chat_messages):
            date_unixtime = safe_int(msg.get("date_unixtime", 0))
            dt = datetime.fromtimestamp(date_unixtime, tz=timezone.utc)
            dt_str = dt.strftime("%d.%m.%Y %H:%M:%S")
            from_name = msg.get("from", "Unknown")
            from_id = safe_int(msg.get("from_id", 0))
            text = self.extract_message_text(msg)

            media_info = self.get_media_info(msg)
            record = {
                "timestamp": dt_str,
                "from": from_name,
                "from_id": from_id,
                "text": text,
                "media_type": None,
                "media_file": None,
                "media_description": None,
            }
            if media_info:
                media_type, rel_path = media_info
                record["media_type"] = media_type
                record["media_file"] = rel_path
                media_tasks[idx] = (media_type, rel_path)
            records.append(record)

        if media_tasks:
            print(f"Медиа для обработки: {len(media_tasks)}")
            with ThreadPoolExecutor(max_workers=MAX_PARALLEL_WORKERS) as executor:
                future_to_idx = {}
                for idx, (media_type, rel_path) in media_tasks.items():
                    future = executor.submit(self.process_media, media_type, rel_path)
                    future_to_idx[future] = idx

                for future in tqdm(as_completed(future_to_idx), total=len(future_to_idx),
                                   desc="Обработка медиа"):
                    idx = future_to_idx[future]
                    try:
                        result = future.result()
                        records[idx]["media_description"] = result
                    except Exception as e:
                        records[idx]["media_description"] = f"(ошибка: {e})"
                        print(f"Ошибка обработки медиа #{idx}: {e}")

        self.save_cache()
        return records

def generate_markdown(records: List[dict]) -> str:
    lines = []
    for r in records:
        header = f"[{r['timestamp']}] {r['from']} (ID: {r['from_id']})"
        lines.append(header)
        if r["text"]: lines.append(r["text"])
        if r["media_type"]:
            lines.append(f"[{r['media_type']}]: {r['media_file']}")
            if r["media_description"]: lines.append(r["media_description"])
        lines.append("")
    return "\n".join(lines)

def split_markdown_by_size(md_text: str, max_bytes: int) -> List[str]:
    pattern = r'(?=\n\n\[\d{2}\.\d{2}\.\d{4}\s\d{2}:\d{2}:\d{2}\])'
    parts = re.split(pattern, "\n\n" + md_text)
    parts = [p.lstrip("\n") for p in parts if p.strip()]
    volumes = []
    cur_vol, cur_size = "", 0
    for part in parts:
        part_size = len(part.encode("utf-8"))
        if cur_size > 0 and cur_size + part_size > max_bytes:
            volumes.append(cur_vol.strip())
            cur_vol, cur_size = "", 0
        cur_vol += part + "\n\n"
        cur_size += part_size + 2
    if cur_vol.strip(): volumes.append(cur_vol.strip())
    return volumes

def save_outputs(records: List[dict], md_full: str, output_dir: Path, timestamp_str: str, max_size_mb: int):
    output_dir.mkdir(parents=True, exist_ok=True)
    base_name = f"telegram_export_{timestamp_str}"
    # CSV
    csv_path = output_dir / f"{base_name}.csv"
    df = pd.DataFrame(records)[["timestamp", "from", "from_id", "text",
                                 "media_type", "media_file", "media_description"]]
    df.to_csv(csv_path, index=False, encoding="utf-8")
    print(f"CSV сохранён: {csv_path}")
    # Markdown
    max_bytes = max_size_mb * 1024 * 1024
    md_bytes = len(md_full.encode("utf-8"))
    if md_bytes <= max_bytes:
        md_path = output_dir / f"{base_name}.md"
        md_path.write_text(md_full, encoding="utf-8")
        print(f"Markdown сохранён: {md_path} ({md_bytes/1024/1024:.2f} MB)")
    else:
        volumes = split_markdown_by_size(md_full, max_bytes)
        for i, vol in enumerate(volumes, 1):
            part_name = f"{base_name}_part{i}.md"
            part_path = output_dir / part_name
            part_path.write_text(vol, encoding="utf-8")
            print(f"Том {i} сохранён: {part_path} ({len(vol.encode('utf-8'))/1024/1024:.2f} MB)")
    # Скачивание в Colab
    if IN_COLAB:
        for f in sorted(output_dir.glob(f"{base_name}*")):
            colab_files.download(str(f))

def main():
    if LOCAL_EXPORT_DIR:
        source_path = Path(LOCAL_EXPORT_DIR)
        if not source_path.exists(): raise FileNotFoundError(f"Не найден: {LOCAL_EXPORT_DIR}")
        if source_path.is_file() and source_path.suffix.lower() == ".zip":
            export_root = TEMP_ROOT / "export"
            if export_root.exists(): shutil.rmtree(export_root)
            export_root.mkdir(parents=True)
            with zipfile.ZipFile(source_path, "r") as zf: zf.extractall(export_root)
        else: export_root = source_path
    elif AUTO_ARCHIVE_NAME:
        zip_path = UPLOAD_DIR / AUTO_ARCHIVE_NAME
        if not zip_path.exists(): raise FileNotFoundError(f"Архив не найден: {zip_path}")
        export_root = TEMP_ROOT / "export"
        if export_root.exists(): shutil.rmtree(export_root)
        export_root.mkdir(parents=True)
        with zipfile.ZipFile(zip_path, "r") as zf: zf.extractall(export_root)
    else:
        if IN_COLAB:
            print("Загрузите ZIP-архив экспорта Telegram")
            uploaded = colab_files.upload()
            if not uploaded: raise RuntimeError("Файл не загружен")
            zip_name = next(iter(uploaded))
            zip_path = UPLOAD_DIR / zip_name
            shutil.move(zip_name, str(zip_path))
            export_root = TEMP_ROOT / "export"
            if export_root.exists(): shutil.rmtree(export_root)
            export_root.mkdir(parents=True)
            with zipfile.ZipFile(zip_path, "r") as zf: zf.extractall(export_root)
        else:
            raise RuntimeError("Не в Colab и не задан LOCAL_EXPORT_DIR/AUTO_ARCHIVE_NAME")

    print(f"Рабочая папка экспорта: {export_root}")

    processor = TelegramExportProcessor(export_root)
    records = processor.process_all_messages()

    total = len(records)
    with_media = sum(1 for r in records if r["media_type"])
    transcribe_done = sum(1 for r in records if r["media_description"] and r["media_description"] not in ("(файл отсутствует)", None))
    errors = sum(1 for r in records if r["media_description"] and r["media_description"].startswith("(ошибка"))
    print(f"\nСтатистика: всего сообщений {total}, с медиа {with_media}, успешно обработано {transcribe_done}, ошибок {errors}")

    if SHOW_SAMPLE_TRANSCRIPTS:
        sample_count = 0
        for r in records:
            if r["media_description"] and not r["media_description"].startswith(("(ошибка","(файл отсутствует")) and r["media_description"] != "[нет аудиодорожки]":
                print(f"\n--- Образец расшифровки ---")
                print(f"Медиа: {r['media_file']}")
                print(r["media_description"][:500])
                sample_count += 1
                if sample_count >= 3: break

    md_text = generate_markdown(records)
    timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")
    save_outputs(records, md_text, Path(OUTPUT_DIR), timestamp, MAX_FILE_SIZE_MB)

        print("Временная папка удалена.")

if __name__ == "__main__":

    except Exception:
        traceback.print_exc()
        print("Скрипт завершился с ошибкой. Проверьте лог.")

## Анализ текущего решения

Ваш код — это мощный конвейер, который превращает сырой JSON‑экспорт Telegram в структурированный Markdown‑файл, готовый для загрузки в LLM. Он работает в Google Colab и состоит из трёх ключевых слоёв:

- **Кэш** — страховка от повторной обработки тысяч медиа при отладке.  
- **Разбиение на части** — чтобы Markdown‑файл можно было скормить даже моделям с ограниченным контекстом.

## 1. Брать архив не из локальной загрузки, а с Облака Mail.ru или Яндекс.Диска

Самый простой и надёжный способ — получить **прямую ссылку на скачивание ZIP‑архива** и загрузить его в Colab через `wget` или `curl`.  
Вы можете разместить экспортированный архив в облаке и предоставить публичный доступ. Дальше код сам скачает его во временную папку.

### 1.1. Получение прямой ссылки

#### Яндекс.Диск
1. Загрузите ZIP‑файл на Диск, откройте к нему доступ по ссылке (кнопка «Поделиться»).  
2. Скопируйте публичную ссылку вида:  
   `https://disk.yandex.ru/d/XXXXXXXXXXXXXXX`  
3. Получите прямую ссылку через API (можно вручную в браузере):  
   ```
   https://cloud-api.yandex.net/v1/disk/public/resources/download?public_key=https://disk.yandex.ru/d/XXXXXXXXXXXXXXX
   ```
   Ответ будет содержать поле `href` — это и есть временная прямая ссылка.  
   Пример:  
   ```bash
   curl -s "https://cloud-api.yandex.net/v1/disk/public/resources/download?public_key=..." | jq -r .href
   ```
   Эту команду можно вставить прямо в Colab.

#### Облако Mail.ru
1. Загрузите файл, нажмите «Поделиться» → «Создать ссылку».  
2. Скопируйте публичную ссылку вида:  
   `https://cloud.mail.ru/public/XXXX/YYYYYYYYYY`  
3. Прямую ссылку можно получить, добавив `?weblink=1` и распарсив JSON.  
   Пример:  
   ```bash
   curl -s "https://cloud.mail.ru/public/XXXX/YYYYYYYYYY?weblink=1" | jq -r '.weblink_get.weblink'
   ```
   Это даст URL для скачивания.

### 1.2. Модификация кода

Вместо блока загрузки через `colab.files.upload()` мы добавим переменную `ARCHIVE_URL` и будем скачивать файл, только если она задана. Логика `main()` изменится так:

```python
ARCHIVE_URL = ""  # вставьте сюда прямую ссылку на ZIP

def main():
    if LOCAL_EXPORT_DIR:
        # ... существующая логика
    elif ARCHIVE_URL:
        zip_path = UPLOAD_DIR / "export.zip"
        print(f"Скачиваю архив из облака...")
        subprocess.run(["wget", "-O", str(zip_path), ARCHIVE_URL], check=True)
        # дальше распаковка как обычно
    elif AUTO_ARCHIVE_NAME:
        # ... как раньше
    else:
        # ... загрузка вручную
```

**Преимущества:**  
- Не нужно хранить файл локально или загружать через медленный интерфейс.  
- Архив загружается напрямую в среду выполнения, объём ограничен только диском Colab (≈70 ГБ).  
- Прямая ссылка одноразовая и не требует хранения токенов облака в коде.

## 2. Боремся с отключением Colab и думаем о сервисе

Colab действительно отключает среду через 6–12 часов (в бесплатной версии) и при простое интерактивного окна. Но ваша задача — обработка большого экспорта — может длиться долго из‑за тысяч медиафайлов. Есть два уровня решения.

### 2.1. Остаёмся в Colab, но повышаем живучесть

1. **Постоянная активность**  
   Вы уже используете `tqdm`, это хорошо. Можно добавить периодический вывод в консоль (каждые 5 минут), чтобы сессия не считалась бездействующей.  
   Вставьте в начало `process_all_messages` отдельный поток:

   ```python
   import time, threading
   def keep_alive():
       while True:
           time.sleep(300)
           print(f"[keepalive] {datetime.now()}", flush=True)
   threading.Thread(target=keep_alive, daemon=True).start()
   ```

2. **Чекпоинты**  
   Сохраняйте промежуточные результаты (кеш медиа уже есть). Можно каждые N сообщений дописывать CSV и Markdown‑фрагмент, чтобы при падении продолжить с того же места.  
   Для этого `TelegramExportProcessor` должен уметь пропускать уже обработанные сообщения по ID. Добавьте поле `processed_ids` и сохранение прогресса в файл.

3. **Colab Pro/Pro+**  
   Дают приоритетные серверы, больше памяти и до 24 часов непрерывной работы. Если бюджет позволяет — самый простой путь.

### 2.2. Переезд на Яндекс.Облако (создание собственного мини‑сервиса)

Вы просили «научить пользоваться» средствами Яндекса. Давайте разберём, как превратить ваш скрипт в автономный сервис, который не зависит от браузера и может работать сутками. Яндекс.Облако предлагает несколько подходящих сервисов:

- **Yandex Compute Cloud** – обычная виртуальная машина (аналог VDS).

Для вашей задачи (продолжительная ресурсоёмкая обработка) лучше всего подойдёт **Compute Cloud** — вы получите полноценный Linux‑сервер, где можно развернуть скрипт и гонять его хоть неделями.

Для вашей задачи превращения архива Telegram в данные, пригодные для анализа языковой моделью, существует несколько готовых инструментов. ### 🧩 Готовые решения с GitHub

Вот несколько интересных проектов, которые можно взять за основу или использовать как отправную точку.

*   **Chatpack** (CLI/Веб): Это, пожалуй, самый близкий к вашей задаче инструмент. Он нацелен именно на оптимизацию чатов для LLM, убирая до 92% токенов, которые тратятся на JSON-разметку. На вход принимает JSON из Telegram, а на выходе дает чистый CSV. Плюсом является скорость и малый расход токенов, минусом — то, что он не работает с медиафайлами, обрабатывая только текст[reference:0][reference:1].
*   **tglens**: Это уже не просто конвертер, а целая дашборд-аналитика с RAG-компонентой. Он работает полностью локально, обеспечивая приватность, и позволяет не только смотреть статистику, но и задавать вопросы по истории чата. Однако для его использования потребуется экспорт из десктопной версии Telegram[reference:2][reference:3].
*   **DialogSmith** и **ai-clone-telegram-chatbot**: Эти проекты ориентированы на файнтюнинг (дообучение) моделей на ваших данных, чтобы воспроизводить ваш стиль общения. Они могут быть полезны, если ваша конечная цель более амбициозна, чем просто анализ[reference:4][reference:5].

### ☁️ Компоненты Яндекс.Облака для вашего конвейера

Переход на Яндекс.Облако дает доступ к мощным нативным сервисам, которые могут закрыть все потребности вашего пайплайна.

*   **Yandex SpeechKit**: Решает задачу расшифровки аудиосообщений. API поддерживает асинхронное распознавание длинных аудио (что важно для больших чатов) и даже определение говорящего. Это более технологичный и точный вариант, чем используемый вами сейчас `Newton CLI`, который может работать нестабильно[reference:6][reference:7].
*   **Yandex Vision OCR**: Идеальный инструмент для извлечения текста из изображений и PDF. Он предлагает разные модели для разных типов контента: `page` для простого текста, `table` для таблиц, `handwritten` для рукописного текста и даже `passport` для распознавания документов. Это полностью покрывает вашу потребность в OCR[reference:8][reference:9].
*   **YandexGPT**: Мощная языковая модель, которая может работать с полученными текстами. Вы можете не только отправлять ей данные на анализ, но и дообучать её в Yandex DataSphere на своих данных, чтобы повысить качество ответов в вашей предметной области[reference:10][reference:11].

### 🚀 Ваш прототип на Yandex.Cloud

Собрать работающий прототип в Yandex.Cloud можно из кирпичиков готового кода:

*   **Telegram-бот на Yandex.Cloud Functions**: Можно взять за основу проект **Sumo**. Это бот, который уже умеет читать сообщения, взаимодействовать с YandexGPT и хранить историю в Yandex Database (YDB). Его код может стать каркасом для вашего решения[reference:12][reference:13].
*   **Распознавание речи через SpeechKit**: Используйте готовые SDK для PHP или Python. Например, библиотека `yandex-gpt-api` работает как с YandexGPT, так и с Yandex SpeechKit, предоставляя удобные функции для синхронного и асинхронного распознавания[reference:14][reference:15].
*   **Распознавание текста через Vision OCR**: API имеет простую интеграцию. Существуют готовые врапперы, например, нода для n8n `n8n-yandex-ocr`, которая позволяет отправлять файлы в Vision OCR и получать текст[reference:16].

### 📝 Пошаговый план: от идеи к прототипу

Вот конкретные шаги, чтобы запустить ваш сервис.

1.  **Настройка окружения**: Зарегистрируйтесь в Yandex.Cloud, создайте каталог и сервисный аккаунт с ролями `ai.speechkit-stt.user`, `ai.vision.user` и `storage.uploader`.
2.  **Хранение данных**: Создайте бакет в Yandex Object Storage для загрузки ZIP-архивов от пользователей.
3.  **Запуск фоновой обработки**: Создайте Cloud Function, которая будет запускаться по триггеру при появлении нового ZIP-файла в бакете. Она должна выполнять следующие шаги:
    *   Распаковать архив во временное хранилище.
    *   Извлечь текстовые сообщения из JSON.
    *   Для каждого аудиофайла отправить его на распознавание в SpeechKit API.
    *   Для каждого изображения отправить его на распознавание в Vision OCR API.
    *   Собрать весь полученный текст в единый Markdown-файл.
    *   Сохранить итоговый Markdown-файл обратно в Object Storage.
4.  **Интерфейс**: Вы можете создать простой веб-интерфейс или Telegram-бота, который будет давать пользователю ссылку на скачивание готового Markdown-файла из Object Storage. Для этого подойдет Yandex API Gateway.
