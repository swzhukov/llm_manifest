class WebDAVCloudManager:
    def __init__(self, login, password):
        self.login = login
        self.setup_webdav_client(login, password)

    def setup_webdav_client(self, login, password):
        """Настраивает WebDAV клиент для Облака Mail.ru"""
        options = {
            'webdav_hostname': 'https://webdav.cloud.mail.ru',
            'webdav_login': login,
            'webdav_password': password,
            'webdav_root': '/',
            'webdav_timeout': 30
        }

        try:
            self.client = Client(options)
            self.client.verify = False

            # Проверяем подключение
            if self.client.check('/'):
                print("✅ Успешное подключение к Облаку Mail.ru")
            else:
                print("❌ Не удалось подключиться к Облаку Mail.ru")
                return False
            return True
        except Exception as e:
            print(f"❌ Ошибка настройки WebDAV клиента: {e}")
            return False

    def ensure_cloud_directories(self):
        """Создает необходимые директории в облаке если их нет"""
        directories = ['/transcription_results', '/transcription_results/progress',
                      '/transcription_results/results', '/transcription_results/chunks']

        for directory in directories:
            try:
                if not self.client.check(directory):
                    self.client.mkdir(directory)
                    print(f"✅ Создана директория в облаке: {directory}")
            except Exception as e:
                print(f"⚠️ Ошибка создания директории {directory}: {e}")

    def upload_file(self, local_path, cloud_path):
        """Загружает файл в облако"""
        try:
            # Создаем родительские директории если нужно
            cloud_dir = os.path.dirname(cloud_path)
            if cloud_dir and not self.client.check(cloud_dir):
                self.client.mkdir(cloud_dir)

            self.client.upload_sync(remote_path=cloud_path, local_path=local_path)
            print(f"✅ Файл загружен в облако: {cloud_path}")
            return True
        except Exception as e:
            print(f"❌ Ошибка загрузки файла {local_path} в {cloud_path}: {e}")
            return False

    def download_file(self, cloud_path, local_path):
        """Скачивает файл из облака"""
        try:
            if not self.client.check(cloud_path):
                return False  # Файла нет в облаке - это нормально

            self.client.download_sync(remote_path=cloud_path, local_path=local_path)
            print(f"✅ Файл скачан из облака: {cloud_path}")
            return True
        except Exception as e:
            # Не выводим ошибку если файла просто нет
            if "not found" in str(e).lower() or "404" in str(e):
                return False
            print(f"❌ Ошибка скачивания файла {cloud_path}: {e}")
            return False

    def file_exists(self, cloud_path):
        """Проверяет существование файла в облаке"""
        try:
            return self.client.check(cloud_path)
        except:
            return False

    def split_long_audio(self, file_path, chunk_duration=3600000):  # 1 час = 3,600,000 мс
        """Разбивает длинное аудио на чанки по 1 часу"""
        print(f"   📦 Разбиваю длинное аудио на чанки по 1 часу...")

        try:
            audio = AudioSegment.from_file(file_path)
            duration_ms = len(audio)

            if duration_ms <= chunk_duration:
                return [file_path]

            chunks = []
            chunk_dir = os.path.join(self.local_base_dir, "chunks")
            os.makedirs(chunk_dir, exist_ok=True)

            base_name = os.path.splitext(os.path.basename(file_path))[0]

            for i, start_time in enumerate(range(0, duration_ms, chunk_duration)):
                end_time = min(start_time + chunk_duration, duration_ms)
                chunk = audio[start_time:end_time]

                chunk_path = os.path.join(chunk_dir, f"{base_name}_chunk_{i+1:03d}.wav")
                chunk.export(chunk_path, format="wav", parameters=["-ac", "1", "-ar", "16000"])
                chunks.append(chunk_path)

                print(f"   ✅ Чанк {i+1}: {start_time/1000/60:.1f}-{end_time/1000/60:.1f} мин")

            print(f"   📊 Создано {len(chunks)} чанков")
            return chunks

        except Exception as e:
            print(f"   ❌ Ошибка разбивки аудио: {e}")
            return [file_path]

    def prepare_audio(self, file_path):
        """Конвертирует аудио в WAV если нужно"""
        try:
            file_ext = os.path.splitext(file_path)[1].lower()

            if file_ext == '.wav':
                return file_path

            if file_ext not in self.supported_formats:
                print(f"   ⚠️ Неподдерживаемый формат: {file_ext}")
                return file_path

            print("   🔄 Конвертирую в WAV...")
            audio = AudioSegment.from_file(file_path)
            audio = audio.set_channels(1).set_frame_rate(16000)

            wav_path = file_path + '.wav'
            audio.export(wav_path, format="wav")
            print("   ✅ Конвертация завершена")
            return wav_path

        except Exception as e:
            print(f"   ❌ Ошибка конвертации: {e}")
            return file_path

    def transcribe_chunk(self, chunk_path, chunk_index, total_chunks):
        """Транскрибирует один чанк"""
        print(f"   🎯 Обрабатываю чанк {chunk_index}/{total_chunks}...")

        try:
            with open(chunk_path, 'rb') as f:
                response = requests.post(
                    f"{self.asr_url}/start_transcribing",
                    files={'file': f},
                    params={'diarize': 'true', 'remove_timestamps': 'false'},
                    headers=self.headers,
                    timeout=120
                )

            if response.status_code != 200:
                print(f"   ❌ Ошибка загрузки: {response.text}")
                return None

            task_data = response.json()
            task_id = task_data.get('task_id')
            if not task_id:
                print("   ❌ Не получили ID задачи")
                return None

            max_attempts = 90
            attempt = 0
            last_progress = 0

            while attempt < max_attempts:
                attempt += 1

                try:
                    status_response = requests.get(
                        f"{self.asr_url}/get_status",
                        params={'task_id': task_id},
                        headers=self.headers,
                        timeout=60
                    )

                    if status_response.status_code == 200:
                        status_data = status_response.json()
                        progress = status_data.get('progress', 0)
                        status = status_data.get('status')

                        if progress > last_progress:
                            bar = "█" * (progress // 5) + "▒" * (20 - progress // 5)
                            print(f"   [{bar}] {progress}% - {status}")
                            last_progress = progress
                        elif attempt % 10 == 0:
                            print(f"   ⏳ Прогресс: {progress}%, Статус: {status}")

                        if status == 'done':
                            print("   ✅ Обработка завершена!")
                            break
                        elif status == 'error':
                            print(f"   ❌ Ошибка обработки: {status_data.get('error', 'Неизвестная ошибка')}")
                            return None
                    else:
                        print(f"   ⚠️ Ошибка статуса: {status_response.status_code}")

                except Exception as e:
                    print(f"   ⚠️ Ошибка запроса: {e}")

                time.sleep(10)

                if attempt % 5 == 0:
                    print("   💾 Сохраняю промежуточный прогресс...")

            result_response = requests.get(
                f"{self.asr_url}/get_file",
                params={'task_id': task_id},
                headers=self.headers,
                timeout=60
            )

            if result_response.status_code == 200:
                try:
                    result_data = result_response.json()
                    text = result_data.get('text', '') or result_data.get('result', '')

                    if text:
                        chunk_header = f"\n\n[ЧАНК {chunk_index}/{total_chunks}]\n"
                        return chunk_header + text
                    else:
                        print("   ❌ Пустой текст в ответе")
                        return None

                except Exception as e:
                    result_text = result_response.text
                    if result_text:
                        chunk_header = f"\n\n[ЧАНК {chunk_index}/{total_chunks}]\n"
                        return chunk_header + result_text
                    else:
                        print("   ❌ Ошибка парсинга ответа")
                        return None
            else:
                print(f"   ❌ Ошибка получения результата: {result_response.status_code}")
                return None

        except Exception as e:
            print(f"   ❌ Ошибка обработки чанка: {e}")
            return None

    def transcribe_file(self, filename):
        """Обрабатывает один файл"""
        file_path = os.path.join(self.local_upload_dir, filename)
        print(f"\n🎧 Обрабатываю: {filename}")

        progress_data = self.load_progress()
        file_key = filename

        if file_key in progress_data and progress_data[file_key].get('completed'):
            print("   ✅ Файл уже обработан ранее")
            result_file = progress_data[file_key].get('result_file')
            if result_file and os.path.exists(result_file):
                with open(result_file, 'r', encoding='utf-8') as f:
                    return f.read()

        audio_path = self.prepare_audio(file_path)

        try:
            chunks = self.split_long_audio(audio_path)
            total_chunks = len(chunks)

            if total_chunks > 1:
                print(f"   📊 Будет обработано {total_chunks} чанков")

            full_text = f"ФАЙЛ: {filename}\n"
            full_text += f"ВРЕМЯ: {time.strftime('%Y-%m-%d %H:%M:%S')}\n"
            full_text += f"ЧАНКОВ: {total_chunks}\n"
            full_text += "=" * 50 + "\n\n"

            completed_chunks = 0
            chunk_results = []

            for i, chunk_path in enumerate(chunks, 1):
                chunk_key = f"{file_key}_chunk_{i}"
                if chunk_key in progress_data:
                    print(f"   ✅ Чанк {i} уже обработан")
                    chunk_text = progress_data[chunk_key].get('text', '')
                    chunk_results.append(chunk_text)
                    completed_chunks += 1
                    continue

                chunk_text = self.transcribe_chunk(chunk_path, i, total_chunks)

                if chunk_text:
                    chunk_results.append(chunk_text)
                    completed_chunks += 1

                    progress_data[chunk_key] = {
                        'text': chunk_text,
                        'timestamp': time.time()
                    }
                    self.save_progress(progress_data)

                    print(f"   ✅ Чанк {i} завершен ({completed_chunks}/{total_chunks})")
                else:
                    print(f"   ❌ Ошибка в чанке {i}")
                    chunk_results.append(f"\n\n[ЧАНК {i} - ОШИБКА ОБРАБОТКИ]\n")

            full_text += "".join(chunk_results)

            result_filename = os.path.splitext(filename)[0] + '_transcript.txt'
            result_filepath = os.path.join(self.local_results_dir, result_filename)

            with open(result_filepath, 'w', encoding='utf-8') as f:
                f.write(full_text)

            self.save_result_to_cloud(result_filepath, result_filename)

            progress_data[file_key] = {
                'completed': True,
                'result_file': result_filepath,
                'cloud_file': f"{self.cloud_results_dir}/{result_filename}",
                'chunks_processed': completed_chunks,
                'total_chunks': total_chunks,
                'completion_time': time.strftime('%Y-%m-%d %H:%M:%S')
            }
            self.save_progress(progress_data)

            print(f"✅ Обработка завершена: {completed_chunks}/{total_chunks} чанков")
            return full_text

        except Exception as e:
            print(f"❌ Общая ошибка обработки: {e}")
            return None
        finally:
            if audio_path != file_path and os.path.exists(audio_path):
                try:
                    os.remove(audio_path)

                    pass
