class FileHashManager:
    """Менеджер для работы с хешами файлов"""

    def __init__(self, cloud_manager: 'WebDAVCloudManager'):
        self.cloud_manager = cloud_manager
        self.local_hashes_file = "/content/transcription_results/file_hashes.json"
        self.cloud_hashes_file = "/transcription_results/file_hashes.json"

        # Создаем локальную директорию
        os.makedirs(os.path.dirname(self.local_hashes_file), exist_ok=True)

    def compute_file_hash(self, file_path: str) -> str:
        """Вычисляет MD5 хеш файла"""
        try:
            hash_md5 = hashlib.md5()
            with open(file_path, "rb") as f:
                for chunk in iter(lambda: f.read(4096), b""):
                    hash_md5.update(chunk)
            return hash_md5.hexdigest()
        except Exception as e:
            print(f"❌ Ошибка вычисления хеша файла {file_path}: {e}")
            return ""

    def load_hashes(self) -> Dict[str, str]:
        """Загружает хеши файлов из облака"""
        try:
            if self.cloud_manager.download_file(self.cloud_hashes_file, self.local_hashes_file):
                with open(self.local_hashes_file, 'r', encoding='utf-8') as f:
                    return json.load(f)
            return {}
        except Exception as e:
            print(f"⚠️ Ошибка загрузки хешей: {e}")
            return {}

    def save_hashes(self, hashes_data: Dict[str, str]) -> bool:
        """Сохраняет хеши файлов в облако"""
        try:
            with open(self.local_hashes_file, 'w', encoding='utf-8') as f:
                json.dump(hashes_data, f, ensure_ascii=False, indent=2)

            return self.cloud_manager.upload_file(self.local_hashes_file, self.cloud_hashes_file)
        except Exception as e:
            print(f"⚠️ Ошибка сохранения хешей: {e}")
            return False

    def is_file_processed(self, filename: str, file_hash: str) -> Tuple[bool, str]:
        """Проверяет, был ли файл уже обработан"""
        hashes_data = self.load_hashes()

        if filename in hashes_data:
            if hashes_data[filename] == file_hash:
                return True, "Файл уже обработан (совпадает хеш)"
            else:
                return False, "Файл с таким именем уже обработан, но изменился"

        return False, "Файл еще не обработан"

    def mark_file_processed(self, filename: str, file_hash: str) -> bool:
        """Помечает файл как обработанный"""
        hashes_data = self.load_hashes()
        hashes_data[filename] = file_hash
        return self.save_hashes(hashes_data)

class APIClient:
    """Клиент для работы с API транскрипции"""

    def __init__(self, token: str):
        self.token = token
        self.base_url = Config.ASR_URL
        self.headers = {"token": self.token}
        self.session = self._create_retry_session()

    def _create_retry_session(self) -> requests.Session:
        """Создает сессию с стратегией повторных попыток"""
        session = requests.Session()
        retry_strategy = Retry(
            total=Config.MAX_RETRIES,
            backoff_factor=Config.BACKOFF_FACTOR,
            status_forcelist=[429, 500, 502, 503, 504],
        )
        adapter = HTTPAdapter(max_retries=retry_strategy)
        session.mount("http://", adapter)
        session.mount("https://", adapter)
        return session

    def start_transcription(self, file_path: str) -> Optional[Dict]:
        """Начинает процесс транскрипции файла"""
        try:
            with open(file_path, 'rb') as f:
                response = self.session.post(
                    f"{self.base_url}/start_transcribing",
                    files={'file': f},
                    params={'diarize': 'true', 'remove_timestamps': 'false'},
                    headers=self.headers,
                    timeout=Config.ASR_TIMEOUT
                )
                response.raise_for_status()
                return response.json()
        except Exception as e:
            print(f"❌ Ошибка при запуске транскрипции: {e}")
            return None

    def get_status(self, task_id: str) -> Optional[Dict]:
        """Получает статус задачи транскрипции"""
        try:
            response = self.session.get(
                f"{self.base_url}/get_status",
                params={'task_id': task_id},
                headers=self.headers,
                timeout=60
            )
            response.raise_for_status()
            return response.json()
        except Exception as e:
            print(f"❌ Ошибка при получении статуса: {e}")
            return None

    def get_result(self, task_id: str) -> Optional[str]:
        """Получает результат транскрипции"""
        try:
            response = self.session.get(
                f"{self.base_url}/get_file",
                params={'task_id': task_id},
                headers=self.headers,
                timeout=60
            )
            response.raise_for_status()

            # Пытаемся разобрать JSON ответ
            try:
                result_data = response.json()
                return result_data.get('text', '') or result_data.get('result', '')
            except:
                # Если не JSON, возвращаем текст
                return response.text

        except Exception as e:
            print(f"❌ Ошибка при получении результата: {e}")
            return None

class AudioProcessor:
    """Класс для обработки аудио файлов"""

    @staticmethod
    def validate_audio_file(file_path: str) -> Tuple[bool, str]:
        """Проверяет валидность аудио файла"""
        try:
            audio = AudioSegment.from_file(file_path)
            if len(audio) == 0:
                return False, "Файл пустой"
            duration_min = len(audio) / 1000 / 60
            return True, f"Длительность: {duration_min:.1f} мин"
        except Exception as e:
            return False, f"Ошибка чтения: {e}"

    @staticmethod
    def prepare_audio(file_path: str) -> str:
        """Конвертирует аудио в оптимальный формат для транскрипции"""
        file_ext = os.path.splitext(file_path)[1].lower()

        # Если уже WAV, проверяем параметры
        if file_ext == '.wav':
            try:
                audio = AudioSegment.from_file(file_path)
                if (audio.channels == Config.TARGET_CHANNELS and
                    audio.frame_rate == Config.TARGET_SAMPLE_RATE):
                    return file_path
            except Exception as e:
                print(f"⚠️ Ошибка проверки WAV файла: {e}")

        print("   🔄 Конвертирую в оптимальный формат...")
        try:
            audio = AudioSegment.from_file(file_path)
            audio = audio.set_channels(Config.TARGET_CHANNELS)
            audio = audio.set_frame_rate(Config.TARGET_SAMPLE_RATE)

            temp_wav = tempfile.NamedTemporaryFile(suffix='.wav', delete=False)
            audio.export(temp_wav.name, format="wav")
            print("   ✅ Конвертация завершена")
            return temp_wav.name
        except Exception as e:
            print(f"❌ Ошибка конвертации: {e}")
            return file_path

    @staticmethod
    def split_long_audio(file_path: str, max_duration_ms: int = None) -> Tuple[List[str], str]:
        """Разбивает длинное аудио на чанки"""
        if max_duration_ms is None:
            max_duration_ms = Config.CHUNK_DURATION_MS

        print(f"   📦 Разбиваю на чанки по {max_duration_ms/1000/60} минут...")

        try:
            audio = AudioSegment.from_file(file_path)
            duration_ms = len(audio)

            # Если файл короче максимальной длительности, возвращаем как есть
            if duration_ms <= max_duration_ms:
                return [file_path], ""

            chunks = []
            chunk_dir = tempfile.mkdtemp(prefix="audio_chunks_")
            base_name = os.path.splitext(os.path.basename(file_path))[0]

            total_chunks = (duration_ms + max_duration_ms - 1) // max_duration_ms
            print(f"   📊 Будет создано {total_chunks} чанков...")

            for i, start_time in enumerate(range(0, duration_ms, max_duration_ms)):
                end_time = min(start_time + max_duration_ms, duration_ms)
                chunk = audio[start_time:end_time]

                chunk_path = os.path.join(chunk_dir, f"{base_name}_chunk_{i+1:03d}.wav")
                chunk.export(chunk_path, format="wav")
                chunks.append(chunk_path)

                print(f"   ✅ Чанк {i+1}/{total_chunks}: {start_time/1000/60:.1f}-{end_time/1000/60:.1f} мин")

            print(f"   🎯 Всего создано {len(chunks)} чанков")
            return chunks, chunk_dir

        except Exception as e:
            print(f"❌ Ошибка разбивки аудио: {e}")
            return [file_path], ""

class WebDAVCloudManager:
    """Менеджер для работы с Облаком Mail.ru"""

    def __init__(self):
        self.login = Config.MAILRU_LOGIN
        self.password = Config.MAILRU_PASSWORD
        self.client = self._setup_webdav_client()

    def _setup_webdav_client(self) -> Optional[Client]:
        """Настраивает WebDAV клиент для Облака Mail.ru"""
        options = {
            'webdav_hostname': 'https://webdav.cloud.mail.ru',
            'webdav_login': self.login,
            'webdav_password': self.password,
            'webdav_root': '/',
            'webdav_timeout': 30
        }

        try:
            client = Client(options)
            client.verify = False

            # Проверяем подключение
            if client.check('/'):
                print("✅ Успешное подключение к Облаку Mail.ru")
                return client
            else:
                print("❌ Не удалось подключиться к Облаку Mail.ru")
                return None
        except Exception as e:
            print(f"❌ Ошибка настройки WebDAV клиента: {e}")
            return None

    def ensure_directories(self) -> bool:
        """Создает необходимые директории в облаке"""
        try:
            for directory in Config.CLOUD_DIRECTORIES:
                if not self.client.check(directory):
                    self.client.mkdir(directory)
            return True
        except Exception as e:
            print(f"❌ Ошибка создания директорий: {e}")
            return False

    def upload_file(self, local_path: str, cloud_path: str) -> bool:
        """Загружает файл в облако"""
        try:
            # Создаем родительские директории если нужно
            cloud_dir = os.path.dirname(cloud_path)
            if cloud_dir and not self.client.check(cloud_dir):
                self.client.mkdir(cloud_dir)

            self.client.upload_sync(remote_path=cloud_path, local_path=local_path)
            return True
        except Exception as e:
            print(f"❌ Ошибка загрузки файла {local_path} в {cloud_path}: {e}")
            return False

    def download_file(self, cloud_path: str, local_path: str) -> bool:
        """Скачивает файл из облака"""
        try:
            if not self.client.check(cloud_path):
                return False

            # Создаем директорию для локального файла если нужно
            os.makedirs(os.path.dirname(local_path), exist_ok=True)

            self.client.download_sync(remote_path=cloud_path, local_path=local_path)
            return True
        except Exception as e:
            if "not found" in str(e).lower() or "404" in str(e):
                return False
            print(f"❌ Ошибка скачивания файла {cloud_path}: {e}")
            return False

class AdvancedTranscriber:
    """Основной класс для транскрипции аудио"""

    def __init__(self, cloud_manager: WebDAVCloudManager):
        self.cloud_manager = cloud_manager

        # Инициализация компонентов
        self.api_client = APIClient(Config.DEFAULT_TOKEN)
        self.hash_manager = FileHashManager(cloud_manager)
        self.progress_manager = ProgressManager(cloud_manager)

        # Настройка путей
        self.local_base_dir = "/content/transcription_results"
        self.local_results_dir = os.path.join(self.local_base_dir, "results")
        self.local_upload_dir = os.path.join(self.local_base_dir, "uploaded_files")
        self.cloud_base_dir = "/transcription_results"
        self.cloud_results_dir = os.path.join(self.cloud_base_dir, "results")

        # Создаем локальные директории
        os.makedirs(self.local_results_dir, exist_ok=True)
        os.makedirs(self.local_upload_dir, exist_ok=True)

        print(f"📁 Локальная директория: {self.local_base_dir}")
        print(f"☁️ Облачная директория: {self.cloud_base_dir}")

    def get_remaining_files(self) -> List[str]:
        """Возвращает список файлов, которые еще не обработаны"""
        progress_data = self.progress_manager.load_progress()

        if not os.path.exists(self.local_upload_dir):
            return []

        all_files = [f for f in os.listdir(self.local_upload_dir)
                    if any(f.lower().endswith(ext) for ext in Config.SUPPORTED_FORMATS)]

        remaining = []
        for file in all_files:
            file_progress = progress_data.get(file, {})
            if not file_progress.get('completed', False):
                remaining.append(file)

        return remaining

    def _format_time(self, seconds: float) -> str:
        """Форматирует время в читаемый вид"""
        if seconds < 60:
            return f"{seconds:.1f} сек"
        elif seconds < 3600:
            return f"{seconds/60:.1f} мин"
        else:
            return f"{seconds/3600:.1f} час"

    def transcribe_chunk(self, chunk_path: str, chunk_index: int, total_chunks: int) -> Optional[str]:
        """Транскрибирует один чанк аудио"""
        print(f"   🎯 Чанк {chunk_index}/{total_chunks}: отправляю на обработку...")

        # Запускаем транскрипцию
        task_data = self.api_client.start_transcription(chunk_path)
        if not task_data or 'task_id' not in task_data:
            print(f"   ❌ Чанк {chunk_index}: ошибка запуска транскрипции")
            return None

        task_id = task_data['task_id']
        start_time = time.time()
        last_progress = 0
        status = 'unknown'
        hundred_percent_time = None

        print(f"   ⏳ Чанк {chunk_index}: отслеживаю прогресс...")

        for attempt in range(Config.STATUS_CHECK_ATTEMPTS):
            current_time = time.time()
            elapsed = current_time - start_time

            status_data = self.api_client.get_status(task_id)

            if not status_data:
                # Показываем статус каждые 30 секунд
                if attempt % 3 == 0:  # Каждые 30 секунд
                    print(f"   ⏰ Чанк {chunk_index}: ожидание ответа сервера ({self._format_time(elapsed)})")
                time.sleep(Config.STATUS_CHECK_INTERVAL)
                continue

            progress = status_data.get('progress', 0)
            status = status_data.get('status', 'unknown')

            # Если прогресс 100%, запоминаем время
            if progress == 100 and hundred_percent_time is None:
                hundred_percent_time = current_time
                print(f"   ✅ Чанк {chunk_index}: достигнут 100% прогресс")

            # Показываем прогресс если он изменился
            if progress != last_progress:
                print(f"   📈 Чанк {chunk_index}: {progress}% ({self._format_time(elapsed)})")
                last_progress = progress

            # Если прогресс 100% и прошло время, пробуем получить результат
            if progress == 100 and hundred_percent_time is not None:
                time_since_100 = current_time - hundred_percent_time
                if time_since_100 > 30:  # Ждем 30 секунд после 100%
                    print(f"   🔄 Чанк {chunk_index}: пробую получить результат...")
                    result_text = self.api_client.get_result(task_id)
                    if result_text:
                        total_time = current_time - start_time
                        print(f"   ✅ Чанк {chunk_index}: результат получен за {self._format_time(total_time)}!")
                        return f"\n\n[ЧАНК {chunk_index}/{total_chunks}]\n" + result_text

            # Проверяем завершение
            if status == 'done':
                total_time = time.time() - start_time
                print(f"   ✅ Чанк {chunk_index}: обработка завершена за {self._format_time(total_time)}!")
                break
            elif status == 'error':
                error_msg = status_data.get('error', 'Неизвестная ошибка')
                print(f"   ❌ Чанк {chunk_index}: ошибка - {error_msg}")
                return None

            time.sleep(Config.STATUS_CHECK_INTERVAL)

        # Пробуем получить результат в любом случае
        result_text = self.api_client.get_result(task_id)
        if result_text:
            total_time = time.time() - start_time
            print(f"   ✅ Чанк {chunk_index}: результат получен за {self._format_time(total_time)}!")
            return f"\n\n[ЧАНК {chunk_index}/{total_chunks}]\n" + result_text
        else:
            total_time = time.time() - start_time
            print(f"   ❌ Чанк {chunk_index}: не удалось получить результат за {self._format_time(total_time)}")
            return None

    def check_existing_file(self, filename: str, file_path: str) -> Tuple[bool, Optional[str]]:
        """Проверяет, был ли файл уже обработан"""
        # Вычисляем хеш файла
        file_hash = self.hash_manager.compute_file_hash(file_path)
        if not file_hash:
            return False, None

        # Проверяем по хешу
        is_processed, message = self.hash_manager.is_file_processed(filename, file_hash)

        if is_processed:
            print(f"✅ {message}")

            # Ищем существующий результат
            result_filename = os.path.splitext(filename)[0] + '_transcript.txt'
            local_result_path = os.path.join(self.local_results_dir, result_filename)
            cloud_result_path = os.path.join(self.cloud_results_dir, result_filename)

            # Пробуем скачать из облака
            if self.cloud_manager.download_file(cloud_result_path, local_result_path):
                if os.path.exists(local_result_path):
                    with open(local_result_path, 'r', encoding='utf-8') as f:
                        return True, f.read()

            # Если не нашли в облаке, проверяем локально
            elif os.path.exists(local_result_path):
                with open(local_result_path, 'r', encoding='utf-8') as f:
                    return True, f.read()

        return False, None

    def transcribe_file(self, filename: str) -> Optional[str]:
        """Обрабатывает один файл"""
        file_path = os.path.join(self.local_upload_dir, filename)
        print(f"\n🎧 Начинаю обработку: {filename}")

        # Проверяем, не обработан ли уже файл
        is_processed, existing_text = self.check_existing_file(filename, file_path)
        if is_processed and existing_text:
            return existing_text

        # Проверяем валидность файла
        is_valid, validation_msg = AudioProcessor.validate_audio_file(file_path)
        if not is_valid:
            print(f"❌ Невалидный файл: {validation_msg}")
            return None

        print(f"✅ Проверка файла: {validation_msg}")

        # Вычисляем хеш для нового файла
        file_hash = self.hash_manager.compute_file_hash(file_path)
        if not file_hash:
            print("❌ Не удалось вычислить хеш файла")
            return None

        progress_data = self.progress_manager.load_progress()
        file_key = filename

        # Проверяем прогресс
        if progress_data.get(file_key, {}).get('completed'):
            print("✅ Файл уже обработан ранее (по прогрессу)")
            result_file = progress_data[file_key].get('result_file')
            if result_file and os.path.exists(result_file):
                with open(result_file, 'r', encoding='utf-8') as f:
                    # Помечаем файл как обработанный в системе хешей
                    self.hash_manager.mark_file_processed(filename, file_hash)
                    return f.read()

        # Начинаем обработку
        start_time = time.time()
        progress_data[file_key] = {
            'start_time': time.strftime('%Y-%m-%d %H:%M:%S'),
            'completed': False,
            'chunks_processed': 0,
            'total_chunks': 0,
            'file_hash': file_hash
        }
        self.progress_manager.save_progress(progress_data)

        prepared_audio = file_path
        chunk_dir = ""

        try:
            # Подготавливаем аудио
            prepared_audio = AudioProcessor.prepare_audio(file_path)

            # Разбиваем на чанки если нужно
            chunks, chunk_dir = AudioProcessor.split_long_audio(prepared_audio)
            total_chunks = len(chunks)

            progress_data[file_key]['total_chunks'] = total_chunks
            self.progress_manager.save_progress(progress_data)

            if total_chunks > 1:
                print(f"📊 Файл разбит на {total_chunks} чанков")

            # Создаем заголовок результата
            full_text = f"ФАЙЛ: {filename}\n"
            full_text += f"ВРЕМЯ: {time.strftime('%Y-%m-%d %H:%M:%S')}\n"
            full_text += f"ЧАНКОВ: {total_chunks}\n"
            full_text += "=" * 50 + "\n\n"

            chunk_results = []
            completed_chunks = 0

            # Обрабатываем каждый чанк
            for i, chunk_path in enumerate(chunks, 1):
                chunk_key = f"{file_key}_chunk_{i}"

                # Проверяем, не обработан ли уже чанк
                if chunk_key in progress_data:
                    print(f"   ✅ Чанк {i} уже обработан")
                    chunk_text = progress_data[chunk_key].get('text', '')
                    chunk_results.append(chunk_text)
                    completed_chunks += 1
                    continue

                # Транскрибируем чанк
                chunk_text = self.transcribe_chunk(chunk_path, i, total_chunks)

                if chunk_text:
                    chunk_results.append(chunk_text)
                    completed_chunks += 1

                    # Сохраняем прогресс чанка
                    progress_data[chunk_key] = {
                        'text': chunk_text,
                        'timestamp': time.time(),
                        'chunk_number': i
                    }
                    progress_data[file_key]['chunks_processed'] = completed_chunks
                    self.progress_manager.save_progress(progress_data)

                    print(f"   ✅ Чанк {i} завершен ({completed_chunks}/{total_chunks})")
                else:
                    print(f"   ❌ Ошибка в чанке {i}")
                    chunk_results.append(f"\n\n[ЧАНК {i} - ОШИБКА ОБРАБОТКИ]\n")

            # Собираем полный текст
            full_text += "".join(chunk_results)

            # Сохраняем результат
            result_filename = os.path.splitext(filename)[0] + '_transcript.txt'
            result_filepath = os.path.join(self.local_results_dir, result_filename)

            with open(result_filepath, 'w', encoding='utf-8') as f:
                f.write(full_text)

            # Загружаем в облако
            cloud_result_path = os.path.join(self.cloud_results_dir, result_filename)
            if self.cloud_manager.upload_file(result_filepath, cloud_result_path):
                print(f"☁️ Результат загружен в облако")

            # Помечаем файл как обработанный
            if self.hash_manager.mark_file_processed(filename, file_hash):
                print("✅ Файл помечен как обработанный в системе хешей")

            # Обновляем прогресс файла
            processing_time = time.time() - start_time
            progress_data[file_key].update({
                'completed': True,
                'result_file': result_filepath,
                'cloud_file': cloud_result_path,
                'chunks_processed': completed_chunks,
                'completion_time': time.strftime('%Y-%m-%d %H:%M:%S'),
                'processing_time_seconds': int(processing_time)
            })
            self.progress_manager.save_progress(progress_data)

            print(f"✅ Обработка завершена: {completed_chunks}/{total_chunks} чанков за {self._format_time(processing_time)}")
            return full_text

        except Exception as e:
            print(f"❌ Общая ошибка обработки: {e}")
            return None
        finally:
            # Очистка временных файлов
            if prepared_audio != file_path and os.path.exists(prepared_audio):
                try:
                    os.unlink(prepared_audio)
                except:
                    pass

            if chunk_dir and os.path.exists(chunk_dir):
                try:
                    shutil.rmtree(chunk_dir)
                except:
                    pass

    def create_results_archive(self, results: Dict[str, str]) -> Optional[str]:
        """Создает архив с результатами транскрипции"""
        if not results:
            return None

        print("\n💾 Создаю архив с результатами...")

        current_time = datetime.now().strftime("%Y-%m-%d_%H-%M-%S")

        if len(results) == 1:
            filename = list(results.keys())[0]
            base_name = os.path.splitext(filename)[0]
            zip_filename = f"{base_name}_{current_time}.zip"
        else:
            zip_filename = f"transcriptions_{current_time}.zip"

        zip_path = os.path.join(self.local_base_dir, zip_filename)

        try:
            with zipfile.ZipFile(zip_path, 'w') as zipf:
                for filename, text in results.items():
                    name = os.path.splitext(filename)[0] + '_transcript.txt'
                    zipf.writestr(name, text.encode('utf-8'))

            print(f"✅ Архив создан: {zip_filename}")

            # Загружаем в облако
            cloud_zip_path = os.path.join(self.cloud_results_dir, zip_filename)
            if self.cloud_manager.upload_file(zip_path, cloud_zip_path):
                print(f"☁️ Архив загружен в облако")

            return zip_path

        except Exception as e:
            print(f"❌ Ошибка создания архива: {e}")
            return None

    def get_statistics(self) -> Dict[str, Any]:
        """Возвращает статистику обработки"""
        progress_data = self.progress_manager.load_progress()
        hashes_data = self.hash_manager.load_hashes()

        total_files = len([f for f in progress_data.values()
                          if isinstance(f, dict) and f.get('completed')])

        total_chunks = sum([f.get('total_chunks', 0) for f in progress_data.values()
                           if isinstance(f, dict) and f.get('completed')])

        unique_processed_files = len(hashes_data)

        return {
            'total_processed_files': total_files,
            'total_processed_chunks': total_chunks,
            'unique_processed_files': unique_processed_files,
            'local_directory': self.local_base_dir,
            'cloud_directory': self.cloud_base_dir
        }

## AI

Данный код представляет собой систему для автоматической транскрипции аудиофайлов с использованием API `bit-asr.1bitai.ru`, интегрированную в среду Google Colab и облачное хранилище Mail.ru. Он загружает аудио, разбивает длинные записи на часовые фрагменты, отправляет на распознавание речи, отслеживает прогресс, собирает текст и сохраняет результаты локально и в облаке, используя WebDAV. Реализованы проверка хешей для избежания повторной обработки, управление состоянием через `progress.json` и создание архива с результатами.

### **Что делает код (основные компоненты):**
- **Загрузка и валидация аудио:** через `google.colab.files.upload()`, проверка формата, длительности.
- **Предобработка:** конвертация в WAV 16 кГц моно, разбиение на чанки по 1 часу.
- **Взаимодействие с API транскрипции:** отправка чанков, опрос статуса, получение текста.
- **Облачное хранилище:** WebDAV-клиент для Mail.ru — создание папок, загрузка/скачивание файлов.
- **Управление прогрессом и хешами:** чтобы при перезапуске Colab не обрабатывать одни и те же файлы повторно.
- **Сборка результатов:** объединение текста чанков, сохранение в `.txt`, архивирование.

## **Критика кода (основные недостатки)**

### **1. Критическая проблема безопасности**
Хранение чувствительных данных **в открытом виде в исходном коде**:
```python
DEFAULT_TOKEN = "xYExUvtcJMUrDYQKFewDXKjODZGXhIN40CHtxfKGsgw"
MAILRU_LOGIN = "swzhukov@mail.ru"
MAILRU_PASSWORD = "rx4aJVqLz86XfG0mGEUU"
```
Это недопустимо, особенно при возможной публикации кода (Colab-ноутбуки часто расшариваются). Токены и пароли должны загружаться из переменных окружения, `google.colab.userdata` или зашифрованных хранилищ.

### **2. Низкая устойчивость к ошибкам и гонкам**
- **Работа с облаком без блокировок:** `file_hashes.json` и `progress.json` синхронизируются без какой-либо координации. При параллельном запуске двух сессий возможна потеря данных или неконсистентное состояние.
- **WebDAV-клиент настроен с `verify=False`** – отключает проверку SSL-сертификата, что снижает безопасность.
- **Метод `WebDAVCloudManager.upload_file`** всегда возвращает `True` при успешном вызове `upload_sync`, но не проверяет реальный результат (если не возникло исключения). Аналогично для `download_file` – проверка только на `check()`, что не гарантирует целостность загрузки.
- **Обработка `get_result`** пытается разобрать ответ как JSON, иначе берёт `response.text`. Но API может вернуть файл (бинарные данные) – код не готов к такому.

### **3. Проблемы с управлением ресурсами**
- **Временные файлы:** `tempfile.NamedTemporaryFile(delete=False)` – файл не удаляется автоматически. Удаление в `finally` есть, но если упадёт до `finally` (например, `KeyboardInterrupt`), останутся мусорные файлы в `/tmp`.
- **Чанки создаются во временной директории `tempfile.mkdtemp`**, очистка в `finally`, но при аварийном завершении могут накапливаться.
- **Файл лога** пишется в `/content/transcription_log.txt`, но не очищается и не синхронизируется с облаком – при перезапуске теряется история.

### **4. Избыточная сложность и запутанная логика**
- **Метод `transcribe_chunk`** содержит хитрую логику с `hundred_percent_time` и ожиданием 30 секунд после достижения 100%. Это попытка обойти особенности API, но выглядит хрупко. Если статус `done` приходит раньше или позже, возможны задержки или пропуск результата.
- **Цикл опроса** делает до 60 попыток с интервалом 10 секунд (10 минут), что может быть избыточно для коротких чанков и недостаточно для длинных.
- **`get_remaining_files`** работает на основе `progress.json`, но не учитывает хеш-менеджер – дублирование логики проверки обработанности.

### **5. Нарушение принципов проектирования**
- **Гигантский класс `AdvancedTranscriber`** отвечает за всё: облако, API, хеши, прогресс. Жёсткая связанность усложняет тестирование и модификацию.
- **Статические методы в `AudioProcessor`** – это нормально, но они используют глобальные константы из `Config`, что не гибко.
- **Прямая зависимость от Colab:** вызов `google.colab.files.upload()` и `files.download()` делает код непригодным вне Colab без модификации.

### **6. Проблемы с удобством использования**
- При запуске `main()` сначала проверяет оставшиеся файлы из предыдущих сессий, но **не показывает их статус** детально (только имена). Пользователь может случайно выбрать `y` и повторно обработать уже готовое.
- Нет возможности пропустить конкретный чанк или файл при ошибке – весь файл падает.
- Статистика в конце показывает общие цифры по всем сессиям, но не разделяет текущую и предыдущие, что может сбивать с толку.

### **7. Потенциальные утечки памяти и производительность**
- При вычислении хеша больших файлов читается весь файл по 4KB, что для multi-GB записей может быть медленно и нагружать диск (особенно в Colab).
- Все чанки загружаются в оперативную память через `AudioSegment`, а затем экспортируются на диск. При большом количестве или размере файлов это может привести к переполнению RAM.

### **8. Неполная обработка краевых случаев**
- Если API возвращает пустой результат или ошибку на одном чанке, в итоговый текст вставляется строка `[ЧАНК X - ОШИБКА ОБРАБОТКИ]`, но сама транскрипция считается выполненной (файл помечается как обработанный). Это может привести к тому, что неполный результат будет считаться завершённым.
- Нет повторных попыток при сбое получения статуса или результата, кроме retry на уровне HTTP-сессии (только для определённых статусов).

# Техническое задание: Система пакетной транскрипции аудио с облачным сохранением прогресса

## 1. Контекст и цель продукта
У пользователя есть множество аудиофайлов (лекции, интервью, созвоны), которые необходимо превратить в текст с помощью стороннего API распознавания речи. Файлы могут быть большого размера (несколько часов), а сеанс обработки может прерваться из-за нестабильного интернета или перезагрузки среды выполнения (Google Colab).  
**Цель** — создать инструмент, который:
- Автоматически загружает аудио, подготавливает его, отправляет на транскрипцию и собирает результат.
- Устойчив к перезапускам: прогресс и результаты сохраняются в облачное хранилище (Mail.ru WebDAV), при повторном запуске необработанные файлы докачиваются, а дубликаты пропускаются.
- Не требует ручного отслеживания состояния и минимизирует повторную работу.

## 2. Ключевая ценность
- **Экономия времени:** пакетная обработка десятков файлов с автоматическим возобновлением.
- **Надёжность:** даже после сбоя не теряются уже распознанные фрагменты.
- **Контроль дубликатов:** файлы проверяются по MD5‑хешу, исключая повторную транскрипцию одного и того же контента.

## 3. Основные функциональные требования

### 3.1. Загрузка и валидация аудио
- Пользователь может загрузить один или несколько файлов через стандартный диалог (в среде Colab) или указать путь к папке с аудио (для локального/серверного режима).
- Поддерживаемые форматы: MP3, WAV, M4A, FLAC, OGG, AAC, WMA.
- Каждый файл проверяется на читаемость, ненулевую длительность и формат.
- Уже обработанные файлы (по хешу) помечаются как пропущенные с сообщением пользователю.

### 3.2. Предобработка аудио
- Все файлы приводятся к оптимальному формату для API: моно, 16 кГц, WAV.
- Если исходный файл уже соответствует требованиям, перекодирование не выполняется.
- Аудио длительностью более 1 часа разбивается на чанки по 1 часу с сохранением порядка.

### 3.3. Взаимодействие с API транскрипции
- Используется заданный внешний сервис (ASR_URL), аутентификация по статическому токену.
- Для каждого чанка:
  - Отправка файла на эндпоинт `/start_transcribing` с параметрами диаризации (если нужно).
  - Периодический опрос статуса `/get_status` (интервал и количество попыток настраиваются).
  - При достижении статуса «done» или прогресса 100% — получение текста через `/get_file`.
- Результат накапливается, собирается в единый текст с заголовками чанков.

### 3.4. Управление прогрессом и сохранение состояния
- В облачном хранилище хранится файл `progress.json`, в котором зафиксированы:
  - Какие файлы начаты, сколько чанков обработано.
  - Для каждого чанка — статус (завершён / не начат) и полученный текст.
- При запуске система загружает `progress.json` и определяет список файлов, требующих доработки.
- Пользователь может выбрать: продолжить незавершённые файлы или начать обработку новой партии.

### 3.5. Контроль дубликатов файлов
- Ведётся отдельная база хешей (`file_hashes.json`) в облаке.
- Перед началом обработки для каждого файла вычисляется MD5-хеш. Если хеш совпадает с ранее обработанным файлом (даже под другим именем), транскрипция пропускается, а готовый текст загружается из облака.
- Хеш‑база синхронизируется с облаком при каждом обновлении.

### 3.6. Сохранение и выдача результатов
- Итоговый текст для файла сохраняется в локальную папку `results/` и параллельно загружается в облако.
- По окончании всей сессии формируется ZIP‑архив со всеми текстами.
- Пользователю предлагается скачать архив.

### 3.7. Логирование
- Ход выполнения пишется в лог‑файл и выводится в консоль.
- Для длительных операций отображается прогресс‑бар (tqdm) или периодические сообщения с процентами.

## 4. Нефункциональные требования

### 4.1. Надёжность и отказоустойчивость
- Все сетевые операции (API, WebDAV) должны содержать повторные попытки с экспоненциальной задержкой при ошибках 5xx, 429 и таймаутах.
- После любого аварийного завершения временные файлы и чанки должны очищаться (механизм `finally` или контекстные менеджеры).
- При отсутствии связи с облаком система не должна падать — ошибка логируется, но локальный прогресс сохраняется для последующей синхронизации (если возможно).

### 4.2. Безопасность
- Токен API и пароль облака **не хранятся в коде**. Они загружаются из переменных окружения или из безопасного хранилища учётных данных Colab (`userdata`).
- Все соединения с облаком идут по HTTPS, но без отключения проверки сертификатов (в production).

### 4.3. Кроссплатформенность и переносимость
- Код должен работать как в Google Colab, так и на локальной машине/сервере. Различия (загрузка файлов через `colab.files`) инкапсулируются за абстрактным интерфейсом «источник аудио».
- Зависимости фиксируются в `requirements.txt`.

### 4.4. Производительность
- Вычисление MD5 файла не должно загружать весь файл в память (читать чанками).
- При конвертации аудио используются временные файлы, а не хранение в оперативной памяти больших объектов.

## 5. Архитектура (рекомендуемая)

Система строится по модульному принципу. Основные компоненты:

1. **Config** — централизованные настройки (пути, тайминги, параметры API).
2. **AudioProcessor** — валидация, конвертация, разбиение на чанки.
3. **APIClient** — инкапсуляция HTTP-запросов к сервису распознавания, обработка ответов, ретраи.
4. **CloudStorage** — интерфейс для операций с WebDAV (upload/download, проверка существования, создание папок).
5. **ProgressManager** — работа с `progress.json`: загрузка, сохранение, обновление прогресса.
6. **HashManager** — ведение словаря «имя/путь: хеш», сохранение в облако, поиск дубликатов.
7. **TranscriptionOrchestrator** — главный оркестратор, который принимает список файлов и координирует их обработку, используя вышеперечисленные сервисы.
8. **CLI / Colab UI** — точка входа, запрос файлов, вывод результатов.

## 6. Ключевые сценарии работы

### Сценарий 1: Первичный запуск с новыми файлами
1. Пользователь загружает 3 аудиофайла.
2. Система аутентифицируется в облаке, синхронизирует `progress.json` и `file_hashes.json`.
3. Для каждого файла вычисляется хеш, проверяется по базе. Все три — новые.
4. Файлы конвертируются, разбиваются на чанки при необходимости.
5. Чанки отправляются в API; по мере готовности текст сохраняется в прогресс и облачную папку `results/`.
6. После завершения всех файлов создаётся архив, предлагается скачать.

### Сценарий 2: Перезапуск после обрыва
1. Пользователь перезапускает среду выполнения.
2. Система обнаруживает в `progress.json` незавершённые файлы.
3. Пользователю предлагается продолжить обработку. При согласии обрабатываются только недостающие чанки.
4. Ранее полученные тексты для завершённых файлов не запрашиваются повторно.
5. После доработки формируется финальный архив.

### Сценарий 3: Дубликат файла
1. Пользователь загружает файл, идентичный ранее обработанному (тот же MD5).
2. Система находит его в хеш‑базе, скачивает готовый результат из облака, уведомляет: «Файл уже обработан, результат восстановлен».
3. Транскрипция не запускается.
