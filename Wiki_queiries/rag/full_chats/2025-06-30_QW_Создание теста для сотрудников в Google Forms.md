function importQuestionsToForm() {
  const form = FormApp.create('Тест по материалам встречи');
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('Questions');
  const data = sheet.getDataRange().getValues();
  
  // Пропускаем заголовок
  data.shift(); 

  data.forEach(row => {
    const [question, type, opt1, opt2, opt3, opt4, correct] = row;
    
    const item = type === 'RADIO' 
      ? form.addMultipleChoiceItem()
      : form.addCheckboxItem();
      
    item.setTitle(question)
      .setChoices([
        item.createChoice(opt1),
        item.createChoice(opt2),
        item.createChoice(opt3),
        opt4 ? item.createChoice(opt4) : null
      ].filter(Boolean))
      .setRequired(true);
    
    if (correct) {
      item.setFeedback(FormApp.FeedbackType.CORRECT, "Верно!")
           .setFeedback(FormApp.FeedbackType.INCORRECT, "Неверно")
           .setPoints(1);
      
      const correctChoices = correct.split(',');
      const choices = item.getChoices();
      const correctAnswers = choices.filter((_, i) => 
        correctChoices.includes(String.fromCharCode(65 + i))
      );
      
      item.setCorrectAnswers(correctAnswers);
    }
  });
  
  // Настройки формы
  form.setDescription('Тест на знание процессов УКП 5.1')
      .setCollectEmailAddress(true)
      .setLimitOneResponsePerUser(true)
      .setIsQuiz(true);
  
  Logger.log('Форма создана: ' + form.getPublishedUrl());
}

function setupResponseCollection(ss, form) {

  const masterSheet = ss.getSheetByName(CONFIG.MASTER_SHEET) || ss.insertSheet(CONFIG.MASTER_SHEET);
  
  // Настраиваем форму для сохранения ответов в текущую таблицу
  form.setDestination(FormApp.DestinationType.SPREADSHEET, ss.getId());
  
  // Переносим ответы из временного листа в общий
  const formResponseSheet = ss.getSheetByName('Ответы на форму');
  if (formResponseSheet) {
    formResponseSheet.getDataRange().copyTo(
      masterSheet.getRange(masterSheet.getLastRow() + 1, 1),
      {contentsOnly: true}
    );
    ss.deleteSheet(formResponseSheet);
  }
  
  // Форматирование заголовков
  if (masterSheet.getLastRow() === 0) {
    masterSheet.appendRow([
      "Метка времени", "ФИО и подразделение", 
      "Email", "Результат", "Баллы", "Процент"
    ]);

         .setFontWeight("bold")

    const timestamp = new Date();
    const email = e.response.getRespondentEmail() || "Неизвестно";
    const score = e.response.getScore() || 0;
    const total = e.source.getPoints();
    const percent = total > 0 ? Math.round((score / total) * 100) : 0;
    
    // Получаем ответ на первый вопрос (ФИО)
    const nameResponse = e.response.getItemResponses()[0].getResponse();
    
    // Добавляем результат

      timestamp,
      nameResponse,
      email,
      score >= CONFIG.PASSING_SCORE ? "Сдал" : "Не сдал",
      `${score}/${total}`,
      `${percent}%`
    ]);
    
    // Автоматическое форматирование

    ]);
    
    // Форматирование

### Как настроить единую таблицу результатов:

1. **Установите триггер** (один раз):
```javascript
function setupTrigger() {
  ScriptApp.newTrigger('onFormSubmit')
    .forSpreadsheet(SpreadsheetApp.getActiveSpreadsheet())
    .onFormSubmit()
    .create();
}
```
- Запустите эту функцию один раз через редактор скриптов
- Дайте необходимые разрешения

2. **Особенности работы системы**:
- Все ответы автоматически сохраняются в лист "**Все_результаты**"
- Данные сохраняются в формате:
  ```
  | Метка времени | ФИО и подразделение | Email | Результат | Баллы | Процент |
  ```
- Система автоматически определяет сдал/не сдал тест
- При создании нового теста не нужно делать дополнительных настроек

### Преимущества решения:
1. **Автоматический сбор** всех результатов в одну таблицу
2. **Единое хранилище** для всех тестов и сотрудников
3. **Автоматическая маркировка** результатов (Сдал/Не сдал)
4. **Процент выполнения** для каждого сотрудника
5. **История прохождений** с датой и временем
6. **Интеграция с email** (если пользователи входят под своими аккаунтами)

### Как использовать:
