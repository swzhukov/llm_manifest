import tkinter as tk
from tkinter import messagebox
import requests
import re
from bs4 import BeautifulSoup
from urllib.parse import urlparse

def download_video():
    url = entry.get()
    try:
        # Проверка валидности ссылки VK
        if "vk.com" not in url and "vkontakte.ru" not in url:
            raise ValueError("Некорректная ссылка VK")

        # Получаем HTML-страницу видео
        headers = {
            'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/58.0.3029.110 Safari/537.3'
        }
        response = requests.get(url, headers=headers)
        response.raise_for_status()

        # Ищем JSON-данные с ссылками на видео
        soup = BeautifulSoup(response.text, 'html.parser')
        scripts = soup.find_all('script')
        video_url = None

        # Поиск URL видео в JavaScript-переменных
        for script in scripts:
            if 'var vars' in script.text:
                match = re.search(r'"url\d+":"(https:\\/\\/[^"]+)"', script.text)
                if match:
                    video_url = match.group(1).replace('\\/', '/')
                    break

        if not video_url:
            raise ValueError("Видео не найдено или доступ ограничен")

        # Скачивание видео
        video_response = requests.get(video_url, stream=True, headers=headers)
        video_response.raise_for_status()

        # Сохранение видео
        filename = urlparse(url).path.split('/')[-1] + ".mp4"
        with open(filename, 'wb') as f:
            for chunk in video_response.iter_content(chunk_size=8192):
                f.write(chunk)

        messagebox.showinfo("Успех", f"Видео сохранено как {filename}")
    except Exception as e:
        messagebox.showerror("Ошибка", str(e))

# Создание GUI
root = tk.Tk()
root.title("Скачатор VK видео")

label = tk.Label(root, text="Введите ссылку на видео VK:")
label.pack(pady=10)

entry = tk.Entry(root, width=50)
entry.pack(pady=5)

button = tk.Button(root, text="Скачать", command=download_video)
button.pack(pady=10)

root.mainloop()

        clear_btn.grid(row=5, column=1, pady=10, sticky=tk.E)

        # Настройка адаптивности
        main_frame.columnconfigure(0, weight=1)
        main_frame.columnconfigure(1, weight=1)

    def start_download(self):
        if not self.downloading:
            url = self.url_entry.get()
            if not url:
                messagebox.showwarning("Ошибка", "Введите ссылку на видео")
                return
            
            threading.Thread(target=self.process_download, daemon=True).start()

    def process_download(self):
        try:
            self.downloading = True
            self.update_ui_state(False)
            
            url = self.url_entry.get()
            video_data = self.get_video_data(url)
            
            if not video_data:
                raise ValueError("Не удалось получить данные видео")
            
            self.root.after(0, self.update_quality_combo, video_data)
            selected_quality = self.quality_combo.get()
            
            video_url = video_data.get(selected_quality)
            if not video_url:
                raise ValueError("Выберите качество видео")
            
            self.download_file(video_url)
            
        except Exception as e:
            self.root.after(0, messagebox.showerror, "Ошибка", str(e))
        finally:
            self.downloading = False
            self.root.after(0, self.update_ui_state, True)

    def get_video_data(self, url):
        if not re.match(r'https?://(vk\.com|vkontakte\.ru)/', url):
            raise ValueError("Некорректная ссылка VK")

        headers = {
            'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/91.0.4472.124 Safari/537.36'
        }
        
        response = requests.get(url, headers=headers)
        response.raise_for_status()
        
        soup = BeautifulSoup(response.text, 'html.parser')
        script = soup.find('script', string=re.compile('var vars'))
        
        if not script:
            raise ValueError("Видео не найдено или доступ ограничен")
        
        video_data = {}
        matches = re.findall(r'"url(\d+)":"(https?:\\/\\/[^"]+)"', script.string)
        for quality, url in matches:
            video_data[f"{quality}p"] = url.replace('\\/', '/')
        
        return video_data

    def download_file(self, url):
        try:
            with requests.get(url, stream=True) as r:
                r.raise_for_status()
                total_size = int(r.headers.get('content-length', 0))
                downloaded = 0
                
                filename = os.path.basename(urlparse(url).path)
                self.root.after(0, self.update_status, f"Скачивание: {filename}")
                
                with open(filename, 'wb') as f:
                    for chunk in r.iter_content(chunk_size=8192):
                        if chunk:
                            f.write(chunk)
                            downloaded += len(chunk)
                            progress = int((downloaded / total_size) * 100) if total_size else 0
                            self.root.after(0, self.update_progress, progress)
            
            self.root.after(0, self.update_status, "Скачивание завершено!")
            self.root.after(0, messagebox.showinfo, "Успех", f"Файл сохранен как: {filename}")
            
        except requests.exceptions.RequestException as e:
            raise ValueError(f"Ошибка скачивания: {str(e)}")

    def update_ui_state(self, enabled):
        state = 'normal' if enabled else 'disabled'
        self.url_entry.config(state=state)
        self.quality_combo.config(state=state)
        self.progress['value'] = 0

    def update_progress(self, value):
        self.progress['value'] = value
        self.root.update_idletasks()

    def update_status(self, text):
        self.status_label.config(text=text)

    def update_quality_combo(self, video_data):
        self.quality_combo['values'] = list(video_data.keys())
        if video_data:
            self.quality_combo.current(0)

    def clear_input(self):
        self.url_entry.delete(0, tk.END)
        self.quality_combo.set('')
        self.progress['value'] = 0
        self.status_label.config(text="")

if __name__ == "__main__":
    root = tk.Tk()
    app = VKVideoDownloader(root)
    root.mainloop()
