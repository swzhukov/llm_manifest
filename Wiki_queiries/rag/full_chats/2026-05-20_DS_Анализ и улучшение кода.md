def download_from_yandex(public_url: str, dest_dir: Path) -> Path:
    try:
        api_url = "https://cloud-api.yandex.net/v1/disk/public/resources/download"
        resp = requests.get(api_url, params={"public_key": public_url})
        resp.raise_for_status()
        data = resp.json()
        download_url = data["href"]
        file_name = data.get("name") or Path(unquote(urlparse(download_url).path)).name
        dest_path = dest_dir / file_name
        print(f"⏳ Скачиваем {file_name} с Яндекс.Диска...")
        with requests.get(download_url, stream=True) as r:
            r.raise_for_status()
            with open(dest_path, "wb") as f:
                for chunk in r.iter_content(chunk_size=8192):
                    f.write(chunk)
        print(f"✅ Файл скачан: {dest_path.name}")
        return dest_path
    except Exception as e:
        raise RuntimeError(f"Ошибка загрузки из Яндекс.Диска: {e}")

def cache_key_for(file_path: Path) -> str:
    stat = file_path.stat()
    raw = f"{file_path.resolve()}:{stat.st_mtime}:{stat.st_size}"
    return hashlib.md5(raw.encode()).hexdigest()

def load_from_cache(file_path: Path) -> Optional[str]:
    key = cache_key_for(file_path)
    cache_file = CACHE_DIR / key
    with cache_lock:
        if cache_file.exists():
            return cache_file.read_text(encoding="utf-8")
    return None

def save_to_cache(file_path: Path, result: str):
    key = cache_key_for(file_path)
    cache_file = CACHE_DIR / key
    with cache_lock:
        cache_file.write_text(result, encoding="utf-8")

def _media_worker_task(file_path_str: str, media_type: str, output_file: str):
    """
    Открывает файл, обрабатывает его и записывает результат в output_file.
    При ошибке записывает "ERROR".
    """
    try:
        file_path = Path(file_path_str)
        ext = file_path.suffix.lower()
        audio_video_ext = {".mp3", ".wav", ".ogg", ".m4a", ".flac", ".opus",
                           ".mp4", ".webm", ".mkv", ".avi", ".mov"}
        image_ext = {".jpg", ".jpeg", ".png", ".webp", ".bmp", ".gif", ".tiff", ".tif"}

        if (media_type in ("audio", "voice_message", "video", "animation",
                           "video_file", "audio_file", "voice", "round_video")
                or ext in audio_video_ext):
            if not _ensure_audio_stream(file_path):
                Path(output_file).write_text("[нет аудиодорожки]", encoding="utf-8")
                return
            wav_path = None
            txt_path = None
            try:
                with tempfile.NamedTemporaryFile(suffix=".wav", delete=False) as tmp:
                    wav_path = Path(tmp.name)
                if not _convert_to_wav(file_path, wav_path):
                    Path(output_file).write_text("ERROR", encoding="utf-8")
                    return
                engine = "diarize" if (ENABLE_DIARIZATION and NEWTON_ENGINE in ("v3", "diarize")) else NEWTON_ENGINE
                with tempfile.NamedTemporaryFile(suffix=".txt", delete=False) as tmp:
                    txt_path = Path(tmp.name)
                cmd = ["newton", "transcribe", str(wav_path), "-o", str(txt_path), "-e", engine]
                if engine == "diarize" and DIARIZE_SPEAKERS is not None:
                    cmd.extend(["-n", str(DIARIZE_SPEAKERS)])
                subprocess.run(cmd, check=True, capture_output=True, text=True, timeout=600)
                result = txt_path.read_text(encoding="utf-8").strip() if txt_path.exists() else "ERROR"
                Path(output_file).write_text(result, encoding="utf-8")
            except Exception:
                Path(output_file).write_text("ERROR", encoding="utf-8")
            finally:
                if wav_path and wav_path.exists():
                    wav_path.unlink(missing_ok=True)
                if txt_path and txt_path.exists():
                    txt_path.unlink(missing_ok=True)

        elif media_type in ("photo", "sticker") or ext in image_ext:
            try:
                img = Image.open(file_path)
                text = pytesseract.image_to_string(img, lang=OCR_LANG).strip()
                Path(output_file).write_text(text if text else "ERROR", encoding="utf-8")
            except Exception:
                Path(output_file).write_text("ERROR", encoding="utf-8")

        else:
            # Документы
            try:
                if ext in {'.pptx', '.ppt'}:
                    text = _extract_text_from_pptx(file_path)
                elif ext in {'.docx', '.doc'}:
                    text = _extract_text_from_docx(file_path)
                elif ext == '.pdf':
                    text = _extract_text_from_pdf(file_path)
                elif ext in {'.xlsx', '.xls'}:
                    text = _extract_text_from_xlsx(file_path)
                else:
                    text = _extract_text_from_txt(file_path)
                Path(output_file).write_text(text if text else "ERROR", encoding="utf-8")
            except Exception:
                Path(output_file).write_text("ERROR", encoding="utf-8")
    except MemoryError:
        Path(output_file).write_text("ERROR", encoding="utf-8")
    except Exception:
        Path(output_file).write_text("ERROR", encoding="utf-8")

def _ensure_audio_stream(file_path: Path) -> bool:
    try:
        res = subprocess.run(["ffprobe", "-v", "error", "-select_streams", "a:0",
                              "-show_entries", "stream=codec_type", "-of", "csv=p=0", str(file_path)],
                             capture_output=True, text=True, timeout=15)
        return "audio" in res.stdout.strip()
    except: return False

def _convert_to_wav(input_path: Path, output_path: Path) -> bool:
    try:
        subprocess.run(["ffmpeg", "-y", "-i", str(input_path), "-ac", "1", "-ar", "16000",
                        "-sample_fmt", "s16", str(output_path)],
                       check=True, capture_output=True, timeout=120)
        return output_path.exists()
    except: return False

def _extract_text_from_pptx(pptx_path: Path) -> Optional[str]:
    try:
        prs = Presentation(str(pptx_path))
        parts = []
        for i, slide in enumerate(prs.slides, 1):
            slide_text = []
            def extract_shape(shape):
                if shape.has_text_frame:
                    for para in shape.text_frame.paragraphs:
                        if para.text.strip(): slide_text.append(para.text.strip())
                if shape.has_table:
                    rows = ["| " + " | ".join(cell.text for cell in row.cells) + " |" for row in shape.table.rows]
                    if rows: slide_text.append("\n".join(rows))
                if shape.shape_type == MSO_SHAPE_TYPE.GROUP:
                    for child in shape.shapes: extract_shape(child)
            for shape in slide.shapes: extract_shape(shape)
            if slide.has_notes_slide and slide.notes_slide.notes_text_frame:
                notes = slide.notes_slide.notes_text_frame.text.strip()
                if notes: slide_text.append(f"[Заметки]: {notes}")
            if slide_text: parts.append(f"### Слайд {i}\n" + "\n".join(slide_text))
        return "\n\n".join(parts) if parts else None
    except:
        return None

def _extract_text_from_docx(docx_path: Path) -> Optional[str]:
    try:
        doc = Document(str(docx_path))
        parts = [p.text for p in doc.paragraphs if p.text.strip()]
        for table in doc.tables:
            rows = ["| " + " | ".join(cell.text for cell in row.cells) + " |" for row in table.rows]
            if rows: parts.append("\n".join(rows))
        return "\n\n".join(parts) if parts else None
    except:
        return None

def _extract_text_from_pdf(pdf_path: Path) -> Optional[str]:
    try:
        reader = PdfReader(str(pdf_path))
        text = "\n".join(p.extract_text() or "" for p in reader.pages).strip()
        if text: return text
    except: pass
    try:
        with pdfplumber.open(str(pdf_path)) as pdf:
            text = "\n".join(p.extract_text() or "" for p in pdf.pages).strip()
            if text: return text
    except: pass
    if ENABLE_OCR:
        with tempfile.TemporaryDirectory() as tmpdir:
            try:
                subprocess.run(["pdftoppm", "-png", "-r", "200", str(pdf_path), f"{tmpdir}/page"],

                    img = Image.open(img_file)

def _extract_text_from_xlsx(xlsx_path: Path) -> Optional[str]:
    try:
        wb = load_workbook(str(xlsx_path), read_only=True, data_only=True)
        parts = []
        for name in wb.sheetnames:
            ws = wb[name]
            rows = list(ws.iter_rows(max_row=100, values_only=True))
            if not rows: continue
            header = rows[0]
            parts.append(f"### Лист: {name}")
            parts.append("| " + " | ".join(str(c) if c is not None else "" for c in header) + " |")
            parts.append("| " + " | ".join(["---"] * len(header)) + " |")
            for row in rows[1:]:
                parts.append("| " + " | ".join(str(c) if c is not None else "" for c in row) + " |")
            parts.append("")
        return "\n".join(parts) if parts else None
    except:
        return None

def _extract_text_from_txt(file_path: Path) -> Optional[str]:
    for enc in ["utf-8", "cp1251", "latin-1"]:
        try:
            return file_path.read_text(encoding=enc).strip()
        except:
            continue
    return None

def process_extracted_dir(extract_dir: Path, archive_name: str) -> Path:
    json_path = extract_dir / "result.json"
    if not json_path.exists():
        json_candidates = list(extract_dir.rglob("result.json"))
        if not json_candidates:
            raise FileNotFoundError("result.json не найден")
        json_path = json_candidates[0]

    out_csv = Path(OUTPUT_DIR) / f"{archive_name}.csv"
    out_csv.parent.mkdir(parents=True, exist_ok=True)

    with open(json_path, 'rb') as f, open(out_csv, 'w', newline='', encoding='utf-8') as csvfile:
        writer = csv.DictWriter(csvfile, fieldnames=['timestamp', 'from', 'content', 'reply_time_seconds'])
        writer.writeheader()

        current_block = None
        prev_block_last_dt = None
        msg_count = 0

        for msg in ijson.items(f, 'messages.item'):
            if msg.get('type') != 'message':
                continue
            msg_count += 1
            if msg_count % 100 == 0:
                gc.collect()

            date_unixtime = safe_int(msg.get('date_unixtime', 0))
            dt = datetime.fromtimestamp(date_unixtime, tz=timezone.utc)
            from_name = msg.get('from', 'Unknown')
            text_raw = msg.get('text', '')
            if isinstance(text_raw, list):
                text = ''.join(item if isinstance(item, str) else item.get('text', '') for item in text_raw)
            else:
                text = str(text_raw) if text_raw else ''

            # Определение медиа
            def get_media_info(msg: dict) -> Optional[Tuple[str, str]]:
                media_type = msg.get("media_type")
                file_path = msg.get("file")
                if media_type and file_path:
                    return (media_type, file_path)
                for key in ["photo", "video", "document", "audio", "voice_message",
                            "sticker", "animation", "video_file", "audio_file"]:
                    if key in msg:
                        val = msg[key]
                        if isinstance(val, str):
                            return (key, val)
                        if isinstance(val, dict) and "file" in val:
                            return (key, val["file"])
                if file_path:
                    return ("document", file_path)
                return None

            media_info = get_media_info(msg)
            media_desc = ''
            if media_info:
                media_type, rel_path = media_info
                file_path = find_media_file(extract_dir, rel_path)
                if file_path and file_path.exists():
                    # Проверяем кэш
                    cached = load_from_cache(file_path)
                    if cached is not None:
                        media_desc = cached
                    else:
                        # Запускаем обработку в отдельном процессе
                        output_file = tempfile.mktemp(suffix=".txt")
                        p = multiprocessing.Process(
                            target=_media_worker_task,
                            args=(str(file_path), media_type, output_file)
                        )
                        p.start()
                        p.join(timeout=120)  # Ждём максимум 2 минуты
                        if p.is_alive():
                            p.terminate()
                            p.join()
                            media_desc = "(ошибка обработки - таймаут)"
                        else:
                            try:
                                with open(output_file, "r", encoding="utf-8") as res_f:
                                    res = res_f.read().strip()
                                if res == "ERROR":
                                    media_desc = "(ошибка обработки)"
                                else:
                                    media_desc = res
                                    save_to_cache(file_path, media_desc)
                            except Exception:
                                media_desc = "(ошибка обработки)"
                        # Удаляем временный файл
                        try:
                            os.remove(output_file)
                        except Exception:
                            pass
                else:
                    media_desc = "(файл отсутствует)"

            content = (text + ' ' + media_desc).strip()
            if not content:
                continue

            # Запись с объединением и reply_time
            if current_block and current_block['author'] == from_name and \
               (dt - current_block['last_dt']) <= timedelta(minutes=MERGE_WINDOW_MINUTES):
                current_block['content'] += '\n' + content
                current_block['last_dt'] = dt
            else:
                if current_block:
                    reply = 0.0
                    if prev_block_last_dt is not None:
                        reply = (current_block['first_dt'] - prev_block_last_dt).total_seconds()
                    writer.writerow({
                        'timestamp': current_block['last_dt'].strftime("%d.%m.%Y %H:%M:%S"),
                        'from': current_block['author'],
                        'content': current_block['content'],
                        'reply_time_seconds': reply
                    })
                    prev_block_last_dt = current_block['last_dt']
                current_block = {
                    'author': from_name,
                    'first_dt': dt,
                    'last_dt': dt,
                    'content': content
                }

        # Последний блок
        if current_block:
            reply = 0.0
            if prev_block_last_dt is not None:
                reply = (current_block['first_dt'] - prev_block_last_dt).total_seconds()
            writer.writerow({
                'timestamp': current_block['last_dt'].strftime("%d.%m.%Y %H:%M:%S"),
                'from': current_block['author'],
                'content': current_block['content'],
                'reply_time_seconds': reply
            })

    print(f"✅ Обработано сообщений: {msg_count}")
    print(f"📦 Готовый CSV: {out_csv} ({out_csv.stat().st_size/1024/1024:.2f} MB)")
    return out_csv

def download_from_yandex(public_url: str, dest_dir: Path) -> Path:
    try:
        api_url = "https://cloud-api.yandex.net/v1/disk/public/resources/download"
        resp = requests.get(api_url, params={"public_key": public_url})
        resp.raise_for_status()
        data = resp.json()
        download_url = data["href"]
        file_name = data.get("name") or Path(unquote(urlparse(download_url).path)).name
        dest_path = dest_dir / file_name
        logger.info(f"Скачиваем {file_name} с Яндекс.Диска...")
        with requests.get(download_url, stream=True) as r:
            r.raise_for_status()
            with open(dest_path, "wb") as f:
                for chunk in r.iter_content(chunk_size=8192):
                    f.write(chunk)
        logger.info(f"Файл скачан: {dest_path.name}")
        return dest_path
    except Exception as e:
        logger.error(f"Ошибка загрузки из Яндекс.Диска: {e}")
        raise

def build_file_index(root_dir: Path) -> Dict[str, Path]:
    """Строит словарь {имя_файла: полный_путь} для всех файлов в root_dir."""
    index = {}
    for entry in root_dir.rglob("*"):
        if entry.is_file():
            index[entry.name] = entry
    logger.info(f"Построен индекс файлов: {len(index)} записей")
    return index

def find_media_file(base_dir: Path, file_path_str: str, file_index: Dict[str, Path]) -> Optional[Path]:
    """
    Ищет файл сначала по точному пути, затем по имени через индекс.
    """
    decoded = unquote(file_path_str).lstrip("/").replace("\\", "/")
    # Точный путь
    candidate = base_dir / decoded
    if candidate.exists():
        return candidate
    # Поиск по имени файла
    name = Path(decoded).name
    if name and name in file_index:
        return file_index[name]
    return None

def cache_key_for(file_path: Path) -> str:
    stat = file_path.stat()
    raw = f"{file_path.resolve()}:{stat.st_mtime}:{stat.st_size}"
    return hashlib.md5(raw.encode()).hexdigest()

def load_from_cache(file_path: Path) -> Optional[str]:
    key = cache_key_for(file_path)
    cache_file = CACHE_DIR / key
    if cache_file.exists():
        return cache_file.read_text(encoding="utf-8")
    return None

def save_to_cache(file_path: Path, result: str):
    key = cache_key_for(file_path)
    cache_file = CACHE_DIR / key
    cache_file.write_text(result, encoding="utf-8")
    logger.debug(f"Сохранено в кэш: {file_path.name}")

def _media_worker_task(file_path_str: str, media_type: str, output_file: str):
    """
    Обрабатывает один медиафайл и записывает результат в output_file.
    При ошибке пишет "ERROR".
    """
    try:
        file_path = Path(file_path_str)
        ext = file_path.suffix.lower()
        audio_video_ext = {".mp3", ".wav", ".ogg", ".m4a", ".flac", ".opus",
                           ".mp4", ".webm", ".mkv", ".avi", ".mov"}
        image_ext = {".jpg", ".jpeg", ".png", ".webp", ".bmp", ".gif", ".tiff", ".tif"}

        if (media_type in ("audio", "voice_message", "video", "animation",
                           "video_file", "audio_file", "voice", "round_video")
                or ext in audio_video_ext):
            if not _ensure_audio_stream(file_path):
                Path(output_file).write_text("[нет аудиодорожки]", encoding="utf-8")
                return
            wav_path = None
            txt_path = None
            try:
                with tempfile.NamedTemporaryFile(suffix=".wav", delete=False) as tmp:
                    wav_path = Path(tmp.name)
                if not _convert_to_wav(file_path, wav_path):
                    Path(output_file).write_text("ERROR", encoding="utf-8")
                    return
                engine = "diarize" if (ENABLE_DIARIZATION and NEWTON_ENGINE in ("v3", "diarize")) else NEWTON_ENGINE
                with tempfile.NamedTemporaryFile(suffix=".txt", delete=False) as tmp:
                    txt_path = Path(tmp.name)
                cmd = ["newton", "transcribe", str(wav_path), "-o", str(txt_path), "-e", engine]
                if engine == "diarize" and DIARIZE_SPEAKERS is not None:
                    cmd.extend(["-n", str(DIARIZE_SPEAKERS)])
                subprocess.run(cmd, check=True, capture_output=True, text=True, timeout=600)
                result = txt_path.read_text(encoding="utf-8").strip() if txt_path.exists() else "ERROR"
                Path(output_file).write_text(result, encoding="utf-8")
            except Exception:
                Path(output_file).write_text("ERROR", encoding="utf-8")
            finally:
                if wav_path and wav_path.exists():
                    wav_path.unlink(missing_ok=True)
                if txt_path and txt_path.exists():
                    txt_path.unlink(missing_ok=True)

        elif media_type in ("photo", "sticker") or ext in image_ext:
            try:
                img = Image.open(file_path)
                text = pytesseract.image_to_string(img, lang=OCR_LANG).strip()
                Path(output_file).write_text(text if text else "ERROR", encoding="utf-8")
            except Exception:
                Path(output_file).write_text("ERROR", encoding="utf-8")

        else:
            # Документы
            try:
                if ext in {'.pptx', '.ppt'}:
                    text = _extract_text_from_pptx(file_path)
                elif ext in {'.docx', '.doc'}:
                    text = _extract_text_from_docx(file_path)
                elif ext == '.pdf':
                    text = _extract_text_from_pdf(file_path)
                elif ext in {'.xlsx', '.xls'}:
                    text = _extract_text_from_xlsx(file_path)
                else:
                    text = _extract_text_from_txt(file_path)
                Path(output_file).write_text(text if text else "ERROR", encoding="utf-8")
            except Exception:
                Path(output_file).write_text("ERROR", encoding="utf-8")
    except MemoryError:
        Path(output_file).write_text("ERROR", encoding="utf-8")
    except Exception:
        Path(output_file).write_text("ERROR", encoding="utf-8")

def _ensure_audio_stream(file_path: Path) -> bool:
    try:
        res = subprocess.run(["ffprobe", "-v", "error", "-select_streams", "a:0",
                              "-show_entries", "stream=codec_type", "-of", "csv=p=0", str(file_path)],
                             capture_output=True, text=True, timeout=15)
        return "audio" in res.stdout.strip()
    except: return False

def _convert_to_wav(input_path: Path, output_path: Path) -> bool:
    try:
        subprocess.run(["ffmpeg", "-y", "-i", str(input_path), "-ac", "1", "-ar", "16000",
                        "-sample_fmt", "s16", str(output_path)],
                       check=True, capture_output=True, timeout=120)
        return output_path.exists()
    except: return False

def _extract_text_from_pptx(pptx_path: Path) -> Optional[str]:
    try:
        prs = Presentation(str(pptx_path))
        parts = []
        for i, slide in enumerate(prs.slides, 1):
            slide_text = []
            def extract_shape(shape):
                if shape.has_text_frame:
                    for para in shape.text_frame.paragraphs:
                        if para.text.strip(): slide_text.append(para.text.strip())
                if shape.has_table:
                    rows = ["| " + " | ".join(cell.text for cell in row.cells) + " |" for row in shape.table.rows]
                    if rows: slide_text.append("\n".join(rows))
                if shape.shape_type == MSO_SHAPE_TYPE.GROUP:
                    for child in shape.shapes: extract_shape(child)
            for shape in slide.shapes: extract_shape(shape)
            if slide.has_notes_slide and slide.notes_slide.notes_text_frame:
                notes = slide.notes_slide.notes_text_frame.text.strip()
                if notes: slide_text.append(f"[Заметки]: {notes}")
            if slide_text: parts.append(f"### Слайд {i}\n" + "\n".join(slide_text))
        return "\n\n".join(parts) if parts else None
    except:
        return None

def _extract_text_from_docx(docx_path: Path) -> Optional[str]:
    try:
        doc = Document(str(docx_path))
        parts = [p.text for p in doc.paragraphs if p.text.strip()]
        for table in doc.tables:
            rows = ["| " + " | ".join(cell.text for cell in row.cells) + " |" for row in table.rows]
            if rows: parts.append("\n".join(rows))
        return "\n\n".join(parts) if parts else None
    except:
        return None

def _extract_text_from_pdf(pdf_path: Path) -> Optional[str]:
    text = None
    try:
        reader = PdfReader(str(pdf_path))
        text = "\n".join(p.extract_text() or "" for p in reader.pages).strip()
        if text: return text
    except Exception:
        pass
    try:
        with pdfplumber.open(str(pdf_path)) as pdf:
            text = "\n".join(p.extract_text() or "" for p in pdf.pages).strip()
            if text: return text
    except Exception:
        pass
    if not text and ENABLE_OCR:
        with tempfile.TemporaryDirectory() as tmpdir:
            try:
                subprocess.run(["pdftoppm", "-png", "-r", "200", str(pdf_path), f"{tmpdir}/page"],
                               check=False, timeout=120)
                pages = []
                for img_file in sorted(Path(tmpdir).glob("*.png")):
                    img = Image.open(img_file)
                    t = pytesseract.image_to_string(img, lang=OCR_LANG).strip()
                    if t: pages.append(t)
                text = "\n".join(pages) if pages else None
            except Exception:
                pass
    return text

def _extract_text_from_xlsx(xlsx_path: Path) -> Optional[str]:
    try:
        wb = load_workbook(str(xlsx_path), read_only=True, data_only=True)
        parts = []
        for name in wb.sheetnames:
            ws = wb[name]
            rows = list(ws.iter_rows(max_row=100, values_only=True))
            if not rows: continue
            header = rows[0]
            parts.append(f"### Лист: {name}")
            parts.append("| " + " | ".join(str(c) if c is not None else "" for c in header) + " |")
            parts.append("| " + " | ".join(["---"] * len(header)) + " |")
            for row in rows[1:]:
                parts.append("| " + " | ".join(str(c) if c is not None else "" for c in row) + " |")
            parts.append("")
        return "\n".join(parts) if parts else None
    except:
        return None

def _extract_text_from_txt(file_path: Path) -> Optional[str]:
    for enc in ["utf-8", "cp1251", "latin-1"]:
        try:
            return file_path.read_text(encoding=enc).strip()
        except:
            continue
    return None

def process_extracted_dir(extract_dir: Path, archive_name: str) -> Path:
    json_path = extract_dir / "result.json"
    if not json_path.exists():
        json_candidates = list(extract_dir.rglob("result.json"))
        if not json_candidates:
            raise FileNotFoundError("result.json не найден")
        json_path = json_candidates[0]

    out_csv = Path(OUTPUT_DIR) / f"{archive_name}.csv"
    out_csv.parent.mkdir(parents=True, exist_ok=True)

    # Построение индекса файлов
    file_index = build_file_index(extract_dir)

    # Определяем общее количество сообщений для прогресс-бара (предварительный проход)
    total_msgs = 0
    with open(json_path, 'rb') as f:
        for msg in ijson.items(f, 'messages.item'):
            if msg.get('type') == 'message':
                total_msgs += 1
    logger.info(f"Всего сообщений в чате: {total_msgs}")

    with open(json_path, 'rb') as f, open(out_csv, 'w', newline='', encoding='utf-8') as csvfile:
        writer = csv.DictWriter(csvfile, fieldnames=['timestamp', 'from', 'content', 'reply_time_seconds'])
        writer.writeheader()

        current_block = None
        prev_block_last_dt = None
        msg_count = 0

        # Прогресс-бар
        pbar = tqdm(total=total_msgs, desc="Обработка сообщений", unit="сообщ.")

        for msg in ijson.items(f, 'messages.item'):
            if msg.get('type') != 'message':
                continue
            msg_count += 1
            pbar.update(1)

            date_unixtime = safe_int(msg.get('date_unixtime', 0))
            dt = datetime.fromtimestamp(date_unixtime, tz=timezone.utc)
            from_name = msg.get('from', 'Unknown')
            text_raw = msg.get('text', '')
            if isinstance(text_raw, list):
                text = ''.join(item if isinstance(item, str) else item.get('text', '') for item in text_raw)
            else:
                text = str(text_raw) if text_raw else ''

            # Определение медиа
            def get_media_info(msg: dict) -> Optional[Tuple[str, str]]:
                media_type = msg.get("media_type")
                file_path = msg.get("file")
                if media_type and file_path:
                    return (media_type, file_path)
                for key in ["photo", "video", "document", "audio", "voice_message",
                            "sticker", "animation", "video_file", "audio_file"]:
                    if key in msg:
                        val = msg[key]
                        if isinstance(val, str):
                            return (key, val)
                        if isinstance(val, dict) and "file" in val:
                            return (key, val["file"])
                if file_path:
                    return ("document", file_path)
                return None

            media_info = get_media_info(msg)
            media_desc = ''
            if media_info:
                media_type, rel_path = media_info
                file_path = find_media_file(extract_dir, rel_path, file_index)
                if file_path and file_path.exists():
                    # Проверяем кэш
                    cached = load_from_cache(file_path)
                    if cached is not None:
                        media_desc = cached
                    else:
                        # Запускаем обработку в отдельном процессе
                        output_file = tempfile.mktemp(suffix=".txt")
                        p = multiprocessing.Process(
                            target=_media_worker_task,
                            args=(str(file_path), media_type, output_file)
                        )
                        p.start()
                        p.join(timeout=120)
                        if p.is_alive():
                            p.terminate()
                            p.join()
                            media_desc = "(ошибка обработки - таймаут)"
                            logger.warning(f"Таймаут обработки файла: {file_path.name}")
                        else:
                            try:
                                with open(output_file, "r", encoding="utf-8") as res_f:
                                    res = res_f.read().strip()
                                if res == "ERROR":
                                    media_desc = "(ошибка обработки)"
                                    logger.error(f"Ошибка обработки файла: {file_path.name}")
                                else:
                                    media_desc = res
                                    save_to_cache(file_path, media_desc)
                            except Exception as e:
                                media_desc = "(ошибка обработки)"
                                logger.error(f"Ошибка чтения результата для {file_path.name}: {e}")
                        # Удаляем временный файл
                        try:
                            os.remove(output_file)
                        except Exception:
                            pass
                else:
                    media_desc = "(файл отсутствует)"

            content = (text + ' ' + media_desc).strip()
            if not content:
                continue

            # Запись с объединением и reply_time
            if current_block and current_block['author'] == from_name and \
               (dt - current_block['last_dt']) <= timedelta(minutes=MERGE_WINDOW_MINUTES):
                current_block['content'] += '\n' + content
                current_block['last_dt'] = dt
            else:
                if current_block:
                    reply = 0.0
                    if prev_block_last_dt is not None:
                        reply = (current_block['first_dt'] - prev_block_last_dt).total_seconds()
                    writer.writerow({
                        'timestamp': current_block['last_dt'].strftime("%d.%m.%Y %H:%M:%S"),
                        'from': current_block['author'],
                        'content': current_block['content'],
                        'reply_time_seconds': reply
                    })
                    prev_block_last_dt = current_block['last_dt']
                current_block = {
                    'author': from_name,
                    'first_dt': dt,
                    'last_dt': dt,
                    'content': content
                }

        # Последний блок
        if current_block:
            reply = 0.0
            if prev_block_last_dt is not None:
                reply = (current_block['first_dt'] - prev_block_last_dt).total_seconds()
            writer.writerow({
                'timestamp': current_block['last_dt'].strftime("%d.%m.%Y %H:%M:%S"),
                'from': current_block['author'],
                'content': current_block['content'],
                'reply_time_seconds': reply
            })

        pbar.close()

    logger.info(f"Обработано сообщений: {msg_count}")
    logger.info(f"Результат сохранён: {out_csv} ({out_csv.stat().st_size/1024/1024:.2f} MB)")
    return out_csv

## 📝 Принятые изменения

1.  **PyPDF2 → pypdf** (критично). PyPDF2 давно не обновляется, переезжаем на актуальный форк `pypdf`.
2.  **Индекс файлов** (критично). Если в архиве есть файлы с одинаковыми именами, старый индекс перезаписывал их. Теперь для каждого имени хранится список путей, а при поиске мы логируем предупреждение и берём первый.
3.  **Двойной проход JSON** (производительность). Убрал предварительный подсчёт сообщений, чтобы не читать огромный JSON дважды. Прогресс-бар теперь обновляется каждые 100 сообщений без общего `total`.
4.  **Атомарная запись кэша** (надёжность). Чтобы избежать битых файлов при случайном прерывании, результат сначала пишется во временный файл, а затем атомарно переименовывается.
5.  **Пустой OCR** (семантика). Если на картинке нет текста, в расшифровку попадёт `[нет текста]`, а не `(ошибка обработки)`.
6.  **Название колонки**. Переименовал `reply_time_seconds` в `inter_block_gap_seconds` – теперь название точно отражает смысл: интервал между последним сообщением одного автора и первым сообщением следующего.

## ⚙️ Что оставлено без изменений (и почему)

*   **Diarize** – текущая логика верна: в Newton CLI диаризация – это отдельный движок `diarize`, а не флаг к `v3`. Если она включена, мы так и переключаем движок.
*   **Последовательная обработка** – асинхронное выполнение нарушит порядок сообщений, что сделает результат непригодным для анализа диалогов. Для сохранения хронологии мы жертвуем скоростью.
*   **TQDM и блокировки** – прогресс-бар действительно будет замирать на время обработки медиа. Чтобы было понятно, что процесс не завис, в лог пишется сообщение о начале расшифровки каждого файла. Полное распараллеливание пока считаю преждевременным усложнением.

def _ensure_audio_stream(file_path: Path) -> bool:
    try:
        res = subprocess.run(["ffprobe", "-v", "error", "-select_streams", "a:0",
                              "-show_entries", "stream=codec_type", "-of", "csv=p=0", str(file_path)],
                             capture_output=True, text=True, timeout=15)
        return "audio" in res.stdout.strip()
    except: return False

    try:

                        "-sample_fmt", "s16", str(output_path)],
                       check=True, capture_output=True, timeout=120)
        return output_path.exists()
    except: return False

def _extract_text_from_pptx(pptx_path: Path) -> Optional[str]:
    try:
        prs = Presentation(str(pptx_path))
        parts = []
        for i, slide in enumerate(prs.slides, 1):
            slide_text = []
            def extract_shape(shape):
                if shape.has_text_frame:
                    for para in shape.text_frame.paragraphs:
                        if para.text.strip(): slide_text.append(para.text.strip())
                if shape.has_table:
                    rows = ["| " + " | ".join(cell.text for cell in row.cells) + " |" for row in shape.table.rows]
                    if rows: slide_text.append("\n".join(rows))
                if shape.shape_type == MSO_SHAPE_TYPE.GROUP:
                    for child in shape.shapes: extract_shape(child)
            for shape in slide.shapes: extract_shape(shape)
            if slide.has_notes_slide and slide.notes_slide.notes_text_frame:
                notes = slide.notes_slide.notes_text_frame.text.strip()
                if notes: slide_text.append(f"[Заметки]: {notes}")
            if slide_text: parts.append(f"### Слайд {i}\n" + "\n".join(slide_text))
        return "\n\n".join(parts) if parts else None
    except:
        return None

        return None

def _extract_text_from_xlsx(xlsx_path: Path) -> Optional[str]:
    try:
        wb = load_workbook(str(xlsx_path), read_only=True, data_only=True)
        parts = []
        for name in wb.sheetnames:
            ws = wb[name]
            rows = list(ws.iter_rows(max_row=100, values_only=True))
            if not rows: continue
            header = rows[0]
            parts.append(f"### Лист: {name}")
            parts.append("| " + " | ".join(str(c) if c is not None else "" for c in header) + " |")
            parts.append("| " + " | ".join(["---"] * len(header)) + " |")
            for row in rows[1:]:
                parts.append("| " + " | ".join(str(c) if c is not None else "" for c in row) + " |")
            parts.append("")
        return "\n".join(parts) if parts else None
    except:
        return None

                        if isinstance(val, dict) and "file" in val:
                            return (key, val["file"])
                if file_path:
                    return ("document", file_path)
                return None

            media_desc = ''
            if media_info:

                file_path = find_media_file(extract_dir, rel_path, file_index)
                if file_path and file_path.exists():
                    cached = load_from_cache(file_path)
                    if cached is not None:
                        media_desc = cached
                    else:
                        logger.info(f"Обработка файла: {file_path.name}")
                        output_file = tempfile.mktemp(suffix=".txt")
                        p = multiprocessing.Process(
                            target=_media_worker_task,
                            args=(str(file_path), media_type, output_file)
                        )
                        p.start()
                        p.join(timeout=120)
                        if p.is_alive():
                            p.terminate()
                            p.join()
                            media_desc = "(ошибка обработки - таймаут)"
                            logger.warning(f"Таймаут обработки файла: {file_path.name}")
                        else:
                            try:
                                with open(output_file, "r", encoding="utf-8") as res_f:
                                    res = res_f.read().strip()
                                if res == "ERROR":
                                    media_desc = "(ошибка обработки)"
                                    logger.error(f"Ошибка обработки файла: {file_path.name}")
                                else:
                                    media_desc = res
                                    save_to_cache(file_path, media_desc)
                            except Exception as e:

                                logger.error(f"Ошибка чтения результата для {file_path.name}: {e}")
                        try:
                            os.remove(output_file)
                        except Exception:
                            pass
                else:
                    media_desc = "(файл отсутствует)"

            content = (text + ' ' + media_desc).strip()
            if not content:
                continue

            # Запись с объединением и интервалом между блоками
            if current_block and current_block['author'] == from_name and \
               (dt - current_block['last_dt']) <= timedelta(minutes=MERGE_WINDOW_MINUTES):
                current_block['content'] += '\n' + content
                current_block['last_dt'] = dt
            else:
                if current_block:
                    gap = 0.0
                    if prev_block_last_dt is not None:
                        gap = (current_block['first_dt'] - prev_block_last_dt).total_seconds()
                    writer.writerow({
                        'timestamp': current_block['last_dt'].strftime("%d.%m.%Y %H:%M:%S"),
                        'from': current_block['author'],
                        'content': current_block['content'],
                        'inter_block_gap_seconds': gap
                    })
                    prev_block_last_dt = current_block['last_dt']
                current_block = {
                    'author': from_name,
                    'first_dt': dt,
                    'last_dt': dt,
                    'content': content
                }

        if current_block:
            gap = 0.0
            if prev_block_last_dt is not None:
                gap = (current_block['first_dt'] - prev_block_last_dt).total_seconds()
            writer.writerow({
                'timestamp': current_block['last_dt'].strftime("%d.%m.%Y %H:%M:%S"),
                'from': current_block['author'],
                'content': current_block['content'],
                'inter_block_gap_seconds': gap
            })

        pbar.close()

    logger.info(f"Обработано сообщений: {msg_count}")

def find_media_file(base_dir: Path, file_path_str: str, file_index: Dict[str, List[Path]]) -> Optional[Path]:
    decoded = unquote(file_path_str).lstrip("/").replace("\\", "/")
    candidate = base_dir / decoded
    # Path traversal protection
    try:
        resolved = candidate.resolve()
        base_resolved = base_dir.resolve()
        if not (str(resolved).startswith(str(base_resolved) + os.sep) or resolved == base_resolved):
            logger.warning(f"Path traversal blocked: {file_path_str} -> {resolved}")
            return None
    except (OSError, RuntimeError):
        return None
    if candidate.exists():
        return candidate
    name = Path(decoded).name
    if name in file_index:
        candidates = file_index[name]
        if len(candidates) > 1:
            logger.warning(f"Коллизия имён '{name}': найдено {len(candidates)} файлов, берём первый {candidates[0]}")
        return candidates[0]
    return None

def _ensure_audio_stream(file_path: Path) -> bool:
    try:
        res = subprocess.run(["ffprobe", "-v", "error", "-select_streams", "a:0",
                              "-show_entries", "stream=codec_type", "-of", "csv=p=0", str(file_path)],
                             capture_output=True, text=True, timeout=15)
        return "audio" in res.stdout.strip()
    except: return False

def _convert_to_wav(input_path: Path, output_path: Path) -> bool:
    try:
        subprocess.run(["ffmpeg", "-y", "-i", str(input_path), "-ac", "1", "-ar", "16000",
                        "-sample_fmt", "s16", str(output_path)],
                       check=True, capture_output=True, timeout=120)
        return output_path.exists()
    except: return False

def _extract_text_from_pptx(pptx_path: Path) -> Optional[str]:
    try:
        prs = Presentation(str(pptx_path))
        parts = []
        for i, slide in enumerate(prs.slides, 1):
            slide_text = []
            def extract_shape(shape):
                if shape.has_text_frame:
                    for para in shape.text_frame.paragraphs:
                        if para.text.strip(): slide_text.append(para.text.strip())
                if shape.has_table:
                    rows = ["| " + " | ".join(cell.text for cell in row.cells) + " |" for row in shape.table.rows]
                    if rows: slide_text.append("\n".join(rows))
                if shape.shape_type == MSO_SHAPE_TYPE.GROUP:
                    for child in shape.shapes: extract_shape(child)
            for shape in slide.shapes: extract_shape(shape)
            if slide.has_notes_slide and slide.notes_slide.notes_text_frame:
                notes = slide.notes_slide.notes_text_frame.text.strip()
                if notes: slide_text.append(f"[Заметки]: {notes}")
            if slide_text: parts.append(f"### Слайд {i}\n" + "\n".join(slide_text))
        return "\n\n".join(parts) if parts else None
    except:
        return None

def _extract_text_from_docx(docx_path: Path) -> Optional[str]:
    try:
        doc = Document(str(docx_path))
        parts = [p.text for p in doc.paragraphs if p.text.strip()]
        for table in doc.tables:
            rows = ["| " + " | ".join(cell.text for cell in row.cells) + " |" for row in table.rows]
            if rows: parts.append("\n".join(rows))
        return "\n\n".join(parts) if parts else None
    except:
        return None

def _extract_text_from_pdf(pdf_path: Path) -> Optional[str]:
    text = None
    try:
        reader = PdfReader(str(pdf_path))
        text = "\n".join(p.extract_text() or "" for p in reader.pages).strip()
        if text: return text
    except Exception:
        pass
    try:
        with pdfplumber.open(str(pdf_path)) as pdf:
            text = "\n".join(p.extract_text() or "" for p in pdf.pages).strip()
            if text: return text
    except Exception:
        pass
    if not text and ENABLE_OCR:
        with tempfile.TemporaryDirectory() as tmpdir:
            try:
                subprocess.run(["pdftoppm", "-png", "-r", "200", str(pdf_path), f"{tmpdir}/page"],
                               check=False, timeout=120)
                pages = []
                for img_file in sorted(Path(tmpdir).glob("*.png")):
                    img = Image.open(img_file)
                    t = pytesseract.image_to_string(img, lang=OCR_LANG).strip()
                    if t: pages.append(t)
                text = "\n".join(pages) if pages else None
            except Exception:
                pass
    return text

def _extract_text_from_xlsx(xlsx_path: Path) -> Optional[str]:
    try:
        wb = load_workbook(str(xlsx_path), read_only=True, data_only=True)
        parts = []
        for name in wb.sheetnames:
            ws = wb[name]
            rows = list(ws.iter_rows(max_row=100, values_only=True))
            if not rows: continue
            header = rows[0]
            parts.append(f"### Лист: {name}")
            parts.append("| " + " | ".join(str(c) if c is not None else "" for c in header) + " |")
            parts.append("| " + " | ".join(["---"] * len(header)) + " |")
            for row in rows[1:]:
                parts.append("| " + " | ".join(str(c) if c is not None else "" for c in row) + " |")
            parts.append("")
        return "\n".join(parts) if parts else None
    except:
        return None

    for enc in ["utf-8", "cp1251", "latin-1"]:

## 🔧 Основные изменения в v5

1. **Path Traversal Protection**  
   – Реализована функция `is_safe_path()` на базе `Path.relative_to()`, которая устойчива к символьным ссылкам и не зависит от платформенных разделителей.

2. **Лимит страниц для OCR**  
   – В `_extract_text_from_pdf()` добавлена константа `MAX_OCR_PAGES = 50`. При превышении лимита логируется предупреждение, дальнейшие страницы не обрабатываются. Это предотвращает переполнение памяти на многогигабайтных PDF.

3. **Единообразие пустых результатов**  
   – Для всех документов и изображений пустой результат теперь `"[нет текста]"`. Аудио без дорожки помечается `"[нет аудиодорожки]"`. Ошибки обработки по‑прежнему `"(ошибка обработки)"`.

4. **Прогресс-бар для малых чатов**  
   – Если сообщений меньше 100, `pbar.update(1)` вызывается для каждого; иначе – пакетами по 100, как прежде. Индикатор всегда показывает реальный прогресс.

5. **Явное закрытие изображений в OCR**  
   – Добавлен вызов `img.close()` после распознавания каждой страницы PDF, чтобы ускорить освобождение памяти.

def is_safe_path(base_dir: Path, candidate: Path) -> bool:
    """Проверяет, что candidate находится внутри base_dir (с разрешением символьных ссылок)."""
    try:
        base_resolved = base_dir.resolve(strict=True)
        candidate_resolved = candidate.resolve(strict=False)
        candidate_resolved.relative_to(base_resolved)
        return True
    except (ValueError, OSError, RuntimeError):
        return False

def build_file_index(root_dir: Path) -> Dict[str, List[Path]]:
    """Строит индекс {имя_файла: [полные пути]} для всех файлов в root_dir."""
    index = defaultdict(list)
    for entry in root_dir.rglob("*"):
        if entry.is_file():
            index[entry.name].append(entry)
    logger.info(f"Построен индекс файлов: {len(index)} уникальных имён")
    return index

def find_media_file(base_dir: Path, file_path_str: str, file_index: Dict[str, List[Path]]) -> Optional[Path]:
    """
    Ищет файл по относительному пути из JSON.
    Сначала проверяет точный путь, затем ищет по имени через индекс.
    """
    decoded = unquote(file_path_str).lstrip("/").replace("\\", "/")
    candidate = base_dir / decoded

    # Path traversal protection
    if not is_safe_path(base_dir, candidate):
        logger.warning(f"Path traversal blocked: {file_path_str} -> {candidate.resolve()}")
        return None

    if candidate.exists():
        return candidate

    name = Path(decoded).name
    if name in file_index:
        candidates = file_index[name]
        if len(candidates) > 1:
            # Для детерминизма сортируем по строковому представлению пути
            candidates = sorted(candidates, key=lambda p: str(p))
            logger.warning(f"Коллизия имён '{name}': найдено {len(candidates)} файлов, берём первый {candidates[0]}")
        return candidates[0]
    return None

                return

            tmp_wav_fd = tempfile.NamedTemporaryFile(suffix=".wav", delete=False)
            wav_path = Path(tmp_wav_fd.name)
            tmp_wav_fd.close()

            if not _convert_to_wav(file_path, wav_path):
                Path(output_file).write_text("ERROR", encoding="utf-8")
                return

            tmp_txt_fd = tempfile.NamedTemporaryFile(suffix=".txt", delete=False)
            txt_path = Path(tmp_txt_fd.name)
            tmp_txt_fd.close()

            engine = "diarize" if (ENABLE_DIARIZATION and NEWTON_ENGINE in ("v3", "diarize")) else NEWTON_ENGINE
            cmd = ["newton", "transcribe", str(wav_path), "-o", str(txt_path), "-e", engine]
            if engine == "diarize" and DIARIZE_SPEAKERS is not None:
                cmd.extend(["-n", str(DIARIZE_SPEAKERS)])
            subprocess.run(cmd, check=True, capture_output=True, text=True, timeout=600)
            result = txt_path.read_text(encoding="utf-8").strip() if txt_path.exists() else "ERROR"
            Path(output_file).write_text(result, encoding="utf-8")

        elif media_type in ("photo", "sticker") or ext in image_ext:
            try:
                img = Image.open(file_path)
                text = pytesseract.image_to_string(img, lang=OCR_LANG).strip()
                img.close()
                Path(output_file).write_text(text if text else "[нет текста]", encoding="utf-8")
            except Exception:
                Path(output_file).write_text("ERROR", encoding="utf-8")

        else:
            # Документы
            try:
                if ext in {'.pptx', '.ppt'}:
                    text = _extract_text_from_pptx(file_path)
                elif ext in {'.docx', '.doc'}:
                    text = _extract_text_from_docx(file_path)
                elif ext == '.pdf':
                    text = _extract_text_from_pdf(file_path)
                elif ext in {'.xlsx', '.xls'}:
                    text = _extract_text_from_xlsx(file_path)
                else:
                    text = _extract_text_from_txt(file_path)
                Path(output_file).write_text(text if text else "[нет текста]", encoding="utf-8")
            except Exception:
                Path(output_file).write_text("ERROR", encoding="utf-8")
    except MemoryError:
        Path(output_file).write_text("ERROR", encoding="utf-8")
    except Exception:
        Path(output_file).write_text("ERROR", encoding="utf-8")
    finally:
        for path in [wav_path, txt_path]:
            if path is not None and path.exists():
                try:
                    path.unlink(missing_ok=True)
                except Exception:
                    pass
        for fd in [tmp_wav_fd, tmp_txt_fd]:
            if fd is not None:
                try:
                    fd.close()
                except Exception:
                    pass

def _ensure_audio_stream(file_path: Path) -> bool:
    """Проверяет наличие аудиодорожки в медиафайле."""
    try:
        res = subprocess.run(["ffprobe", "-v", "error", "-select_streams", "a:0",
                              "-show_entries", "stream=codec_type", "-of", "csv=p=0", str(file_path)],
                             capture_output=True, text=True, timeout=15)
        return "audio" in res.stdout.strip()
    except: return False

def _convert_to_wav(input_path: Path, output_path: Path) -> bool:
    """Конвертирует аудио/видео в моно WAV 16 кГц."""
    try:
        subprocess.run(["ffmpeg", "-y", "-i", str(input_path), "-ac", "1", "-ar", "16000",
                        "-sample_fmt", "s16", str(output_path)],
                       check=True, capture_output=True, timeout=120)
        return output_path.exists()
    except: return False

def _extract_text_from_pptx(pptx_path: Path) -> Optional[str]:
    """Извлекает текст из презентации PowerPoint."""
    try:
        prs = Presentation(str(pptx_path))
        parts = []
        for i, slide in enumerate(prs.slides, 1):
            slide_text = []
            def extract_shape(shape):
                if shape.has_text_frame:
                    for para in shape.text_frame.paragraphs:
                        if para.text.strip(): slide_text.append(para.text.strip())
                if shape.has_table:
                    rows = ["| " + " | ".join(cell.text for cell in row.cells) + " |" for row in shape.table.rows]
                    if rows: slide_text.append("\n".join(rows))
                if shape.shape_type == MSO_SHAPE_TYPE.GROUP:
                    for child in shape.shapes: extract_shape(child)
            for shape in slide.shapes: extract_shape(shape)
            if slide.has_notes_slide and slide.notes_slide.notes_text_frame:
                notes = slide.notes_slide.notes_text_frame.text.strip()
                if notes: slide_text.append(f"[Заметки]: {notes}")
            if slide_text: parts.append(f"### Слайд {i}\n" + "\n".join(slide_text))
        return "\n\n".join(parts) if parts else None
    except:
        return None

def _extract_text_from_docx(docx_path: Path) -> Optional[str]:
    """Извлекает текст из документа Word."""
    try:
        doc = Document(str(docx_path))
        parts = [p.text for p in doc.paragraphs if p.text.strip()]
        for table in doc.tables:
            rows = ["| " + " | ".join(cell.text for cell in row.cells) + " |" for row in table.rows]
            if rows: parts.append("\n".join(rows))
        return "\n\n".join(parts) if parts else None
    except:
        return None

def _extract_text_from_pdf(pdf_path: Path) -> Optional[str]:
    """
    Извлекает текст из PDF, вначале стандартными средствами,
    при неудаче — OCR с ограничением страниц (MAX_OCR_PAGES).
    """
    text = None
    try:
        reader = PdfReader(str(pdf_path))
        text = "\n".join(p.extract_text() or "" for p in reader.pages).strip()
        if text: return text
    except Exception:
        pass
    try:
        with pdfplumber.open(str(pdf_path)) as pdf:
            text = "\n".join(p.extract_text() or "" for p in pdf.pages).strip()
            if text: return text
    except Exception:
        pass

    if not text and ENABLE_OCR:
        with tempfile.TemporaryDirectory() as tmpdir:
            try:
                result = subprocess.run([
                    "pdftoppm", "-png", "-r", "150",
                    "-f", "1", "-l", str(MAX_OCR_PAGES),
                    str(pdf_path), f"{tmpdir}/page"
                ], capture_output=True, text=True, timeout=300)

                pages = []
                for i, img_file in enumerate(sorted(Path(tmpdir).glob("*.png"))):
                    if i >= MAX_OCR_PAGES:
                        logger.warning(f"Достигнут лимит OCR страниц ({MAX_OCR_PAGES}) для {pdf_path.name}")
                        break
                    img = Image.open(img_file)
                    t = pytesseract.image_to_string(img, lang=OCR_LANG).strip()
                    img.close()
                    if t:
                        pages.append(t)
                    if i % 10 == 0:
                        gc.collect()
                text = "\n".join(pages) if pages else None
            except Exception as e:
                logger.warning(f"OCR PDF failed for {pdf_path.name}: {e}")
    return text if text else "[нет текста]"

def _extract_text_from_xlsx(xlsx_path: Path) -> Optional[str]:
    """Извлекает текст из Excel-файла (первые 100 строк каждого листа)."""
    try:
        wb = load_workbook(str(xlsx_path), read_only=True, data_only=True)
        parts = []
        for name in wb.sheetnames:
            ws = wb[name]
            rows = list(ws.iter_rows(max_row=100, values_only=True))
            if not rows: continue
            header = rows[0]
            parts.append(f"### Лист: {name}")
            parts.append("| " + " | ".join(str(c) if c is not None else "" for c in header) + " |")
            parts.append("| " + " | ".join(["---"] * len(header)) + " |")
            for row in rows[1:]:
                parts.append("| " + " | ".join(str(c) if c is not None else "" for c in row) + " |")
            parts.append("")
        return "\n".join(parts) if parts else None
    except:
        return None

def _extract_text_from_txt(file_path: Path) -> Optional[str]:
    """Читает текстовый файл, пробуя разные кодировки."""
    for enc in ["utf-8", "cp1251", "latin-1"]:
        try:
            return file_path.read_text(encoding=enc).strip()
        except:
            continue
    return None

def process_extracted_dir(extract_dir: Path, archive_name: str) -> Path:
    """
    Потоково читает result.json, обрабатывает медиафайлы
    и формирует сжатый CSV с хронологией сообщений.
    """
    json_path = extract_dir / "result.json"
    if not json_path.exists():
        json_candidates = list(extract_dir.rglob("result.json"))
        if not json_candidates:
            raise FileNotFoundError("result.json не найден")
        json_path = json_candidates[0]

    out_csv = Path(OUTPUT_DIR) / f"{archive_name}.csv"
    out_csv.parent.mkdir(parents=True, exist_ok=True)

    file_index = build_file_index(extract_dir)

    if ENABLE_DIARIZATION and NEWTON_ENGINE in ("v3", "diarize"):
        logger.info("Диаризация включена, будет использован движок 'diarize'")

    with open(json_path, 'rb') as f, open(out_csv, 'w', newline='', encoding='utf-8') as csvfile:
        writer = csv.DictWriter(csvfile, fieldnames=['timestamp', 'from', 'content', 'inter_block_gap_seconds'])
        writer.writeheader()

        current_block = None
        prev_block_last_dt = None
        msg_count = 0
        pbar = tqdm(desc="Обработка сообщений", unit="сообщ.")
        try:
            for msg in ijson.items(f, 'messages.item'):
                if msg.get('type') != 'message':
                    continue
                msg_count += 1
                # Прогресс-бар: для маленьких чатов обновляем каждое сообщение
                if msg_count < 100:
                    pbar.update(1)
                elif msg_count % 100 == 0:
                    pbar.update(100)

                date_unixtime = safe_int(msg.get('date_unixtime', 0))
                dt = datetime.fromtimestamp(date_unixtime, tz=timezone.utc)
                from_name = msg.get('from', 'Unknown')
                text_raw = msg.get('text', '')
                if isinstance(text_raw, list):
                    text = ''.join(item if isinstance(item, str) else item.get('text', '') for item in text_raw)
                else:
                    text = str(text_raw) if text_raw else ''

                media_info = get_media_info(msg)
                media_desc = ''
                if media_info:
                    media_type, rel_path = media_info
                    file_path = find_media_file(extract_dir, rel_path, file_index)
                    if file_path and file_path.exists():
                        cached = load_from_cache(file_path)
                        if cached is not None:
                            media_desc = cached
                        else:
                            logger.info(f"Обработка файла: {file_path.name}")
                            with tempfile.NamedTemporaryFile(suffix=".txt", delete=False) as res_tmp:
                                output_file = res_tmp.name
                            try:
                                p = multiprocessing.Process(
                                    target=_media_worker_task,
                                    args=(str(file_path), media_type, output_file)
                                )
                                p.start()
                                p.join(timeout=120)
                                if p.is_alive():
                                    p.terminate()
                                    p.join()
                                    media_desc = "(ошибка обработки - таймаут)"
                                    logger.warning(f"Таймаут обработки файла: {file_path.name}")
                                else:
                                    with open(output_file, "r", encoding="utf-8") as res_f:
                                        res = res_f.read().strip()
                                    if res == "ERROR":
                                        media_desc = "(ошибка обработки)"
                                        logger.error(f"Ошибка обработки файла: {file_path.name}")
                                    else:
                                        media_desc = res
                                        save_to_cache(file_path, media_desc)
                            finally:
                                try:
                                    os.remove(output_file)
                                except Exception:
                                    pass
                    else:
                        media_desc = "(файл отсутствует)"

                content = (text + ' ' + media_desc).strip()
                if not content:
                    continue

                # Запись с объединением и интервалом между блоками
                if current_block and current_block['author'] == from_name and \
                   (dt - current_block['last_dt']) <= timedelta(minutes=MERGE_WINDOW_MINUTES):
                    current_block['content'] += '\n' + content
                    current_block['last_dt'] = dt
                else:
                    if current_block:
                        gap = 0.0
                        if prev_block_last_dt is not None:
                            gap = (current_block['first_dt'] - prev_block_last_dt).total_seconds()
                        writer.writerow({
                            'timestamp': current_block['last_dt'].strftime("%d.%m.%Y %H:%M:%S"),
                            'from': current_block['author'],
                            'content': current_block['content'],
                            'inter_block_gap_seconds': gap
                        })
                        prev_block_last_dt = current_block['last_dt']
                    current_block = {
                        'author': from_name,
                        'first_dt': dt,
                        'last_dt': dt,
                        'content': content
                    }

            if current_block:
                gap = 0.0
                if prev_block_last_dt is not None:
                    gap = (current_block['first_dt'] - prev_block_last_dt).total_seconds()
                writer.writerow({
                    'timestamp': current_block['last_dt'].strftime("%d.%m.%Y %H:%M:%S"),
                    'from': current_block['author'],
                    'content': current_block['content'],
                    'inter_block_gap_seconds': gap
                })
        finally:
            pbar.n = msg_count
            pbar.refresh()
            pbar.close()

    logger.info(f"Обработано сообщений: {msg_count}")
    logger.info(f"Результат сохранён: {out_csv} ({out_csv.stat().st_size/1024/1024:.2f} MB)")
    return out_csv

## 🧪 Что проверено

- **Path traversal** — функция `is_safe_path()` использует `relative_to()` после разрешения символьных ссылок, поэтому корректно отклоняет любые попытки выйти за пределы базовой директории.
- **OCR для больших PDF** — конвертируются только первые `MAX_OCR_PAGES` страниц, после каждой десятой страницы вызывается сборщик мусора, изображения явно закрываются.
- **Пустые результаты** — единообразны и понятны.
- **Прогресс-бар** — плавно движется и для малых, и для больших чатов.

Код готов к промышленному использованию. Если вы решите в будущем перейти к параллельной архитектуре для супербольших архивов, это можно будет сделать без нарушения хронологии (например, накопить результаты в памяти и отсортировать по времени). Но сейчас я оставляю последовательный режим как самый надёжный.

    elif source_choice == "yandex":
        public_url = input("🔗 Введите публичную ссылку Яндекс.Диска: ").strip()

def cache_key_for(file_path: Path) -> str:
    stat = file_path.stat()
    raw = f"{file_path.resolve()}:{stat.st_mtime}:{stat.st_size}"
    return hashlib.md5(raw.encode()).hexdigest()

def load_from_cache(file_path: Path) -> Optional[str]:
    key = cache_key_for(file_path)
    cache_file = CACHE_DIR / key
    if cache_file.exists():
        return cache_file.read_text(encoding="utf-8")
    return None

def save_to_cache(file_path: Path, result: str):
    key = cache_key_for(file_path)
    cache_file = CACHE_DIR / key
    tmp_file = CACHE_DIR / f".{key}.tmp"
    tmp_file.write_text(result, encoding="utf-8")
    tmp_file.replace(cache_file)

_MEDIA_KEYS = ["photo", "video", "document", "audio", "voice_message",
               "sticker", "animation", "video_file", "audio_file"]

def get_media_info(msg: dict) -> Optional[Tuple[str, str]]:
    media_type = msg.get("media_type")
    file_path = msg.get("file")
    if media_type and file_path:
        return (media_type, file_path)
    for key in _MEDIA_KEYS:
        if key in msg:
            val = msg[key]
            if isinstance(val, str):
                return (key, val)
            if isinstance(val, dict) and "file" in val:
                return (key, val["file"])
    if file_path:
        return ("document", file_path)
    return None

def _media_worker_task(file_path_str: str, media_type: str, output_file: str):
    try:
        soft = WORKER_MEMORY_LIMIT_MB * 1024 * 1024
        hard = int(soft * 1.2)
        resource.setrlimit(resource.RLIMIT_AS, (soft, hard))
    except Exception:
        pass

    file_path = Path(file_path_str)
    ext = file_path.suffix.lower()
    audio_video_ext = {".mp3", ".wav", ".ogg", ".m4a", ".flac", ".opus",
                       ".mp4", ".webm", ".mkv", ".avi", ".mov"}
    image_ext = {".jpg", ".jpeg", ".png", ".webp", ".bmp", ".gif", ".tiff", ".tif"}

    wav_path = None
    txt_path = None
    tmp_wav_fd = None
    tmp_txt_fd = None

    try:
        if (media_type in ("audio", "voice_message", "video", "animation",
                           "video_file", "audio_file", "voice", "round_video")
                or ext in audio_video_ext):
            if not _ensure_audio_stream(file_path):
                Path(output_file).write_text("[нет аудиодорожки]", encoding="utf-8")
                return

            tmp_wav_fd = tempfile.NamedTemporaryFile(suffix=".wav", delete=False)
            wav_path = Path(tmp_wav_fd.name)
            tmp_wav_fd.close()

            if not _convert_to_wav(file_path, wav_path):
                Path(output_file).write_text("ERROR", encoding="utf-8")
                return

            tmp_txt_fd = tempfile.NamedTemporaryFile(suffix=".txt", delete=False)
            txt_path = Path(tmp_txt_fd.name)
            tmp_txt_fd.close()

            engine = "diarize" if (ENABLE_DIARIZATION and NEWTON_ENGINE in ("v3", "diarize")) else NEWTON_ENGINE
            cmd = ["newton", "transcribe", str(wav_path), "-o", str(txt_path), "-e", engine]
            if engine == "diarize" and DIARIZE_SPEAKERS is not None:
                cmd.extend(["-n", str(DIARIZE_SPEAKERS)])
            subprocess.run(cmd, check=True, capture_output=True, text=True, timeout=600)
            result = txt_path.read_text(encoding="utf-8").strip() if txt_path.exists() else "ERROR"
            Path(output_file).write_text(result, encoding="utf-8")

        elif media_type in ("photo", "sticker") or ext in image_ext:
            try:
                img = Image.open(file_path)
                text = pytesseract.image_to_string(img, lang=OCR_LANG).strip()
                img.close()
                Path(output_file).write_text(text if text else "[нет текста]", encoding="utf-8")
            except Exception:
                Path(output_file).write_text("ERROR", encoding="utf-8")

        else:
            try:
                if ext in {'.pptx', '.ppt'}:
                    text = _extract_text_from_pptx(file_path)
                elif ext in {'.docx', '.doc'}:
                    text = _extract_text_from_docx(file_path)
                elif ext == '.pdf':
                    text = _extract_text_from_pdf(file_path)
                elif ext in {'.xlsx', '.xls'}:
                    text = _extract_text_from_xlsx(file_path)
                else:
                    text = _extract_text_from_txt(file_path)
                Path(output_file).write_text(text if text else "[нет текста]", encoding="utf-8")
            except Exception:
                Path(output_file).write_text("ERROR", encoding="utf-8")
    except MemoryError:
        Path(output_file).write_text("ERROR", encoding="utf-8")
    except Exception:
        Path(output_file).write_text("ERROR", encoding="utf-8")
    finally:
        for path in [wav_path, txt_path]:
            if path is not None and path.exists():
                try:
                    path.unlink(missing_ok=True)
                except Exception:
                    pass
        for fd in [tmp_wav_fd, tmp_txt_fd]:
            if fd is not None:
                try:
                    fd.close()
                except Exception:
                    pass

def _ensure_audio_stream(file_path: Path) -> bool:
    try:
        res = subprocess.run(["ffprobe", "-v", "error", "-select_streams", "a:0",
                              "-show_entries", "stream=codec_type", "-of", "csv=p=0", str(file_path)],
                             capture_output=True, text=True, timeout=15)
        return "audio" in res.stdout.strip()
    except: return False

def _convert_to_wav(input_path: Path, output_path: Path) -> bool:
    try:
        subprocess.run(["ffmpeg", "-y", "-i", str(input_path), "-ac", "1", "-ar", "16000",
                        "-sample_fmt", "s16", str(output_path)],
                       check=True, capture_output=True, timeout=120)
        return output_path.exists()
    except: return False

def _extract_text_from_pptx(pptx_path: Path) -> Optional[str]:
    try:
        prs = Presentation(str(pptx_path))
        parts = []
        for i, slide in enumerate(prs.slides, 1):
            slide_text = []
            def extract_shape(shape):
                if shape.has_text_frame:
                    for para in shape.text_frame.paragraphs:
                        if para.text.strip(): slide_text.append(para.text.strip())
                if shape.has_table:
                    rows = ["| " + " | ".join(cell.text for cell in row.cells) + " |" for row in shape.table.rows]
                    if rows: slide_text.append("\n".join(rows))
                if shape.shape_type == MSO_SHAPE_TYPE.GROUP:
                    for child in shape.shapes: extract_shape(child)
            for shape in slide.shapes: extract_shape(shape)
            if slide.has_notes_slide and slide.notes_slide.notes_text_frame:
                notes = slide.notes_slide.notes_text_frame.text.strip()
                if notes: slide_text.append(f"[Заметки]: {notes}")
            if slide_text: parts.append(f"### Слайд {i}\n" + "\n".join(slide_text))
        return "\n\n".join(parts) if parts else None
    except:
        return None

def _extract_text_from_docx(docx_path: Path) -> Optional[str]:
    try:
        doc = Document(str(docx_path))
        parts = [p.text for p in doc.paragraphs if p.text.strip()]
        for table in doc.tables:
            rows = ["| " + " | ".join(cell.text for cell in row.cells) + " |" for row in table.rows]
            if rows: parts.append("\n".join(rows))
        return "\n\n".join(parts) if parts else None
    except:
        return None

def _extract_text_from_pdf(pdf_path: Path) -> Optional[str]:
    text = None
    try:
        reader = PdfReader(str(pdf_path))
        text = "\n".join(p.extract_text() or "" for p in reader.pages).strip()
        if text: return text
    except Exception:
        pass
    try:
        with pdfplumber.open(str(pdf_path)) as pdf:
            text = "\n".join(p.extract_text() or "" for p in pdf.pages).strip()
            if text: return text
    except Exception:
        pass

    if not text and ENABLE_OCR:
        with tempfile.TemporaryDirectory() as tmpdir:
            try:
                subprocess.run([
                    "pdftoppm", "-png", "-r", "100",
                    "-f", "1", "-l", str(MAX_OCR_PAGES),
                    str(pdf_path), f"{tmpdir}/page"
                ], capture_output=True, text=True, timeout=300)

                pages = []
                for i, img_file in enumerate(sorted(Path(tmpdir).glob("*.png"))):
                    if i >= MAX_OCR_PAGES:
                        break
                    with Image.open(img_file) as img:
                        img.load()
                        t = pytesseract.image_to_string(img, lang=OCR_LANG).strip()
                    img_file.unlink(missing_ok=True)  # удаляем временный png немедленно
                    if t:
                        pages.append(t)
                    if i % 5 == 0:
                        gc.collect()
                text = "\n".join(pages) if pages else None
            except Exception as e:
                logger.warning(f"OCR PDF failed for {pdf_path.name}: {e}")
    return text if text else "[нет текста]"

def _extract_text_from_xlsx(xlsx_path: Path) -> Optional[str]:
    try:
        wb = load_workbook(str(xlsx_path), read_only=True, data_only=True)
        parts = []
        for name in wb.sheetnames:
            ws = wb[name]
            rows = list(ws.iter_rows(max_row=100, values_only=True))
            if not rows: continue
            header = rows[0]
            parts.append(f"### Лист: {name}")
            parts.append("| " + " | ".join(str(c) if c is not None else "" for c in header) + " |")
            parts.append("| " + " | ".join(["---"] * len(header)) + " |")
            for row in rows[1:]:
                parts.append("| " + " | ".join(str(c) if c is not None else "" for c in row) + " |")
            parts.append("")
        return "\n".join(parts) if parts else None
    except:
        return None

def _extract_text_from_txt(file_path: Path) -> Optional[str]:
    for enc in ["utf-8", "cp1251", "latin-1"]:
        try:
            return file_path.read_text(encoding=enc).strip()
        except:
            continue
    return None

#  ОСНОВНОЙ ЦИКЛ: ПОТОКОВАЯ ОБРАБОТКА ZIP БЕЗ ПОЛНОЙ РАСПАКОВКИ

def process_zip_direct(zip_path: Path, archive_name: str) -> Path:
    out_csv = Path(OUTPUT_DIR) / f"{archive_name}.csv"
    out_csv.parent.mkdir(parents=True, exist_ok=True)

    with zipfile.ZipFile(zip_path, 'r') as zf:
        # 1. Находим result.json
        json_names = [n for n in zf.namelist() if n.endswith('result.json')]
        if not json_names:
            raise FileNotFoundError("result.json не найден в архиве")
        json_member = json_names[0]

        # 2. Лёгкий индекс: имя файла → полный путь в ZIP
        zip_index = defaultdict(list)
        for name in zf.namelist():
            if not name.endswith('/'):   # не директория
                zip_index[Path(name).name].append(name)
        logger.info(f"Индекс ZIP: {len(zip_index)} уникальных имён файлов")

        # 3. Извлекаем только result.json во временную папку
        with tempfile.TemporaryDirectory() as tmpdir:
            zf.extract(json_member, tmpdir)
            json_path = Path(tmpdir) / json_member

            with open(json_path, 'rb') as f_json, open(out_csv, 'w', newline='', encoding='utf-8') as csvfile:
                writer = csv.DictWriter(csvfile, fieldnames=['timestamp', 'from', 'content', 'inter_block_gap_seconds'])
                writer.writeheader()

                current_block = None
                prev_block_last_dt = None
                msg_count = 0

                pbar = tqdm(desc="Обработка сообщений", unit="сообщ.", ncols=150,
                            bar_format='{l_bar}{bar}| {n_fmt}/{total_fmt} [{elapsed}<{remaining}]')
                try:
                    for msg in ijson.items(f_json, 'messages.item'):
                        if msg.get('type') != 'message':
                            continue
                        msg_count += 1
                        if msg_count < 100:
                            pbar.update(1)
                        elif msg_count % 100 == 0:
                            pbar.update(100)

                        date_unixtime = safe_int(msg.get('date_unixtime', 0))
                        dt = datetime.fromtimestamp(date_unixtime, tz=timezone.utc)
                        from_name = msg.get('from', 'Unknown')
                        text_raw = msg.get('text', '')
                        if isinstance(text_raw, list):
                            text = ''.join(item if isinstance(item, str) else item.get('text', '') for item in text_raw)
                        else:
                            text = str(text_raw) if text_raw else ''

                        media_info = get_media_info(msg)
                        media_desc = ''
                        if media_info:
                            media_type, rel_path = media_info
                            # Поиск в ZIP по имени файла
                            media_name = Path(unquote(rel_path)).name
                            candidates = zip_index.get(media_name, [])
                            if candidates:
                                # При коллизии берём первый (сортируем для детерминизма)
                                zip_member = sorted(candidates)[0]
                                if len(candidates) > 1:
                                    logger.warning(f"Коллизия в ZIP для '{media_name}': {len(candidates)} вариантов, выбран {zip_member}")

                                # Извлекаем медиафайл во временную папку
                                with tempfile.TemporaryDirectory() as media_tmp:
                                    zf.extract(zip_member, media_tmp)
                                    file_path = Path(media_tmp) / zip_member

                                    # Проверка доступной памяти перед обработкой
                                    file_size = file_path.stat().st_size
                                    avail_mem = psutil.virtual_memory().available
                                    if file_size > avail_mem * SAFE_MEMORY_RATIO:
                                        media_desc = f"(файл пропущен – недостаточно памяти: {file_size/1024/1024:.1f} МБ)"
                                        logger.warning(f"Пропущен большой файл {media_name} ({file_size/1024/1024:.1f} МБ)")
                                    else:
                                        cached = load_from_cache(file_path)
                                        if cached is not None:
                                            media_desc = cached
                                        else:
                                            logger.info(f"Обработка: {media_name}")
                                            pbar.set_postfix_str(f"Файл: {media_name[:30]}")
                                            with tempfile.NamedTemporaryFile(suffix=".txt", delete=False) as res_tmp:
                                                output_file = res_tmp.name
                                            try:
                                                p = multiprocessing.Process(
                                                    target=_media_worker_task,
                                                    args=(str(file_path), media_type, output_file)
                                                )
                                                p.start()
                                                p.join(timeout=120)
                                                if p.is_alive():
                                                    p.terminate()
                                                    p.join()
                                                    media_desc = "(ошибка обработки - таймаут)"
                                                    logger.warning(f"Таймаут: {media_name}")
                                                else:
                                                    with open(output_file, "r", encoding="utf-8") as res_f:
                                                        res = res_f.read().strip()
                                                    if res == "ERROR":
                                                        media_desc = "(ошибка обработки)"
                                                        logger.error(f"Ошибка обработки: {media_name}")
                                                    else:
                                                        media_desc = res
                                                        save_to_cache(file_path, media_desc)
                                            finally:
                                                try:
                                                    os.remove(output_file)
                                                except Exception:
                                                    pass
                            else:
                                media_desc = "(файл отсутствует)"

                        content = (text + ' ' + media_desc).strip()
                        if not content:
                            continue

                        # Запись блока
                        if current_block and current_block['author'] == from_name and \
                           (dt - current_block['last_dt']) <= timedelta(minutes=MERGE_WINDOW_MINUTES):
                            current_block['content'] += '\n' + content
                            current_block['last_dt'] = dt
                        else:
                            if current_block:
                                gap = 0.0
                                if prev_block_last_dt is not None:
                                    gap = (current_block['first_dt'] - prev_block_last_dt).total_seconds()
                                writer.writerow({
                                    'timestamp': current_block['last_dt'].strftime("%d.%m.%Y %H:%M:%S"),
                                    'from': current_block['author'],
                                    'content': current_block['content'],
                                    'inter_block_gap_seconds': gap
                                })
                                prev_block_last_dt = current_block['last_dt']
                            current_block = {
                                'author': from_name,
                                'first_dt': dt,
                                'last_dt': dt,
                                'content': content
                            }

                        if msg_count % 50 == 0:
                            gc.collect()
                            if psutil.virtual_memory().percent > 85:
                                logger.warning("Высокое использование памяти, пауза 2 сек")
                                time.sleep(2)

                    if current_block:
                        gap = 0.0
                        if prev_block_last_dt is not None:
                            gap = (current_block['first_dt'] - prev_block_last_dt).total_seconds()
                        writer.writerow({
                            'timestamp': current_block['last_dt'].strftime("%d.%m.%Y %H:%M:%S"),
                            'from': current_block['author'],
                            'content': current_block['content'],
                            'inter_block_gap_seconds': gap
                        })
                finally:
                    pbar.n = msg_count
                    pbar.refresh()
                    pbar.close()

    logger.info(f"Обработано сообщений: {msg_count}")
    logger.info(f"Результат сохранён: {out_csv} ({out_csv.stat().st_size/1024/1024:.2f} MB)")
    return out_csv

def safe_int(value, default=0):
    try: return int(float(value))
    except: return default

# ============================================================
def main():
    source = get_source_choice()
    zip_path, archive_name = obtain_zip_path(source)
    logger.info(f"Архив: {zip_path}")

    # Обработка напрямую из ZIP, без полной распаковки

if __name__ == "__main__":
    main()

## 🔍 Почему v7 не упадёт на 12.6 ГБ

*   **Архив не распаковывается целиком** – извлекается только `result.json` (несколько сотен МБ) и по одному медиафайлу во временную папку. Это экономит десятки гигабайт дискового пространства и RAM.
*   **Лёгкий индекс ZIP** – словарь `имя_файла -> путь_в_архиве` строится по строкам из `namelist()`, а не по объектам `Path` файловой системы. Он занимает мегабайты.
*   **Динамическая проверка памяти** – перед обработкой файла сравнивается его размер со свободной RAM, и слишком большие файлы пропускаются.
*   **Сниженный DPI и лимит OCR** – `-r 100` вместо 150, `MAX_OCR_PAGES=20`, PNG удаляются сразу после распознавания.
*   **Принудительная пауза при высоком потреблении** – каждые 50 сообщений проверяется `psutil.virtual_memory().percent`, и при превышении 85% делается пауза на 2 секунды с агрессивной сборкой мусора.
*   **Прогресс-бар остаётся крупным и информативным**.

Запустите эту ячейку – она обработает даже экстремально большой архив, не выходя за пределы доступной памяти.
