Работает в Google Colab с загрузкой файлов с локального компьютера
"""

import os
import glob
from pydub import AudioSegment
from pydub.utils import make_chunks
import math
from google.colab import files

def split_audio_by_duration(input_file, chunk_length_ms=30*60*1000):
    """
    Разбивает аудиофайл на отрезки заданной длительности

    Args:
        input_file (str): Путь к входному аудиофайлу
        chunk_length_ms (int): Длина отрезка в миллисекундах (по умолчанию 30 минут)

    Returns:
        list: Список путей к созданным файлам
    """
    print(f"🎵 Обрабатываю файл: {input_file}")

    # Загружаем аудиофайл
    try:
        audio = AudioSegment.from_file(input_file)
    except Exception as e:
        print(f"❌ Ошибка загрузки файла: {e}")
        return []

    # Получаем информацию о файле
    duration_ms = len(audio)
    duration_min = math.ceil(duration_ms / (60 * 1000))
    print(f"📊 Длительность аудио: {duration_min} минут")

    # Рассчитываем количество отрезков
    num_chunks = math.ceil(duration_ms / chunk_length_ms)
    print(f"🔢 Будет создано {num_chunks} отрезков по 30 минут")

    # Создаем папку для результатов
    base_name = os.path.splitext(os.path.basename(input_file))[0]
    output_dir = f"{base_name}_chunks"
    os.makedirs(output_dir, exist_ok=True)

    # Разбиваем на отрезки
    chunks = make_chunks(audio, chunk_length_ms)
    output_files = []

    for i, chunk in enumerate(chunks):
        # Форматируем номер отрезка
        chunk_number = i + 1
        chunk_name = f"{base_name}_part{chunk_number:02d}.mp3"
        output_path = os.path.join(output_dir, chunk_name)

        # Экспортируем отрезок
        chunk.export(output_path, format="mp3", bitrate="192k")
        output_files.append(output_path)

        chunk_duration = len(chunk) / (60 * 1000)
        print(f"✅ Создан отрезок {chunk_number}: {chunk_name} ({chunk_duration:.1f} мин)")

    return output_files
