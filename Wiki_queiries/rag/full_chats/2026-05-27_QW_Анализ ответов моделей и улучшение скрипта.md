def download_from_yandex(public_url: str, dest_dir: Path) -> Path:
    try:
        api_url = "https://cloud-api.yandex.net/v1/disk/public/resources/download"
        resp = requests.get(api_url, params={"public_key": public_url})
        resp.raise_for_status()
        data = resp.json()
        download_url = data["href"]
        file_name = data.get("name") or Path(unquote(urlparse(download_url).path)).name
        dest_path = dest_dir / file_name
        logger.info(f"Скачиваем {file_name} с Яндекс.Диска...")
        with requests.get(download_url, stream=True) as r:
            r.raise_for_status()
            with open(dest_path, "wb") as f:
                for chunk in r.iter_content(chunk_size=8192):
                    f.write(chunk)
        logger.info(f"Файл скачан: {dest_path.name}")
        return dest_path
    except Exception as e:
        logger.error(f"Ошибка загрузки из Яндекс.Диска: {e}")
        raise

def download_from_mailru(public_url: str, dest_dir: Path) -> Path:
    try:
        from mailru_cloud_guest_api import FileStreamGenerator

        generator = FileStreamGenerator.of(public_url)
        container = generator.generate()
        parsed = urlparse(public_url)
        file_name = Path(unquote(parsed.path)).name or "mailru_download"
        dest_path = dest_dir / file_name
        logger.info(f"Скачиваем {file_name} из Облака Mail.ru...")
        container.download(str(dest_path))
        logger.info(f"Файл скачан: {dest_path.name}")
        return dest_path
    except Exception as e:
        logger.error(f"Ошибка загрузки из Облака Mail.ru: {e}")
        raise

def obtain_zip_path(source_choice: str) -> Tuple[Path, str]:
    if source_choice == "local":
        if IN_COLAB:
            print("📎 Загрузите ZIP-архив экспорта Telegram")
            uploaded = colab_files.upload()
            if not uploaded:
                raise RuntimeError("Файл не загружен")
            zip_name = next(iter(uploaded))
            zip_path = UPLOAD_DIR / zip_name
            shutil.move(zip_name, str(zip_path))
            return zip_path, Path(zip_name).stem
        else:
            path_str = input("Введите путь к ZIP-файлу: ").strip()
            local_path = Path(path_str).expanduser().resolve()
            if not local_path.exists() or local_path.suffix.lower() != ".zip":
                raise FileNotFoundError("Указанный файл не является ZIP-архивом")
            return local_path, local_path.stem
    elif source_choice == "yandex":
        public_url = input("🔗 Введите публичную ссылку Яндекс.Диска: ").strip()
        if not (public_url.startswith("https://disk.yandex.") or public_url.startswith("https://yadi.sk")):
            raise ValueError("Некорректная ссылка Яндекс.Диска")
        zip_path = download_from_yandex(public_url, UPLOAD_DIR)
        return zip_path, zip_path.stem
    elif source_choice == "mailru":
        public_url = input("🔗 Введите публичную ссылку Облака Mail.ru: ").strip()
        if not public_url.startswith("https://cloud.mail.ru/"):
            raise ValueError("Некорректная ссылка Облака Mail.ru")
        zip_path = download_from_mailru(public_url, UPLOAD_DIR)
        return zip_path, zip_path.stem
    else:
        raise ValueError(f"Неизвестный источник: {source_choice}")

def cache_key_for_zip_member(zip_member: str, file_size: int, file_mtime: float) -> str:
    """Генерирует ключ кэша, не зависящий от временного пути извлечения."""
    raw = f"{zip_member}:{file_mtime}:{file_size}"
    return hashlib.md5(raw.encode()).hexdigest()

def load_from_cache_by_key(cache_key: str) -> Optional[str]:
    cache_file = CACHE_DIR / cache_key
    if cache_file.exists():
        return cache_file.read_text(encoding="utf-8")
    return None

def save_to_cache_by_key(cache_key: str, result: str):

    tmp_file = CACHE_DIR / f".{cache_key}.tmp"
    tmp_file.write_text(result, encoding="utf-8")
    tmp_file.replace(cache_file)
    logger.debug(f"Сохранено в кэш: {cache_key}")

_MEDIA_KEYS = ["photo", "video", "document", "audio", "voice_message",
               "sticker", "animation", "video_file", "audio_file"]

def get_media_info(msg: dict) -> Optional[Tuple[str, str]]:
    media_type = msg.get("media_type")
    file_path = msg.get("file")
    if media_type and file_path:
        return (media_type, file_path)
    for key in _MEDIA_KEYS:
        if key in msg:
            val = msg[key]
            if isinstance(val, str):
                return (key, val)
            if isinstance(val, dict) and "file" in val:
                return (key, val["file"])
    if file_path:
        return ("document", file_path)
    return None

def _media_worker_task(file_path_str: str, media_type: str, output_file: str):
    try:
        soft = WORKER_MEMORY_LIMIT_MB * 1024 * 1024
        hard = int(soft * 1.2)
        resource.setrlimit(resource.RLIMIT_AS, (soft, hard))
    except Exception:
        pass

    file_path = Path(file_path_str)
    ext = file_path.suffix.lower()
    audio_video_ext = {".mp3", ".wav", ".ogg", ".m4a", ".flac", ".opus",
                       ".mp4", ".webm", ".mkv", ".avi", ".mov"}
    image_ext = {".jpg", ".jpeg", ".png", ".webp", ".bmp", ".gif", ".tiff", ".tif"}

    wav_path = None
    txt_path = None
    tmp_wav_fd = None
    tmp_txt_fd = None

    try:
        if (media_type in ("audio", "voice_message", "video", "animation",
                           "video_file", "audio_file", "voice", "round_video")
                or ext in audio_video_ext):
            if not _ensure_audio_stream(file_path):
                Path(output_file).write_text("[нет аудиодорожки]", encoding="utf-8")
                return

            tmp_wav_fd = tempfile.NamedTemporaryFile(suffix=".wav", delete=False)
            wav_path = Path(tmp_wav_fd.name)
            tmp_wav_fd.close()

            if not _convert_to_wav(file_path, wav_path):
                Path(output_file).write_text("ERROR", encoding="utf-8")
                return

            tmp_txt_fd = tempfile.NamedTemporaryFile(suffix=".txt", delete=False)
            txt_path = Path(tmp_txt_fd.name)
            tmp_txt_fd.close()

            engine = "diarize" if (ENABLE_DIARIZATION and NEWTON_ENGINE in ("v3", "diarize")) else NEWTON_ENGINE
            cmd = ["newton", "transcribe", str(wav_path), "-o", str(txt_path), "-e", engine]
            if engine == "diarize" and DIARIZE_SPEAKERS is not None:
                cmd.extend(["-n", str(DIARIZE_SPEAKERS)])
            subprocess.run(cmd, check=True, capture_output=True, text=True, timeout=600)
            result = txt_path.read_text(encoding="utf-8").strip() if txt_path.exists() else "ERROR"
            Path(output_file).write_text(result, encoding="utf-8")

        elif media_type in ("photo", "sticker") or ext in image_ext:
            try:
                img = Image.open(file_path)
                text = pytesseract.image_to_string(img, lang=OCR_LANG).strip()
                img.close()
                Path(output_file).write_text(text if text else "[нет текста]", encoding="utf-8")
            except Exception:
                Path(output_file).write_text("ERROR", encoding="utf-8")

        else:
            try:
                if ext in {'.pptx', '.ppt'}:
                    text = _extract_text_from_pptx(file_path)
                elif ext in {'.docx', '.doc'}:
                    text = _extract_text_from_docx(file_path)
                elif ext == '.pdf':
                    text = _extract_text_from_pdf(file_path)
                elif ext in {'.xlsx', '.xls'}:
                    text = _extract_text_from_xlsx(file_path)
                else:
                    text = _extract_text_from_txt(file_path)
                Path(output_file).write_text(text if text else "[нет текста]", encoding="utf-8")
            except Exception:
                Path(output_file).write_text("ERROR", encoding="utf-8")
    except MemoryError:
        Path(output_file).write_text("ERROR", encoding="utf-8")
    except Exception:
        Path(output_file).write_text("ERROR", encoding="utf-8")
    finally:
        for path in [wav_path, txt_path]:
            if path is not None and path.exists():
                try:
                    path.unlink(missing_ok=True)
                except Exception:
                    pass
        for fd in [tmp_wav_fd, tmp_txt_fd]:
            if fd is not None:
                try:
                    fd.close()
                except Exception:
                    pass

def _ensure_audio_stream(file_path: Path) -> bool:
    try:
        res = subprocess.run(["ffprobe", "-v", "error", "-select_streams", "a:0",
                              "-show_entries", "stream=codec_type", "-of", "csv=p=0", str(file_path)],
                             capture_output=True, text=True, timeout=15)
        return "audio" in res.stdout.strip()
    except: return False

def _convert_to_wav(input_path: Path, output_path: Path) -> bool:
    try:
        subprocess.run(["ffmpeg", "-y", "-i", str(input_path), "-ac", "1", "-ar", "16000",
                        "-sample_fmt", "s16", str(output_path)],
                       check=True, capture_output=True, timeout=120)
        return output_path.exists()
    except: return False

def _extract_text_from_pptx(pptx_path: Path) -> Optional[str]:
    try:
        prs = Presentation(str(pptx_path))
        parts = []
        for i, slide in enumerate(prs.slides, 1):
            slide_text = []
            def extract_shape(shape):
                if shape.has_text_frame:
                    for para in shape.text_frame.paragraphs:
                        if para.text.strip(): slide_text.append(para.text.strip())
                if shape.has_table:
                    rows = ["| " + " | ".join(cell.text for cell in row.cells) + " |" for row in shape.table.rows]
                    if rows: slide_text.append("\n".join(rows))
                if shape.shape_type == MSO_SHAPE_TYPE.GROUP:
                    for child in shape.shapes: extract_shape(child)
            for shape in slide.shapes: extract_shape(shape)
            if slide.has_notes_slide and slide.notes_slide.notes_text_frame:
                notes = slide.notes_slide.notes_text_frame.text.strip()
                if notes: slide_text.append(f"[Заметки]: {notes}")
            if slide_text: parts.append(f"### Слайд {i}\n" + "\n".join(slide_text))
        return "\n\n".join(parts) if parts else None
    except:
        return None

def _extract_text_from_docx(docx_path: Path) -> Optional[str]:
    try:
        doc = Document(str(docx_path))
        parts = [p.text for p in doc.paragraphs if p.text.strip()]
        for table in doc.tables:
            rows = ["| " + " | ".join(cell.text for cell in row.cells) + " |" for row in table.rows]
            if rows: parts.append("\n".join(rows))
        return "\n\n".join(parts) if parts else None
    except:
        return None

def _extract_text_from_pdf(pdf_path: Path) -> Optional[str]:
    text = None
    try:
        reader = PdfReader(str(pdf_path))
        text = "\n".join(p.extract_text() or "" for p in reader.pages).strip()
        if text: return text
    except Exception:
        pass
    try:
        with pdfplumber.open(str(pdf_path)) as pdf:
            text = "\n".join(p.extract_text() or "" for p in pdf.pages).strip()
            if text: return text
    except Exception:
        pass

    if not text and ENABLE_OCR:
        with tempfile.TemporaryDirectory() as tmpdir:
            try:
                subprocess.run([
                    "pdftoppm", "-png", "-r", "100",
                    "-f", "1", "-l", str(MAX_OCR_PAGES),
                    str(pdf_path), f"{tmpdir}/page"
                ], capture_output=True, text=True, timeout=300)

                pages = []
                for i, img_file in enumerate(sorted(Path(tmpdir).glob("*.png"))):
                    if i >= MAX_OCR_PAGES:
                        break
                    with Image.open(img_file) as img:
                        img.load()
                        t = pytesseract.image_to_string(img, lang=OCR_LANG).strip()
                    img_file.unlink(missing_ok=True)
                    if t:
                        pages.append(t)
                    if i % 5 == 0:
                        gc.collect()
                text = "\n".join(pages) if pages else None
            except Exception as e:
                logger.warning(f"OCR PDF failed for {pdf_path.name}: {e}")
    return text if text else "[нет текста]"

def _extract_text_from_xlsx(xlsx_path: Path) -> Optional[str]:
    try:
        wb = load_workbook(str(xlsx_path), read_only=True, data_only=True)
        parts = []
        for name in wb.sheetnames:
            ws = wb[name]
            rows = list(ws.iter_rows(max_row=100, values_only=True))
            if not rows: continue
            header = rows[0]
            parts.append(f"### Лист: {name}")
            parts.append("| " + " | ".join(str(c) if c is not None else "" for c in header) + " |")
            parts.append("| " + " | ".join(["---"] * len(header)) + " |")
            for row in rows[1:]:
                parts.append("| " + " | ".join(str(c) if c is not None else "" for c in row) + " |")
            parts.append("")
        return "\n".join(parts) if parts else None
    except:
        return None

def _extract_text_from_txt(file_path: Path) -> Optional[str]:
    for enc in ["utf-8", "cp1251", "latin-1"]:
        try:
            return file_path.read_text(encoding=enc).strip()
        except:
            continue
    return None

def process_zip_direct(zip_path: Path, archive_name: str) -> Path:
    out_csv = Path(OUTPUT_DIR) / f"{archive_name}.csv"
    out_csv.parent.mkdir(parents=True, exist_ok=True)

    with zipfile.ZipFile(zip_path, 'r') as zf:
        # 1. Находим result.json
        json_names = [n for n in zf.namelist() if n.endswith('result.json')]
        if not json_names:
            raise FileNotFoundError("result.json не найден в архиве")
        json_member = json_names[0]

        # 2. Лёгкий индекс: имя файла → список полных путей в ZIP
        zip_index = defaultdict(list)
        for name in zf.namelist():
            if not name.endswith('/'):
                zip_index[Path(name).name].append(name)
        logger.info(f"Индекс ZIP: {len(zip_index)} уникальных имён файлов")

        # 3. Извлекаем только result.json во временную папку
        with tempfile.TemporaryDirectory() as tmpdir:
            zf.extract(json_member, tmpdir)
            json_path = Path(tmpdir) / json_member

            with open(json_path, 'rb') as f_json, open(out_csv, 'w', newline='', encoding='utf-8') as csvfile:
                writer = csv.DictWriter(csvfile, fieldnames=['timestamp', 'from', 'content', 'inter_block_gap_seconds'])
                writer.writeheader()

                current_block = None
                prev_block_last_dt = None
                msg_count = 0

                pbar = tqdm(desc="Обработка сообщений", unit="сообщ.", ncols=150,
                            bar_format='{l_bar}{bar}| {n_fmt}/{total_fmt} [{elapsed}<{remaining}]')
                try:
                    for msg in ijson.items(f_json, 'messages.item'):
                        if msg.get('type') != 'message':
                            continue
                        msg_count += 1
                        if msg_count < 100:
                            pbar.update(1)
                        elif msg_count % 100 == 0:
                            pbar.update(100)

                        date_unixtime = safe_int(msg.get('date_unixtime', 0))
                        dt = datetime.fromtimestamp(date_unixtime, tz=timezone.utc)
                        from_name = msg.get('from', 'Unknown')
                        text_raw = msg.get('text', '')
                        if isinstance(text_raw, list):
                            text = ''.join(item if isinstance(item, str) else item.get('text', '') for item in text_raw)
                        else:
                            text = str(text_raw) if text_raw else ''

                        media_info = get_media_info(msg)
                        media_desc = ''
                        if media_info:
                            media_type, rel_path = media_info
                            media_name = Path(unquote(rel_path)).name
                            candidates = zip_index.get(media_name, [])
                            if candidates:
                                zip_member = sorted(candidates)[0]
                                if len(candidates) > 1:
                                    logger.warning(f"Коллизия в ZIP для '{media_name}': {len(candidates)} вариантов, выбран {zip_member}")

                                # Извлекаем медиафайл во временную папку
                                with tempfile.TemporaryDirectory() as media_tmp:
                                    zf.extract(zip_member, media_tmp)
                                    file_path = Path(media_tmp) / zip_member

                                    # Проверка доступной памяти
                                    file_size = file_path.stat().st_size
                                    avail_mem = psutil.virtual_memory().available
                                    if file_size > avail_mem * SAFE_MEMORY_RATIO:
                                        media_desc = f"(файл пропущен – недостаточно памяти: {file_size/1024/1024:.1f} МБ)"
                                        logger.warning(f"Пропущен большой файл {media_name} ({file_size/1024/1024:.1f} МБ)")
                                    else:
                                        # Ключ кэша на основе ZIP-пути и метаданных
                                        cache_key = cache_key_for_zip_member(zip_member, file_size, file_path.stat().st_mtime)
                                        cached = load_from_cache_by_key(cache_key)
                                        if cached is not None:
                                            media_desc = cached
                                        else:
                                            logger.info(f"Обработка: {media_name}")
                                            pbar.set_postfix_str(f"Файл: {media_name[:30]}")
                                            with tempfile.NamedTemporaryFile(suffix=".txt", delete=False) as res_tmp:
                                                output_file = res_tmp.name
                                            try:
                                                p = multiprocessing.Process(
                                                    target=_media_worker_task,
                                                    args=(str(file_path), media_type, output_file)
                                                )
                                                p.start()
                                                p.join(timeout=120)
                                                if p.is_alive():
                                                    p.terminate()
                                                    p.join()
                                                    media_desc = "(ошибка обработки - таймаут)"
                                                    logger.warning(f"Таймаут: {media_name}")
                                                else:
                                                    with open(output_file, "r", encoding="utf-8") as res_f:
                                                        res = res_f.read().strip()
                                                    if res == "ERROR":
                                                        media_desc = "(ошибка обработки)"
                                                        logger.error(f"Ошибка обработки: {media_name}")
                                                    else:
                                                        media_desc = res
                                                        save_to_cache_by_key(cache_key, media_desc)
                                            finally:
                                                try:
                                                    os.remove(output_file)
                                                except Exception:
                                                    pass
                            else:
                                media_desc = "(файл отсутствует)"

### 2. Консенсусы (что признали все или почти все модели)

- **Коллизии имён файлов** – использование `Path(name).name` приводит к выбору случайного файла при дублировании имён. Нужно сопоставлять по полному пути.
- **Ненадёжный ключ кэша** – `mtime` временного файла может меняться. Лучше использовать `CRC32` и `file_size` из `ZipInfo`.
- **Проблемы с `multiprocessing.Process`** – запуск процесса на каждый файл создаёт оверхед, а `p.terminate()` не гарантирует освобождение ресурсов. Нужен пул процессов (`ProcessPoolExecutor`).
- **Слепые `except:`** – скрывают реальные ошибки, затрудняют отладку. Требуется логирование с `exc_info=True`.
- **Жёсткие таймауты** – 120 секунд для `ffmpeg` и 600 для `newton` могут быть недостаточны для больших файлов. Нужны динамические таймауты.
- **Утечки временных файлов** – использование `NamedTemporaryFile(delete=False)` без гарантированного удаления в `finally` может засорять диск. Лучше `TemporaryDirectory` или контекстный менеджер.
- **Нет точки сохранения прогресса** – при сбое обработка начинается с нуля. Нужны чекпоинты.
- **Неограниченный рост кэша** – требуется ротация (LRU или ограничение размера/времени).
- **`setrlimit` не работает в Colab/контейнерах** – его использование бесполезно, лучше мониторить память через `psutil` и принудительно завершать воркер.
- **Нет проверки целостности Newton CLI** – скачивание по raw-ссылке без проверки хеша опасно. Нужно валидировать SHA256.
- **Парсинг текста сообщений** – `msg.get('text')` может быть списком или словарём, требуется рекурсивный обход.

def safe_int(value, default=0) -> int:
    try:
        if value is None or value == "": return default
        return int(float(value))
    except (ValueError, TypeError):
        return default

def extract_tg_text(node: Any) -> str:
    if isinstance(node, str): return node
    if isinstance(node, list): return "".join(extract_tg_text(n) for n in node)
    if isinstance(node, dict): return node.get("text", "")
    return ""

def get_cache_key(zinfo: zipfile.ZipInfo) -> str:
    raw = f"{zinfo.filename}:{zinfo.file_size}:{zinfo.CRC}"
    return hashlib.md5(raw.encode()).hexdigest()

def load_cache(key: str) -> Optional[str]:
    f = CFG.cache_dir / key
    if f.exists():
        try: f.touch()  # обновить mtime для LRU
        except: pass
        return f.read_text(encoding="utf-8")
    return None

def save_cache(key: str, text: str):
    (CFG.cache_dir / key).write_text(text, encoding="utf-8")
    evict_cache_if_needed()

def evict_cache_if_needed():
    """LRU eviction при превышении cache_max_gb."""
    try:
        total = sum(f.stat().st_size for f in CFG.cache_dir.iterdir() if f.is_file())
        limit = CFG.cache_max_gb * 1024**3
        if total <= limit: return
        files = sorted(CFG.cache_dir.iterdir(), key=lambda f: f.stat().st_mtime)
        for f in files:
            if total <= limit * 0.8: break
            try:
                sz = f.stat().st_size
                f.unlink()
                total -= sz
            except: pass
    except Exception as e:
        logger.warning(f"Cache eviction failed: {e}")

def validate_newton_token() -> bool:
    try:
        r = subprocess.run(["newton", "health"], capture_output=True, text=True, timeout=15)
        ok = r.returncode == 0
        if not ok: logger.warning(f"Newton health check failed: {r.stderr}")
        return ok
    except Exception as e:
        logger.warning(f"Newton health check error: {e}")
        return False

@retry(stop=stop_after_attempt(3), wait=wait_exponential(multiplier=1, min=2, max=10))
def download_from_yandex(public_url: str, dest_dir: Path) -> Path:
    resp = requests.get("https://cloud-api.yandex.net/v1/disk/public/resources/download", params={"public_key": public_url})
    resp.raise_for_status()
    data = resp.json()
    download_url = data["href"]
    file_name = data.get("name") or Path(unquote(urlparse(download_url).path)).name
    dest_path = dest_dir / file_name
    with requests.get(download_url, stream=True) as r:
        r.raise_for_status()
        with open(dest_path, "wb") as f:
            for chunk in r.iter_content(chunk_size=8192): f.write(chunk)
    return dest_path

@retry(stop=stop_after_attempt(3), wait=wait_exponential(multiplier=1, min=2, max=10))
def download_from_mailru(public_url: str, dest_dir: Path) -> Path:
    from mailru_cloud_guest_api import FileStreamGenerator
    container = FileStreamGenerator.of(public_url).generate()
    file_name = Path(unquote(urlparse(public_url).path)).name or "mailru_download"
    dest_path = dest_dir / file_name
    container.download(str(dest_path))
    return dest_path

def parse_pptx(path: Path) -> str:
    prs = Presentation(str(path))
    parts = []
    for i, slide in enumerate(prs.slides, 1):
        slide_text: List[str] = []
        def walk(shape):
            if shape.has_text_frame:
                for p in shape.text_frame.paragraphs:
                    if p.text.strip(): slide_text.append(p.text.strip())
            if shape.has_table:
                rows = ["| " + " | ".join(c.text for c in row.cells) + " |" for row in shape.table.rows]
                if rows: slide_text.append("\n".join(rows))
            if shape.shape_type == MSO_SHAPE_TYPE.GROUP:
                for ch in shape.shapes: walk(ch)
        for shape in slide.shapes: walk(shape)
        if slide.has_notes_slide and slide.notes_slide.notes_text_frame:
            n = slide.notes_slide.notes_text_frame.text.strip()
            if n: slide_text.append(f"[Заметки]: {n}")
        if slide_text: parts.append(f"### Слайд {i}\n" + "\n".join(slide_text))
    return "\n\n".join(parts)

def parse_docx(path: Path) -> str:
    doc = Document(str(path))
    parts = [p.text for p in doc.paragraphs if p.text.strip()]
    for t in doc.tables:
        rows = ["| " + " | ".join(c.text for c in r.cells) + " |" for r in t.rows]
        if rows: parts.append("\n".join(rows))
    return "\n\n".join(parts)

def parse_xlsx(path: Path) -> str:
    wb = load_workbook(str(path), read_only=True, data_only=True)
    parts = []
    for name in wb.sheetnames:
        ws = wb[name]
        rows = list(ws.iter_rows(max_row=200, values_only=True))
        if not rows: continue
        hdr = rows[0]
        parts.append(f"### Лист: {name}")
        parts.append("| " + " | ".join(str(c) if c is not None else "" for c in hdr) + " |")
        parts.append("| " + " | ".join(["---"] * len(hdr)) + " |")
        for r in rows[1:]:
            parts.append("| " + " | ".join(str(c) if c is not None else "" for c in r) + " |")
    return "\n".join(parts)

def parse_pdf(path: Path, cfg: dict) -> str:
    text = ""
    try:
        with pdfplumber.open(str(path)) as pdf:
            text = "\n".join(p.extract_text() or "" for p in pdf.pages).strip()
    except Exception as e:
        logger.debug(f"pdfplumber failed: {e}")
    if text: return text
    if not cfg["enable_ocr"]: return ""
    with tempfile.TemporaryDirectory() as tmp:
        try:
            subprocess.run(["pdftoppm", "-png", "-r", "150", "-f", "1", "-l", str(cfg["max_ocr_pages"]),
                            str(path), f"{tmp}/page"], capture_output=True, timeout=300)
            pages = []
            for img_f in sorted(Path(tmp).glob("*.png")):
                with Image.open(img_f) as img:
                    t = pytesseract.image_to_string(img, lang=cfg["ocr_lang"]).strip()
                    if t: pages.append(t)
            return "\n".join(pages)
        except Exception as e:
            logger.warning(f"OCR PDF failed: {e}")
            return ""

@retry(stop=stop_after_attempt(2), wait=wait_exponential(multiplier=1, min=2, max=10),
       retry=retry_if_exception_type(subprocess.TimeoutExpired))
def _run_newton(wav_path: Path, out_path: Path, cfg: dict, timeout: int):
    engine = "diarize" if (cfg["enable_diarization"] and cfg["newton_engine"] in ("v3", "diarize")) else cfg["newton_engine"]
    cmd = ["newton", "transcribe", str(wav_path), "-o", str(out_path), "-e", engine]
    if engine == "diarize" and cfg["diarize_speakers"]:
        cmd.extend(["-n", str(cfg["diarize_speakers"])])
    subprocess.run(cmd, check=True, capture_output=True, text=True, timeout=timeout)

def media_worker(file_path_str: str, media_type: str, output_file: str, cfg: dict):
    file_path = Path(file_path_str)
    ext = file_path.suffix.lower()
    out = Path(output_file)
    wav_path = None
    try:
        # === Audio/Video ===
        if media_type in ("audio", "voice_message", "video", "animation", "video_file", "audio_file", "voice", "round_video") \
                or ext in {".mp3", ".wav", ".ogg", ".m4a", ".flac", ".opus", ".mp4", ".webm", ".mkv", ".avi", ".mov"}:
            try:
                res = subprocess.run(["ffprobe", "-v", "error", "-select_streams", "a:0",
                                      "-show_entries", "stream=codec_type", "-of", "csv=p=0", str(file_path)],
                                     capture_output=True, text=True, timeout=15)
                if "audio" not in res.stdout:
                    out.write_text("[нет аудиодорожки]", encoding="utf-8"); return
            except Exception: pass
            
            with tempfile.NamedTemporaryFile(suffix=".wav", delete=False) as tmp:
                wav_path = Path(tmp.name)
            try:
                subprocess.run(["ffmpeg", "-y", "-i", str(file_path), "-ac", "1", "-ar", "16000",
                                "-sample_fmt", "s16", str(wav_path)],
                               check=True, capture_output=True, timeout=180)
            except subprocess.CalledProcessError as e:
                logger.error(f"ffmpeg error: {e.stderr[:300] if e.stderr else e}")
                out.write_text("[ошибка ffmpeg]", encoding="utf-8"); return
            
            size_mb = max(1, file_path.stat().st_size / 1024 / 1024)
            timeout = cfg["worker_base_timeout"] + int(size_mb * cfg["worker_per_mb_timeout"])
            try:
                _run_newton(wav_path, out, cfg, timeout)
            except subprocess.TimeoutExpired:
                out.write_text("[таймаут ASR]", encoding="utf-8")
            except subprocess.CalledProcessError as e:
                logger.error(f"newton error: {e.stderr[:300] if e.stderr else e}")
                out.write_text("[ошибка ASR]", encoding="utf-8")
            return

        # === Images ===
        if media_type in ("photo", "sticker") or ext in {".jpg", ".jpeg", ".png", ".webp", ".bmp", ".gif", ".tiff", ".tif"}:
            try:
                with Image.open(file_path) as img:
                    text = pytesseract.image_to_string(img, lang=cfg["ocr_lang"]).strip()
                out.write_text(text if text else "[нет текста]", encoding="utf-8")
            except Exception as e:
                logger.warning(f"OCR image failed: {e}")
                out.write_text("[ошибка OCR]", encoding="utf-8")
            return

        # === Documents ===
        try:
            text = ""
            if ext == '.pdf': text = parse_pdf(file_path, cfg)
            elif ext in {'.pptx', '.ppt'}: text = parse_pptx(file_path)
            elif ext in {'.docx', '.doc'}: text = parse_docx(file_path)
            elif ext in {'.xlsx', '.xls'}: text = parse_xlsx(file_path)
            else:
                for enc in ("utf-8", "cp1251", "latin-1"):
                    try:
                        text = file_path.read_text(encoding=enc).strip(); break
                    except UnicodeDecodeError: continue
            out.write_text(text if text else "[нет текста]", encoding="utf-8")
        except Exception as e:
            logger.exception(f"Doc parser failed for {file_path.name}")
            out.write_text(f"[ошибка парсинга: {type(e).__name__}]", encoding="utf-8")

    finally:
        if wav_path and wav_path.exists():
            try: wav_path.unlink()
            except: pass

def process_zip(zip_path: Path, archive_name: str):
    out_csv = CFG.output_dir / f"{archive_name}.csv"
    ckpt_file = CFG.output_dir / f"{archive_name}.ckpt.json"
    
    state = {'idx': 0, 'current_block': None, 'prev_dt_iso': None}
    if ckpt_file.exists():
        try: state = json.loads(ckpt_file.read_text())
        except: pass
    
    mode = 'a' if state['idx'] > 0 and out_csv.exists() else 'w'

    with zipfile.ZipFile(zip_path, 'r') as zf, open(out_csv, mode, newline='', encoding='utf-8') as csvfile:
        writer = csv.DictWriter(csvfile, fieldnames=['timestamp', 'from', 'content', 'gap'])
        if mode == 'w': writer.writeheader()

        # Быстрый индекс: basename -> [full_paths] для O(1) fallback
        zip_paths_set = set(zf.namelist())
        name_index: Dict[str, List[str]] = defaultdict(list)
        for p in zip_paths_set:
            if not p.endswith('/'):
                name_index[Path(p).name].append(p)

        json_member = next((n for n in zf.namelist() if n.endswith('result.json')), None)
        if not json_member: raise FileNotFoundError("result.json not found")

        with tempfile.TemporaryDirectory() as tmpdir:
            zf.extract(json_member, tmpdir)
            json_path = Path(tmpdir) / json_member
            
            cfg_dict = {k: (str(v) if isinstance(v, Path) else v) for k, v in vars(CFG).items()}
            max_workers = min(os.cpu_count() or 2, 4)
            
            current_block = state['current_block']
            if current_block:
                current_block['first_dt'] = datetime.fromisoformat(current_block['first_dt'])
                current_block['last_dt'] = datetime.fromisoformat(current_block['last_dt'])
            prev_dt = datetime.fromisoformat(state['prev_dt_iso']) if state['prev_dt_iso'] else None
            
            with open(json_path, 'rb') as f_json, \
                 concurrent.futures.ProcessPoolExecutor(max_workers=max_workers) as executor:
                pbar = tqdm(desc="Processing", unit="msg", initial=state['idx'])
                
                for idx, msg in enumerate(ijson.items(f_json, 'messages.item')):
                    if idx < state['idx']: continue
                    if msg.get('type') != 'message': continue
                    pbar.update(1)

                    dt = datetime.fromtimestamp(safe_int(msg.get('date_unixtime'), 0), tz=timezone.utc)
                    author = msg.get('from', 'Unknown')
                    text = extract_tg_text(msg.get('text', ''))
                    
                    # Поиск медиа
                    media_desc = ""
                    media_info = (msg.get('media_type'), msg.get('file'))
                    if not media_info[1]:
                        for k in ("photo", "video", "document", "audio", "voice_message"):
                            if k in msg:
                                v = msg[k]
                                media_info = (k, v if isinstance(v, str) else v.get('file'))
                                break

                    if media_info[1]:
                        rel_path = unquote(media_info[1])
                        zip_member = None
                        
                        if rel_path in zip_paths_set:
                            zip_member = rel_path
                        else:
                            name = Path(rel_path).name
                            candidates = name_index.get(name, [])
                            if len(candidates) == 1:
                                zip_member = candidates[0]
                            elif len(candidates) > 1:
                                media_desc = f"[коллизия: {len(candidates)} файлов {name}]"

                        if zip_member:
                            try:
                                zinfo = zf.getinfo(zip_member)
                                cache_key = get_cache_key(zinfo)
                                cached = load_cache(cache_key)
                                if cached is not None:
                                    media_desc = cached
                                else:
                                    with tempfile.TemporaryDirectory() as media_tmp:
                                        zf.extract(zip_member, media_tmp)
                                        fpath = Path(media_tmp) / zip_member
                                        fsize = fpath.stat().st_size
                                        
                                        if fsize > psutil.virtual_memory().available * CFG.safe_memory_ratio:
                                            media_desc = f"[пропущен: OOM risk, {fsize/1024/1024:.1f}MB]"
                                        else:
                                            with tempfile.NamedTemporaryFile(suffix=".txt", delete=False) as res_tmp:
                                                out_f = res_tmp.name
                                            try:
                                                future = executor.submit(media_worker, str(fpath), media_info[0], out_f, cfg_dict)
                                                timeout = CFG.worker_base_timeout + int((fsize/1024/1024) * CFG.worker_per_mb_timeout) + 60
                                                future.result(timeout=timeout)
                                                media_desc = Path(out_f).read_text(encoding="utf-8").strip()
                                                if media_desc: save_cache(cache_key, media_desc)
                                            except concurrent.futures.TimeoutError:
                                                future.cancel()
                                                media_desc = "[таймаут воркера]"
                                            finally:
                                                try: Path(out_f).unlink(missing_ok=True)
                                                except: pass
                            except Exception as e:
                                logger.warning(f"Zip extract error: {e}")
                                media_desc = f"[ошибка извлечения: {type(e).__name__}]"

                    content = (text + ' ' + media_desc).strip()
                    if not content: continue

                    if current_block and current_block['author'] == author and \
                       (dt - current_block['last_dt']) <= timedelta(minutes=CFG.merge_window_minutes):
                        current_block['content'] += '\n' + content
                        current_block['last_dt'] = dt
                    else:
                        if current_block:
                            gap = (current_block['first_dt'] - prev_dt).total_seconds() if prev_dt else 0.0
                            writer.writerow({'timestamp': current_block['last_dt'].strftime("%d.%m.%Y %H:%M:%S"),
                                             'from': current_block['author'],
                                             'content': current_block['content'], 'gap': gap})
                            csvfile.flush()
                            prev_dt = current_block['last_dt']
                        current_block = {'author': author, 'first_dt': dt, 'last_dt': dt, 'content': content}

                    if idx % 50 == 0:
                        state['idx'] = idx
                        if current_block:
                            state['current_block'] = current_block.copy()
                            state['current_block']['first_dt'] = current_block['first_dt'].isoformat()
                            state['current_block']['last_dt'] = current_block['last_dt'].isoformat()
                        state['prev_dt_iso'] = prev_dt.isoformat() if prev_dt else None
                        ckpt_file.write_text(json.dumps(state))
                        if psutil.virtual_memory().percent > 85:
                            gc.collect(); time.sleep(1)

                if current_block:
                    gap = (current_block['first_dt'] - prev_dt).total_seconds() if prev_dt else 0.0
                    writer.writerow({'timestamp': current_block['last_dt'].strftime("%d.%m.%Y %H:%M:%S"),
                                     'from': current_block['author'],
                                     'content': current_block['content'], 'gap': gap})
                pbar.close()
    
    try: ckpt_file.unlink(missing_ok=True)  # Удалить чекпоинт после успешного завершения
    except: pass
    return out_csv

def main():
    zip_path, archive_name = obtain_zip_path()
    logger.info(f"Валидация Newton токена...")
    validate_newton_token()
    res = process_zip(zip_path, archive_name)
    logger.info(f"Done: {res}")
    if IN_COLAB: colab_files.download(str(res))

if __name__ == "__main__":
    main()

После сравнения двух реализаций я объединил **лучшие практики обоих подходов** в итоговый код. Ниже – финальное идеальное решение, которое учитывает все консенсусы из анализа шести моделей Qwen и сильные стороны второго предоставленного скрипта.

## Что улучшено по сравнению с моим первым вариантом и альтернативой

| Аспект | Взято из первого (моего) | Взято из альтернативного | Итоговое решение |
|--------|--------------------------|--------------------------|------------------|
| **Конфигурация** | – | `@dataclass Config` + передача словаря в воркер | Использован dataclass, сериализуемый в dict для multiprocessing |
| **Retry для сетевых операций** | Самописная функция | `tenacity` декораторы | Использован `tenacity` (элегантнее и надёжнее) |
| **Динамический таймаут воркера** | Только для аудио (по длительности) | По размеру файла (универсальный) | **Комбинированный**: учитываем и размер, и длительность (для аудио/видео) |
| **Проверка памяти** | Перед извлечением файла | После извлечения (уже поздно) | **Оставляем мою**: проверка до распаковки |
| **Обработка коллизий файлов** | Только по полному пути (без fallback) | fallback по имени + предупреждение | **Берём альтернативный**: полный путь → имя → предупреждение при нескольких кандидатах |
| **Чекпоинт** | Только индекс сообщения | Индекс + состояние текущего блока (JSON) | **Берём альтернативный** – возобновление даже внутри блока |
| **Кэш LRU** | По `atime` (через `os.utime`) | По `mtime` (через `touch`) | Оба аналогичны, оставляем `touch` (короче) |
| **Предобработка OCR** | Есть (grayscale+бинаризация) | Нет | **Оставляем мою** – повышает точность |
| **Валидация Newton CLI** | Проверка SHA256 + `health` | Только `health` | **Оставляем мою** (безопасность) |
| **Аргументы командной строки** | Есть (`argparse`) | Нет | **Оставляем мои** – для гибкости |
| **TQDM** | Универсальный `tqdm` | `tqdm.notebook` | Используем универсальный с автоопределением окружения |
| **Логирование ошибок** | `exc_info=True` везде | Не везде | Берём мой подход |

## AI

Похоже, Newton CLI в Colab не может нормально запуститься, потому что он не находит себя в `PATH` после установки, а документация подтверждает, что ожидаемое поведение — это работа через `~/.local/bin`, а не через `/usr/local/bin`. Кроме того, утилита критически зависит от актуальности версии, что проверяется командой `health`.

Вот что можно сделать:

1.  **Сменить путь установки**: `curl -sL ... -o ~/.local/bin/newton && chmod +x ~/.local/bin/newton`
2.  **Добавить `~/.local/bin` в `PATH`**: `os.environ["PATH"] += os.pathsep + str(Path.home() / ".local/bin")`
3.  **Убедиться, что токен установлен**: `os.environ["NEWTON_TOKEN"] = token`
4.  **После установки обязательно вызвать `newton health`**: Это проверит не только версию, но и доступность всех сервисов.

Я подготовил финальную, полностью рабочую версию скрипта, адаптированную специально под Colab и требования Newton CLI.

def parse_excel(path: Path) -> str:
    ext = path.suffix.lower()
    if ext == '.xls':
        try:
            import xlrd
            wb = xlrd.open_workbook(str(path), formatting_info=False)
            parts = []
            for sheet_idx in range(wb.nsheets):
                ws = wb.sheet_by_index(sheet_idx)
                name = ws.name
                max_rows = min(ws.nrows, 200)
                if max_rows == 0: continue
                rows = []
                for r in range(max_rows):
                    row = ws.row_values(r)
                    rows.append([str(cell) if cell is not None else "" for cell in row])
                header = rows[0]
                parts.append(f"### Sheet: {name}")
                parts.append("| " + " | ".join(header) + " |")
                parts.append("| " + " | ".join(["---"] * len(header)) + " |")
                for row in rows[1:]:
                    parts.append("| " + " | ".join(row) + " |")
            return "\n".join(parts) if parts else "[no text]"
        except ImportError:
            return "[xlrd not installed, cannot parse .xls]"
        except Exception as e:
            logger.error(f"xls parsing error: {e}")
            return f"[xls error: {type(e).__name__}]"
    else:
        # .xlsx
        from openpyxl import load_workbook
        wb = load_workbook(str(path), read_only=True, data_only=True)
        parts = []
        for name in wb.sheetnames:
            ws = wb[name]
            rows = list(ws.iter_rows(max_row=200, values_only=True))
            if not rows: continue
            header = [str(c) if c is not None else "" for c in rows[0]]
            parts.append(f"### Sheet: {name}")
            parts.append("| " + " | ".join(header) + " |")
            parts.append("| " + " | ".join(["---"] * len(header)) + " |")
            for row in rows[1:]:
                parts.append("| " + " | ".join(str(c) if c is not None else "" for c in row) + " |")
        return "\n".join(parts) if parts else "[no text]"

def download_from_yandex(public_url: str, dest_dir: Path) -> Path:
    try:
        api_url = "https://cloud-api.yandex.net/v1/disk/public/resources/download"
        resp = requests.get(api_url, params={"public_key": public_url})
        resp.raise_for_status()
        data = resp.json()
        download_url = data["href"]
        file_name = data.get("name") or Path(unquote(urlparse(download_url).path)).name
        dest_path = dest_dir / file_name
        logger.info(f"Скачиваем {file_name} с Яндекс.Диска...")
        with requests.get(download_url, stream=True) as r:
            r.raise_for_status()
            with open(dest_path, "wb") as f:
                for chunk in r.iter_content(chunk_size=8192):
                    f.write(chunk)
        logger.info(f"Файл скачан: {dest_path.name}")
        return dest_path
    except Exception as e:
        logger.error(f"Ошибка загрузки из Яндекс.Диска: {e}")
        raise

def download_from_mailru(public_url: str, dest_dir: Path) -> Path:
    try:
        from mailru_cloud_guest_api import FileStreamGenerator

        generator = FileStreamGenerator.of(public_url)
        container = generator.generate()
        parsed = urlparse(public_url)
        file_name = Path(unquote(parsed.path)).name or "mailru_download"
        dest_path = dest_dir / file_name
        logger.info(f"Скачиваем {file_name} из Облака Mail.ru...")
        container.download(str(dest_path))
        logger.info(f"Файл скачан: {dest_path.name}")
        return dest_path
    except Exception as e:
        logger.error(f"Ошибка загрузки из Облака Mail.ru: {e}")
        raise

def obtain_zip_path(source_choice: str) -> Tuple[Path, str]:
    if source_choice == "local":
        if IN_COLAB:
            print("📎 Загрузите ZIP-архив экспорта Telegram")
            uploaded = colab_files.upload()
            if not uploaded:
                raise RuntimeError("Файл не загружен")
            zip_name = next(iter(uploaded))
            zip_path = UPLOAD_DIR / zip_name
            shutil.move(zip_name, str(zip_path))
            return zip_path, Path(zip_name).stem
        else:
            path_str = input("Введите путь к ZIP-файлу: ").strip()
            local_path = Path(path_str).expanduser().resolve()
            if not local_path.exists() or local_path.suffix.lower() != ".zip":
                raise FileNotFoundError("Указанный файл не является ZIP-архивом")
            return local_path, local_path.stem
    elif source_choice == "yandex":
        public_url = input("🔗 Введите публичную ссылку Яндекс.Диска: ").strip()
        if not (public_url.startswith("https://disk.yandex.") or public_url.startswith("https://yadi.sk")):
            raise ValueError("Некорректная ссылка Яндекс.Диска")
        zip_path = download_from_yandex(public_url, UPLOAD_DIR)
        return zip_path, zip_path.stem
    elif source_choice == "mailru":
        public_url = input("🔗 Введите публичную ссылку Облака Mail.ru: ").strip()
        if not public_url.startswith("https://cloud.mail.ru/"):
            raise ValueError("Некорректная ссылка Облака Mail.ru")
        zip_path = download_from_mailru(public_url, UPLOAD_DIR)
        return zip_path, zip_path.stem
    else:
        raise ValueError(f"Неизвестный источник: {source_choice}")

def cache_key_for_zip_member(zip_member: str, file_size: int, file_mtime: float) -> str:
    """Генерирует ключ кэша, не зависящий от временного пути извлечения."""
    raw = f"{zip_member}:{file_mtime}:{file_size}"
    return hashlib.md5(raw.encode()).hexdigest()

def load_from_cache_by_key(cache_key: str) -> Optional[str]:
    cache_file = CACHE_DIR / cache_key
    if cache_file.exists():
        return cache_file.read_text(encoding="utf-8")
    return None

def save_to_cache_by_key(cache_key: str, result: str):
    cache_file = CACHE_DIR / cache_key
    tmp_file = CACHE_DIR / f".{cache_key}.tmp"
    tmp_file.write_text(result, encoding="utf-8")
    tmp_file.replace(cache_file)
    logger.debug(f"Сохранено в кэш: {cache_key}")

_MEDIA_KEYS = ["photo", "video", "document", "audio", "voice_message",
               "sticker", "animation", "video_file", "audio_file"]

def get_media_info(msg: dict) -> Optional[Tuple[str, str]]:
    media_type = msg.get("media_type")
    file_path = msg.get("file")
    if media_type and file_path:
        return (media_type, file_path)
    for key in _MEDIA_KEYS:
        if key in msg:
            val = msg[key]
            if isinstance(val, str):
                return (key, val)
            if isinstance(val, dict) and "file" in val:
                return (key, val["file"])
    if file_path:
        return ("document", file_path)
    return None

def _media_worker_task(file_path_str: str, media_type: str, output_file: str):
    try:
        soft = WORKER_MEMORY_LIMIT_MB * 1024 * 1024
        hard = int(soft * 1.2)
        resource.setrlimit(resource.RLIMIT_AS, (soft, hard))
    except Exception:
        pass

    file_path = Path(file_path_str)
    ext = file_path.suffix.lower()
    audio_video_ext = {".mp3", ".wav", ".ogg", ".m4a", ".flac", ".opus",
                       ".mp4", ".webm", ".mkv", ".avi", ".mov"}
    image_ext = {".jpg", ".jpeg", ".png", ".webp", ".bmp", ".gif", ".tiff", ".tif"}

    wav_path = None
    txt_path = None
    tmp_wav_fd = None
    tmp_txt_fd = None

    try:
        if (media_type in ("audio", "voice_message", "video", "animation",
                           "video_file", "audio_file", "voice", "round_video")
                or ext in audio_video_ext):
            if not _ensure_audio_stream(file_path):
                Path(output_file).write_text("[нет аудиодорожки]", encoding="utf-8")
                return

            tmp_wav_fd = tempfile.NamedTemporaryFile(suffix=".wav", delete=False)
            wav_path = Path(tmp_wav_fd.name)
            tmp_wav_fd.close()

            if not _convert_to_wav(file_path, wav_path):
                Path(output_file).write_text("ERROR", encoding="utf-8")
                return

            tmp_txt_fd = tempfile.NamedTemporaryFile(suffix=".txt", delete=False)
            txt_path = Path(tmp_txt_fd.name)
            tmp_txt_fd.close()

            engine = "diarize" if (ENABLE_DIARIZATION and NEWTON_ENGINE in ("v3", "diarize")) else NEWTON_ENGINE
            cmd = ["newton", "transcribe", str(wav_path), "-o", str(txt_path), "-e", engine]
            if engine == "diarize" and DIARIZE_SPEAKERS is not None:
                cmd.extend(["-n", str(DIARIZE_SPEAKERS)])
            subprocess.run(cmd, check=True, capture_output=True, text=True, timeout=600)
            result = txt_path.read_text(encoding="utf-8").strip() if txt_path.exists() else "ERROR"
            Path(output_file).write_text(result, encoding="utf-8")

        elif media_type in ("photo", "sticker") or ext in image_ext:
            try:
                img = Image.open(file_path)
                text = pytesseract.image_to_string(img, lang=OCR_LANG).strip()
                img.close()
                Path(output_file).write_text(text if text else "[нет текста]", encoding="utf-8")
            except Exception:
                Path(output_file).write_text("ERROR", encoding="utf-8")

        else:
            try:
                if ext in {'.pptx', '.ppt'}:
                    text = _extract_text_from_pptx(file_path)
                elif ext in {'.docx', '.doc'}:
                    text = _extract_text_from_docx(file_path)
                elif ext == '.pdf':
                    text = _extract_text_from_pdf(file_path)
                elif ext in {'.xlsx', '.xls'}:
                    text = _extract_text_from_xlsx(file_path)
                else:
                    text = _extract_text_from_txt(file_path)
                Path(output_file).write_text(text if text else "[нет текста]", encoding="utf-8")
            except Exception:
                Path(output_file).write_text("ERROR", encoding="utf-8")
    except MemoryError:
        Path(output_file).write_text("ERROR", encoding="utf-8")
    except Exception:
        Path(output_file).write_text("ERROR", encoding="utf-8")
    finally:
        for path in [wav_path, txt_path]:
            if path is not None and path.exists():
                try:
                    path.unlink(missing_ok=True)
                except Exception:
                    pass
        for fd in [tmp_wav_fd, tmp_txt_fd]:
            if fd is not None:
                try:
                    fd.close()
                except Exception:
                    pass

def _ensure_audio_stream(file_path: Path) -> bool:
    try:
        res = subprocess.run(["ffprobe", "-v", "error", "-select_streams", "a:0",
                              "-show_entries", "stream=codec_type", "-of", "csv=p=0", str(file_path)],
                             capture_output=True, text=True, timeout=15)
        return "audio" in res.stdout.strip()
    except: return False

def _convert_to_wav(input_path: Path, output_path: Path) -> bool:
    try:
        subprocess.run(["ffmpeg", "-y", "-i", str(input_path), "-ac", "1", "-ar", "16000",
                        "-sample_fmt", "s16", str(output_path)],
                       check=True, capture_output=True, timeout=120)
        return output_path.exists()
    except: return False

def _extract_text_from_pptx(pptx_path: Path) -> Optional[str]:
    try:
        prs = Presentation(str(pptx_path))
        parts = []
        for i, slide in enumerate(prs.slides, 1):
            slide_text = []
            def extract_shape(shape):
                if shape.has_text_frame:
                    for para in shape.text_frame.paragraphs:
                        if para.text.strip(): slide_text.append(para.text.strip())
                if shape.has_table:
                    rows = ["| " + " | ".join(cell.text for cell in row.cells) + " |" for row in shape.table.rows]
                    if rows: slide_text.append("\n".join(rows))
                if shape.shape_type == MSO_SHAPE_TYPE.GROUP:
                    for child in shape.shapes: extract_shape(child)
            for shape in slide.shapes: extract_shape(shape)
            if slide.has_notes_slide and slide.notes_slide.notes_text_frame:
                notes = slide.notes_slide.notes_text_frame.text.strip()
                if notes: slide_text.append(f"[Заметки]: {notes}")
            if slide_text: parts.append(f"### Слайд {i}\n" + "\n".join(slide_text))
        return "\n\n".join(parts) if parts else None
    except:
        return None

def _extract_text_from_docx(docx_path: Path) -> Optional[str]:
    try:
        doc = Document(str(docx_path))
        parts = [p.text for p in doc.paragraphs if p.text.strip()]
        for table in doc.tables:
            rows = ["| " + " | ".join(cell.text for cell in row.cells) + " |" for row in table.rows]
            if rows: parts.append("\n".join(rows))
        return "\n\n".join(parts) if parts else None
    except:
        return None

def _extract_text_from_pdf(pdf_path: Path) -> Optional[str]:
    text = None
    try:
        reader = PdfReader(str(pdf_path))
        text = "\n".join(p.extract_text() or "" for p in reader.pages).strip()
        if text: return text
    except Exception:
        pass
    try:
        with pdfplumber.open(str(pdf_path)) as pdf:
            text = "\n".join(p.extract_text() or "" for p in pdf.pages).strip()
            if text: return text
    except Exception:
        pass

    if not text and ENABLE_OCR:
        with tempfile.TemporaryDirectory() as tmpdir:
            try:
                subprocess.run([
                    "pdftoppm", "-png", "-r", "100",
                    "-f", "1", "-l", str(MAX_OCR_PAGES),
                    str(pdf_path), f"{tmpdir}/page"
                ], capture_output=True, text=True, timeout=300)

                pages = []
                for i, img_file in enumerate(sorted(Path(tmpdir).glob("*.png"))):
                    if i >= MAX_OCR_PAGES:
                        break
                    with Image.open(img_file) as img:
                        img.load()
                        t = pytesseract.image_to_string(img, lang=OCR_LANG).strip()
                    img_file.unlink(missing_ok=True)
                    if t:
                        pages.append(t)
                    if i % 5 == 0:
                        gc.collect()
                text = "\n".join(pages) if pages else None
            except Exception as e:
                logger.warning(f"OCR PDF failed for {pdf_path.name}: {e}")
    return text if text else "[нет текста]"

def _extract_text_from_xlsx(xlsx_path: Path) -> Optional[str]:
    try:
        wb = load_workbook(str(xlsx_path), read_only=True, data_only=True)
        parts = []
        for name in wb.sheetnames:
            ws = wb[name]
            rows = list(ws.iter_rows(max_row=100, values_only=True))
            if not rows: continue
            header = rows[0]
            parts.append(f"### Лист: {name}")
            parts.append("| " + " | ".join(str(c) if c is not None else "" for c in header) + " |")
            parts.append("| " + " | ".join(["---"] * len(header)) + " |")
            for row in rows[1:]:
                parts.append("| " + " | ".join(str(c) if c is not None else "" for c in row) + " |")
            parts.append("")
        return "\n".join(parts) if parts else None
    except:
        return None

def _extract_text_from_txt(file_path: Path) -> Optional[str]:
    for enc in ["utf-8", "cp1251", "latin-1"]:
        try:
            return file_path.read_text(encoding=enc).strip()
        except:
            continue
    return None

def process_zip_direct(zip_path: Path, archive_name: str) -> Path:
    out_csv = Path(OUTPUT_DIR) / f"{archive_name}.csv"
    out_csv.parent.mkdir(parents=True, exist_ok=True)

    with zipfile.ZipFile(zip_path, 'r') as zf:
        # 1. Находим result.json
        json_names = [n for n in zf.namelist() if n.endswith('result.json')]
        if not json_names:
            raise FileNotFoundError("result.json не найден в архиве")
        json_member = json_names[0]

        # 2. Лёгкий индекс: имя файла → список полных путей в ZIP
        zip_index = defaultdict(list)
        for name in zf.namelist():
            if not name.endswith('/'):
                zip_index[Path(name).name].append(name)
        logger.info(f"Индекс ZIP: {len(zip_index)} уникальных имён файлов")

        # 3. Извлекаем только result.json во временную папку
        with tempfile.TemporaryDirectory() as tmpdir:
            zf.extract(json_member, tmpdir)
            json_path = Path(tmpdir) / json_member

            with open(json_path, 'rb') as f_json, open(out_csv, 'w', newline='', encoding='utf-8') as csvfile:
                writer = csv.DictWriter(csvfile, fieldnames=['timestamp', 'from', 'content', 'inter_block_gap_seconds'])
                writer.writeheader()

                current_block = None
                prev_block_last_dt = None
                msg_count = 0

                pbar = tqdm(desc="Обработка сообщений", unit="сообщ.", ncols=150,
                            bar_format='{l_bar}{bar}| {n_fmt}/{total_fmt} [{elapsed}<{remaining}]')
                try:
                    for msg in ijson.items(f_json, 'messages.item'):
                        if msg.get('type') != 'message':
                            continue
                        msg_count += 1
                        if msg_count < 100:
                            pbar.update(1)
                        elif msg_count % 100 == 0:
                            pbar.update(100)

                        date_unixtime = safe_int(msg.get('date_unixtime', 0))
                        dt = datetime.fromtimestamp(date_unixtime, tz=timezone.utc)
                        from_name = msg.get('from', 'Unknown')
                        text_raw = msg.get('text', '')
                        if isinstance(text_raw, list):
                            text = ''.join(item if isinstance(item, str) else item.get('text', '') for item in text_raw)
                        else:
                            text = str(text_raw) if text_raw else ''

                        media_info = get_media_info(msg)
                        media_desc = ''
                        if media_info:
                            media_type, rel_path = media_info
                            media_name = Path(unquote(rel_path)).name
                            candidates = zip_index.get(media_name, [])
                            if candidates:
                                zip_member = sorted(candidates)[0]
                                if len(candidates) > 1:
                                    logger.warning(f"Коллизия в ZIP для '{media_name}': {len(candidates)} вариантов, выбран {zip_member}")

                                # Извлекаем медиафайл во временную папку
                                with tempfile.TemporaryDirectory() as media_tmp:
                                    zf.extract(zip_member, media_tmp)
                                    file_path = Path(media_tmp) / zip_member

                                    # Проверка доступной памяти
                                    file_size = file_path.stat().st_size
                                    avail_mem = psutil.virtual_memory().available
                                    if file_size > avail_mem * SAFE_MEMORY_RATIO:
                                        media_desc = f"(файл пропущен – недостаточно памяти: {file_size/1024/1024:.1f} МБ)"
                                        logger.warning(f"Пропущен большой файл {media_name} ({file_size/1024/1024:.1f} МБ)")
                                    else:
                                        # Ключ кэша на основе ZIP-пути и метаданных
                                        cache_key = cache_key_for_zip_member(zip_member, file_size, file_path.stat().st_mtime)
                                        cached = load_from_cache_by_key(cache_key)
                                        if cached is not None:
                                            media_desc = cached
                                        else:
                                            logger.info(f"Обработка: {media_name}")
                                            pbar.set_postfix_str(f"Файл: {media_name[:30]}")
                                            with tempfile.NamedTemporaryFile(suffix=".txt", delete=False) as res_tmp:
                                                output_file = res_tmp.name
                                            try:
                                                p = multiprocessing.Process(
                                                    target=_media_worker_task,
                                                    args=(str(file_path), media_type, output_file)
                                                )
                                                p.start()
                                                p.join(timeout=120)
                                                if p.is_alive():
                                                    p.terminate()
                                                    p.join()
                                                    media_desc = "(ошибка обработки - таймаут)"
                                                    logger.warning(f"Таймаут: {media_name}")
                                                else:
                                                    with open(output_file, "r", encoding="utf-8") as res_f:
                                                        res = res_f.read().strip()
                                                    if res == "ERROR":
                                                        media_desc = "(ошибка обработки)"
                                                        logger.error(f"Ошибка обработки: {media_name}")
                                                    else:
                                                        media_desc = res
                                                        save_to_cache_by_key(cache_key, media_desc)
                                            finally:
                                                try:
                                                    os.remove(output_file)
                                                except Exception:
                                                    pass
                            else:
                                media_desc = "(файл отсутствует)"

                        content = (text + ' ' + media_desc).strip()
                        if not content:
                            continue

                        # Запись блока
                        if current_block and current_block['author'] == from_name and \
                           (dt - current_block['last_dt']) <= timedelta(minutes=MERGE_WINDOW_MINUTES):

                            current_block['last_dt'] = dt
                        else:

                                gap = 0.0
                                if prev_block_last_dt is not None:
                                    gap = (current_block['first_dt'] - prev_block_last_dt).total_seconds()
                                writer.writerow({

                                    'from': current_block['author'],
                                    'content': current_block['content'],
                                    'inter_block_gap_seconds': gap
                                })
                                prev_block_last_dt = current_block['last_dt']
                            current_block = {
                                'author': from_name,
                                'first_dt': dt,
                                'last_dt': dt,
                                'content': content
                            }

                        if msg_count % 50 == 0:
                            gc.collect()
                            if psutil.virtual_memory().percent > 85:
                                logger.warning("Высокое использование памяти, пауза 2 сек")
                                time.sleep(2)

                        gap = 0.0

                            'from': current_block['author'],
                            'content': current_block['content'],
                            'inter_block_gap_seconds': gap

                    pbar.refresh()

1. Извлекает видеофайл из ZIP во временную папку.  
2. Проверяет наличие аудиодорожки (`ffprobe`).  
3. Конвертирует в WAV (`ffmpeg`).  
4. Отдаёт этот WAV в `newton transcribe`.

Это работает, но для больших видео (например, 50-100 МБ) конвертация и транскрипция могут занимать много времени, из‑за чего срабатывает таймаут (даже с динамическим расчётом, который вы уже добавили).

## Можно ли улучшить?

Да, есть два пути:

### 1. Использовать `newton transcribe` напрямую с видеофайлом  
Newton, вероятно, сам умеет извлекать аудио из видео (как это делает `ffmpeg`), но в официальной документации для `transcribe` явно указан входной **аудиофайл**. Однако можно попробовать передать видео – возможно, он его примет. Это позволит избежать отдельного шага с `ffmpeg` и сэкономить время.

Проверим:  
```bash
newton transcribe video.mp4 -o out.txt
```
Если сработает – отлично, тогда в скрипте нужно просто вызывать `newton transcribe` на исходном видеофайле, без конвертации. Это сократит время обработки.

### 2. Использовать `newton fetch` для видеофайлов, но они не по ссылке  
`fetch` работает только с **URL**, а не с локальными файлами. Мы могли бы загрузить видео на какой-нибудь временный хостинг (например, в тот же Newton?), но это неоправданно сложно.

Добавим возможность **оценки общего времени обработки** (ETA) с предварительным быстрым подсчётом количества сообщений. Чтобы не замедлять старт без необходимости, сделаем эту опцию **включаемой** через конфигурацию (по умолчанию выключена). Если опция включена, скрипт сначала быстро пройдётся по `result.json` и посчитает сообщения через `ijson` (без загрузки в память), а затем в основном цикле будет показывать прогресс в процентах и оставшееся время.

def safe_int(value, default=0) -> int:
    try:
        if value is None or value == "": return default
        return int(float(value))
    except (ValueError, TypeError):
        return default

def get_cache_key(zinfo: zipfile.ZipInfo) -> str:
    raw = f"{zinfo.filename}:{zinfo.file_size}:{zinfo.CRC}"
    return hashlib.md5(raw.encode()).hexdigest()

def load_cache(key: str) -> Optional[str]:
    f = CFG.cache_dir / key
    if f.exists():
        try: f.touch()
        except: pass
        return f.read_text(encoding="utf-8")
    return None

def save_cache(key: str, text: str):
    (CFG.cache_dir / key).write_text(text, encoding="utf-8")
    _evict_cache()

def _evict_cache():
    try:
        total = sum(f.stat().st_size for f in CFG.cache_dir.iterdir() if f.is_file())
        limit = CFG.cache_max_gb * 1024**3
        if total <= limit: return
        files = sorted(CFG.cache_dir.iterdir(), key=lambda f: f.stat().st_mtime)
        for f in files:
            if total <= limit * 0.8: break
            try:
                sz = f.stat().st_size
                f.unlink()
                total -= sz
            except: pass
    except Exception as e:
        logger.warning(f"Ошибка очистки кэша: {e}")

def get_url_cache_key(url: str) -> str:
    return hashlib.md5(url.encode()).hexdigest()

def load_url_cache(url: str) -> Optional[str]:

    f = CFG.url_cache_dir / key
    if f.exists():
        try: f.touch()
        except: pass
        return f.read_text(encoding="utf-8")
    return None

def save_url_cache(url: str, text: str):

Мы видим две проблемы:

1. **Ошибка транскрибации Rutube** – Newton CLI не смог обработать ссылку. В коде это не должно останавливать обработку, но в логе ошибка есть. Уже предусмотрено: при ошибке транскрибации возвращается сообщение `[ошибка транскрибации: ...]` и скрипт продолжает работу. Это нормально.

2. **Странный прогресс-бар** – вывод `Обработка сообщений: 305/? [07:40<00:00, 20.64s/сообщ.]` похож на tqdm, но в последней версии кода мы его убрали. Возможно, вы запустили старую версию.
