def classify_project_type(project_name, description):
    """Классифицирует проект как административный или проектный"""
    text = str(project_name).lower() + ' ' + str(description).lower()
    
    admin_keywords = [
        'административный', 'гкп', 'админ', 'внутренний',
        'совещание', 'планерка', 'корпоратив',
        'отпуск', 'больничный', 'отгул',
        'обучение', 'самообучение', 'курсы', 'тренинг'
    ]
    
    for keyword in admin_keywords:
        if keyword in text:
            if 'отпуск' in text or 'больничный' in text:
                return 'vacation'
            elif 'обучение' in text or 'курсы' in text or 'тренинг' in text:
                return 'training'
            else:
                return 'administrative'
    return 'project'

def analyze_employee(employee_name, employee_data, working_days_list, expected_hours):
    """Анализирует одного сотрудника"""
    result = {
        'name': employee_name,
        'total_gap_hours': 0,
        'project_hours': 0,
        'admin_hours': 0,
        'training_hours': 0,
        'vacation_hours': 0,
        'efficiency_percent': 0,
        'quality_score': 100,
        'total_score': 0,
        'vacation_days': 0
    }
    
    # Анализ простоев
    daily_hours = employee_data.groupby(employee_data['Дата'].dt.date)['Часы'].sum().to_dict()
    daily_types = employee_data.groupby(employee_data['Дата'].dt.date)['Тип_работы'].apply(list).to_dict()
    
    vacation_days = set()
    for work_day in working_days_list:
        day_hours = daily_hours.get(work_day, 0)
        day_type_list = daily_types.get(work_day, [])
        
        if 'vacation' in day_type_list:
            vacation_days.add(work_day)
            continue
        
        if day_hours < 8:
            gap = 8 - day_hours
            result['total_gap_hours'] += gap
    
    result['vacation_days'] = len(vacation_days)
    
    # Распределение по типам
    type_hours = employee_data.groupby('Тип_работы')['Часы'].sum().to_dict()
    result['project_hours'] = type_hours.get('project', 0)
    result['admin_hours'] = type_hours.get('administrative', 0)
    result['training_hours'] = type_hours.get('training', 0)
    result['vacation_hours'] = type_hours.get('vacation', 0)
    
    # Эффективность
    adjusted_expected = expected_hours - (result['vacation_days'] * 8)

            quality_issues += 1
    
    empty_count = sum(1 for desc in descriptions if len(str(desc).strip()) < 3)
    quality_issues += (empty_count > 0)
    
    result['quality_score'] = max(0, 100 - (quality_issues * 10))

    quality_score = result['quality_score'] * 0.3

    gap_score = max(0, 100 - gap_penalty) * 0.3

def analyze_report_quality(descriptions):
    """Расширенный анализ качества описаний"""
    quality_score = 100
    issues = []
    
    descriptions = [str(d).strip() for d in descriptions if str(d).strip()]
    
    # Анализ пустых описаний
    empty_count = sum(1 for d in descriptions if len(d) < 5)
    if empty_count > 0:
        quality_score -= 20
        issues.append(f"Пустые описания: {empty_count}")
    
    # Анализ повторяемости
    desc_counter = Counter(descriptions)
    repeated = [(desc, count) for desc, count in desc_counter.items() 
                if count > 3 and len(desc) < 50]
    if repeated:
        quality_score -= len(repeated) * 5
        issues.append(f"Повторяющиеся формулировки: {len(repeated)}")
    
    # Анализ конкретности (длина описания)
    vague_descriptions = sum(1 for d in descriptions if 5 < len(d) < 20)
    if vague_descriptions > len(descriptions) * 0.3:  # 30% vague
        quality_score -= 15
        issues.append("Слишком краткие описания")
    
    return max(quality_score, 0), issues
