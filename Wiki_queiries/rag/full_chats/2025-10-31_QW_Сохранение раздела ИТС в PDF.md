import os
import time
from urllib.parse import urlparse
from selenium import webdriver
from selenium.webdriver.common.by import By
from selenium.webdriver.support.ui import WebDriverWait
from selenium.webdriver.support import expected_conditions as EC
from selenium.webdriver.chrome.options import Options

def setup_driver():
    chrome_options = Options()
    chrome_options.add_argument('--headless')
    chrome_options.add_argument('--no-sandbox')
    chrome_options.add_argument('--disable-dev-shm-usage')
    chrome_options.add_argument('--disable-gpu')
    chrome_options.add_argument('--window-size=1920,1080')
    return webdriver.Chrome(options=chrome_options)

def save_page_to_pdf(url, output_path):
    driver = setup_driver()
    try:
        driver.get(url)
        
        # Ожидание загрузки контента
        WebDriverWait(driver, 10).until(
            EC.presence_of_element_located((By.TAG_NAME, "body"))
        )
        
        # Скроллинг для активации ленивой загрузки
        driver.execute_script("window.scrollTo(0, document.body.scrollHeight);")
        time.sleep(2)
        
        # Создание PDF
        pdf_options = {
            'page-size': 'A4',
            'margin-top': '0.5in',
            'margin-right': '0.5in',
            'margin-bottom': '0.5in',
            'margin-left': '0.5in',
            'encoding': "UTF-8",
        }
        
        driver.print_page()  # В новых версиях Selenium
        
    except Exception as e:
        print(f"Ошибка: {str(e)}")
    finally:
        driver.quit()

# Интерактивный ввод
url = input("Введите URL раздела ИТС: ").strip()
if not url.startswith(('http://', 'https://')):
    url = 'https://' + url

# Сохранение PDF
domain = urlparse(url).netloc
filename = f"{domain}_{time.strftime('%Y%m%d-%H%M%S')}.pdf"
save_page_to_pdf(url, filename)

print(f"\nPDF сохранен как: {filename}")

import os

def setup_driver():
    """Настройка Chrome драйвера для Colab"""
    chrome_options = Options()
    chrome_options.add_argument('--headless')
    chrome_options.add_argument('--no-sandbox')
    chrome_options.add_argument('--disable-dev-shm-usage')
    chrome_options.add_argument('--disable-gpu')
    chrome_options.add_argument('--window-size=1920,1080')
    return webdriver.Chrome(options=chrome_options)

def login_to_its(driver, login, password):
    """Авторизация на сайте ИТС"""
    print("Выполняю вход на сайт ИТС...")
    
    # Переход на страницу авторизации
    driver.get("https://its.1c.ru/login")
    
    # Ожидание появления формы входа
    WebDriverWait(driver, 10).until(
        EC.presence_of_element_located((By.ID, "login"))
    )
    
    # Ввод логина
    login_field = driver.find_element(By.ID, "login")
    login_field.clear()
    login_field.send_keys(login)
    
    # Ввод пароля
    password_field = driver.find_element(By.ID, "password")
    password_field.clear()
    password_field.send_keys(password)
    
    # Нажатие кнопки входа
    login_button = driver.find_element(By.CSS_SELECTOR, "button[type='submit']")
    login_button.click()
    
    # Ожидание завершения авторизации
    time.sleep(3)
    
    # Проверка успешного входа
    if "login" in driver.current_url:
        print("Ошибка: Не удалось авторизоваться")
        return False
    
    print("Успешный вход!")
    return True

def save_page_as_pdf(driver, url, filename):
    """Сохранение страницы в PDF"""
    print(f"Сохраняю страницу: {url}")
    
    try:
        # Переход на целевую страницу
        driver.get(url)
        
        # Ожидание загрузки контента
        WebDriverWait(driver, 15).until(
            EC.presence_of_element_located((By.TAG_NAME, "body"))
        )
        
        # Прокрутка для загрузки всего контента
        driver.execute_script("window.scrollTo(0, document.body.scrollHeight);")
        time.sleep(2)
        
        # Создание PDF через команду печати
        pdf_options = {
            'landscape': False,
            'displayHeaderFooter': False,
            'printBackground': True,
            'paperWidth': 8.3,
            'paperHeight': 11.7,
        }
        
        # Получение PDF через Chrome DevTools Protocol
        result = driver.execute_cdp_cmd("Page.printToPDF", pdf_options)
        pdf_data = base64.b64decode(result['data'])
        
        # Сохранение файла
        with open(filename, 'wb') as f:
            f.write(pdf_data)
        
        print(f"Страница успешно сохранена как: {filename}")
        return True
        
    except Exception as e:
        print(f"Ошибка при сохранении страницы: {str(e)}")
        return False

def main():

from selenium.common.exceptions import TimeoutException, NoSuchElementException

def setup_driver():
    """Настройка драйвера"""
    chrome_options = Options()
    chrome_options.add_argument('--headless=new')
    chrome_options.add_argument('--no-sandbox')
    chrome_options.add_argument('--disable-dev-shm-usage')
    chrome_options.add_argument('--disable-gpu')
    chrome_options.add_argument('--window-size=1920,1080')
    chrome_options.add_argument('--disable-blink-features=AutomationControlled')
    chrome_options.add_experimental_option("excludeSwitches", ["enable-automation"])
    chrome_options.add_experimental_option('useAutomationExtension', False)
    
    driver = webdriver.Chrome(options=chrome_options)
    driver.execute_script("Object.defineProperty(navigator, 'webdriver', {get: () => undefined})")
    return driver

def get_all_subsection_links(driver, base_url):
    """Получаем все ссылки на подразделы"""
    print(f"🔍 Ищу все подразделы на странице: {base_url}")
    
    try:
        driver.get(base_url)
        time.sleep(3)
        
        # Сохраняем скриншот для диагностики структуры
        driver.save_screenshot("section_structure.png")
        
        # Различные селекторы для поиска ссылок на подразделы
        subsection_selectors = [
            ".content a",
            ".main-content a",
            "#content a",
            "article a",
            ".section a",
            ".subsection a",
            ".tree a",
            ".nav a",
            "ul a",
            "ol a",
            "li a",
            "table a",
            ".list a",
            ".menu a"
        ]
        
        all_links = set()
        
        for selector in subsection_selectors:
            try:
                elements = driver.find_elements(By.CSS_SELECTOR, selector)
                for element in elements:
                    href = element.get_attribute('href')
                    if href and base_url in href and href != base_url:
                        # Проверяем, что это действительно ссылка на контент (исключаем навигацию и т.д.)
                        link_text = element.text.strip().lower()
                        if (len(link_text) > 3 and 
                            not any(word in link_text for word in ['главная', 'home', 'назад', 'back', 'вперед', 'next']) and
                            href not in all_links):
                            all_links.add(href)
                            print(f"  📎 Найдена ссылка: {link_text} -> {href}")
            except:
                continue
        
        # Также ищем ссылки в основных структурных элементах
        structural_selectors = ["main", "article", "section", ".content", "#content"]
        for selector in structural_selectors:
            try:
                container = driver.find_element(By.CSS_SELECTOR, selector)
                links = container.find_elements(By.TAG_NAME, "a")
                for link in links:
                    href = link.get_attribute('href')
                    link_text = link.text.strip()
                    if (href and base_url in href and href != base_url and 
                        len(link_text) > 3 and href not in all_links):
                        all_links.add(href)
                        print(f"  📎 Найдена в {selector}: {link_text} -> {href}")
            except:
                continue
        
        print(f"✅ Всего найдено ссылок на подразделы: {len(all_links)}")
        return list(all_links)
        
    except Exception as e:
        print(f"❌ Ошибка при поиске подразделов: {str(e)}")
        return []

def save_page_as_pdf(driver, url, filename):
    """Сохранение страницы в PDF"""
    print(f"📄 Сохраняю: {url}")
    
    try:
        driver.get(url)
        time.sleep(4)
        
        # Прокрутка для загрузки всего контента
        last_height = driver.execute_script("return document.body.scrollHeight")
        for i in range(3):
            driver.execute_script("window.scrollTo(0, document.body.scrollHeight);")
            time.sleep(2)
            new_height = driver.execute_script("return document.body.scrollHeight")
            if new_height == last_height:
                break
            last_height = new_height
        
        driver.execute_script("window.scrollTo(0, 0);")
        time.sleep(1)
        
        # Создаем PDF
        pdf_options = {
            'landscape': False,
            'displayHeaderFooter': False,
            'printBackground': True,
            'paperWidth': 8.3,
            'paperHeight': 11.7,
            'scale': 0.8,
            'marginTop': 0.5,
            'marginBottom': 0.5,
            'marginLeft': 0.5,
            'marginRight': 0.5,
        }
        
        result = driver.execute_cdp_cmd("Page.printToPDF", pdf_options)
        pdf_data = base64.b64decode(result['data'])
        
        with open(filename, 'wb') as f:
            f.write(pdf_data)
        
        print(f"✅ Успешно сохранено: {filename}")
        return True
        
    except Exception as e:
        print(f"❌ Ошибка при сохранении {url}: {str(e)}")
        return False
