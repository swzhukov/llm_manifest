import requests
import zipfile
import os
import time
from google.colab import files
from pydub import AudioSegment

class SimpleTranscriber:
    def __init__(self, token):
        self.token = token
        self.asr_url = "https://bit-asr.1bitai.ru"
        self.headers = {"token": token}

    def prepare_audio(self, file_path):
        """Конвертирует аудио в WAV если нужно"""
        try:
            if file_path.lower().endswith('.wav'):
                return file_path
                
            print("1. Конвертирую в WAV...")
            audio = AudioSegment.from_file(file_path)
            audio = audio.set_channels(1).set_frame_rate(16000)
            
            wav_path = file_path + '.wav'
            audio.export(wav_path, format="wav")
            print("   ✅ Конвертация завершена")
            return wav_path
        except Exception as e:
            print(f"   ❌ Ошибка конвертации: {e}")
            return file_path

    def transcribe(self, file_path):
        """Основная функция транскрибации"""
        print("2. Отправляю на распознавание...")
        
        # Подготавливаем аудио
        audio_path = self.prepare_audio(file_path)
        
        try:
            # Загружаем на сервер
            print("3. Загружаю файл на сервер...")
            with open(audio_path, 'rb') as f:
                response = requests.post(
                    f"{self.asr_url}/start_transcribing",
                    files={'file': f},
                    params={'diarize': 'true', 'remove_timestamps': 'false'},
                    headers=self.headers,
                    timeout=60
                )
            
            print(f"4. Ответ сервера: {response.status_code}")
            
            if response.status_code != 200:
                print(f"   ❌ Ошибка: {response.text}")
                return None
                
            task_data = response.json()
            task_id = task_data.get('task_id')
            if not task_id:
                print("   ❌ Не получили ID задачи")
                return None
                
            print(f"   ✅ Задача создана: {task_id}")
            
            # Ожидаем завершения
            print("5. Ожидаю завершения обработки...")
            max_attempts = 20
            attempt = 0
            
            while attempt < max_attempts:
                attempt += 1
                print(f"   Проверка {attempt}/{max_attempts}...")
                
                try:
                    status_response = requests.get(
                        f"{self.asr_url}/get_status",
                        params={'task_id': task_id},
                        headers=self.headers,
                        timeout=30
                    )
                    
                    if status_response.status_code == 200:
                        status_data = status_response.json()
                        progress = status_data.get('progress', 0)
                        status = status_data.get('status')
                        
                        print(f"   Прогресс: {progress}%, Статус: {status}")
                        
                        if status == 'done':
                            print("   ✅ Обработка завершена!")
                            break
                        elif status == 'error':
                            print(f"   ❌ Ошибка: {status_data.get('error', 'Неизвестная ошибка')}")
                            return None
                    else:
                        print(f"   ⚠️ Ошибка статуса: {status_response.status_code}")
                    
                except Exception as e:
                    print(f"   ⚠️ Ошибка запроса: {e}")
                
                time.sleep(5)
            
            # Получаем результат
            print("6. Получаю результат...")
            result_response = requests.get(
                f"{self.asr_url}/get_file",
                params={'task_id': task_id},
                headers=self.headers,
                timeout=30
            )
            
            print(f"7. Статус получения результата: {result_response.status_code}")
            
            if result_response.status_code == 200:

                    if not text:
                        # Если текста нет в корне, ищем в других местах

                    if text:

                        header = f"ФАЙЛ: {os.path.basename(file_path)}\n"
                        header += f"ВРЕМЯ: {time.strftime('%Y-%m-%d %H:%M:%S')}\n"
                        header += "=" * 50 + "\n\n"
                        
                        final_text = header + text

                    else:

### 💾 Работа с сессионным хранилищем Colab

Вы можете загружать файлы непосредственно в оперативную память Colab. Это делается через панель управления файлами или кодом, как в вашем предыдущем скрипте.

- **Загрузка через интерфейс**: В левой панели Colab есть вкладка «Файлы». Нажав на иконку 📤 (**Upload to session storage**), вы можете загрузить файлы с компьютера. Они будут сохранены в папке `/content/`[citation:2][citation:4].
- **Загрузка кодом**: Используйте модуль `files` из `google.colab` для загрузки файлов напрямую в ноутбук[citation:2][citation:8].
    ```python
    from google.colab import files
    uploaded = files.upload()  # Откроется диалоговое окно для выбора файла
    ```
    После загрузки файлы также будут доступны в рабочей директории `/content/`.

Все файлы, загруженные этими способами, находятся в **сессионном хранилище** — временной памяти Colab[citation:8]. Colab не гарантирует сохранность этих данных. Сессия может прерваться по нескольким причинам[citation:1][citation:5]:
*   **Неактивность**: После примерно **90 минут** бездействия.
*   **Максимальное время жизни**: Даже при активной работе сессия завершится через **12 часов**.
*   **Технические сбои**: Переподключение к runtime или другие неполадки.

После разрыва сессии все загруженные аудиофайлы, созданные чанки и временные результаты будут безвозвратно утеряны.

### ⚙️ Подключение к Облаку Mail.ru через API

Прямое использование API обеспечивает максимальный контроль над операциями с файлами. Судя по коду на Go[citation:4] и обсуждениям в rclone[citation:8], общая схема работы выглядит так:

1.  **Аутентификация**: Получите токен доступа с помощью логина и пароля от аккаунта Mail.ru.
2.  **Получение шардов**: Запросите URL-адреса для загрузки (`Upload`) и скачивания (`Get`) файлов.
3.  **Загрузка файла (PUT-запрос)**: Отправьте файл на полученный URL-адрес шарда для загрузки.

К сожалению, в результатах поиска нет готовой официальной Python-библиотеки, поэтому запросы нужно будет делать самостоятельно, например, через библиотеку `requests`.

Вот примерный каркас класса для работы с API, построенный на основе доступной информации:

```python
import requests
import json
from pathlib import Path

class MailRuCloudAPI:
    def __init__(self, login, password):
        self.login = login
        self.password = password
        self.auth_token = None
        self.upload_url = None
        self.session = requests.Session()
        # Предполагаемые эндпоинты API (могут потребовать уточнения)
        self.auth_url = "https://auth.mail.ru/cgi-bin/auth"
        self.token_url = "https://oauth.mail.ru/token"
        self.api_url = "https://cloud.mail.ru/api/v2"
        
    def auth(self):
        """Аутентификация в Облаке Mail.ru и получение токена"""
        # ВНИМАНИЕ: Процесс аутентификации может быть сложным и
        # требовать нескольких шагов (OAuth 2.0).
        # Это упрощенный пример, который, вероятно, нуждается в доработке.
        auth_data = {
            'login': self.login,
            'password': self.password,
            'client_id': 'cloud-win', # Пример client_id
            'grant_type': 'password'
        }
        try:
            response = self.session.post(self.token_url, data=auth_data)
            response.raise_for_status()
            token_data = response.json()
            self.auth_token = token_data['access_token']
            print("✅ Успешная аутентификация")
        except Exception as e:
            print(f"❌ Ошибка аутентификации: {e}")
            
    def get_shards(self):
        """Получение URL-адресов шардов для загрузки/скачивания"""
        if not self.auth_token:
            print("Сначала выполните аутентификацию")
            return
            
        shard_params = {'token': self.auth_token}
        try:
            response = self.session.get(f"{self.api_url}/dispatcher", params=shard_params)
            response.raise_for_status()
            shard_info = response.json()
            # Извлекаем URL для загрузки (предполагаемая структура JSON)
            self.upload_url = shard_info['body']['upload'][0]['url']
            print("✅ Шарды получены")
        except Exception as e:
            print(f"❌ Ошибка получения шардов: {e}")
    
    def upload_file(self, local_file_path, cloud_file_path):
        """Загрузка файла в Облако Mail.ru"""
        if not self.upload_url:
            print("Сначала получите шарды")
            return False
            
        # Кодируем путь в облаке для использования в URL
        from urllib.parse import quote
        encoded_cloud_path = quote(cloud_file_path, safe='')
        
        # 1. Запрашиваем URL для загрузки конкретного файла
        upload_get_url = f"{self.upload_url}/?token={self.auth_token}&path={encoded_cloud_path}"
        
        try:
            response = self.session.get(upload_get_url)
            response.raise_for_status()
            upload_data = response.json()
            
            # 2. Отправляем файл на полученный URL (HTTP PUT)
            with open(local_file_path, 'rb') as file:
                put_response = requests.put(upload_data['body']['url'], data=file)
            
            put_response.raise_for_status()
            print(f"✅ Файл {local_file_path} успешно загружен в Облако")
            return True
            
        except Exception as e:
            print(f"❌ Ошибка загрузки файла: {e}")
            return False
```
