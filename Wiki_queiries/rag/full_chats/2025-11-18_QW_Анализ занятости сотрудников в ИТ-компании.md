def is_working_day_russia(check_date, holidays, extra_working_days):
    """
    Определяет, является ли день рабочим по российскому календарю
    с учётом праздников и переносов
    """
    # Если день в списке дополнительных рабочих дней (переносы)
    if check_date in extra_working_days:
        return True
    
    # Если день выходной (суббота, воскресенье)
    if check_date.weekday() >= 5:  # 5=суббота, 6=воскресенье
        return False
    
    # Если день праздничный
    if check_date in holidays:
        return False
    
    return True

def get_russian_working_days(start_date, end_date, holidays, extra_working_days):
    """Определяет рабочие дни по российскому производственному календарю"""
    working_days = []
    current = start_date.date()
    end = end_date.date()
    
    while current <= end:
        if is_working_day_russia(current, holidays, extra_working_days):
            working_days.append(current)
        current += timedelta(days=1)
    
    return working_days

def analyze_report_quality(descriptions):
    """Анализирует качество описаний работ"""
    descriptions = [str(d).strip() for d in descriptions if pd.notna(d)]
    
    if not descriptions:
        return 0, ["Все описания пустые"]
    
    quality_score = 100
    issues = []
    
    # Анализ пустых описаний
    empty_count = sum(1 for d in descriptions if len(d) < 5)
    if empty_count > 0:
        penalty = CONFIG['quality_penalties']['empty_description']
        quality_score -= penalty
        issues.append(f"Пустые описания: {empty_count}")
    
    # Анализ повторяемости
    desc_counter = Counter(descriptions)
    repeated = [(desc, count) for desc, count in desc_counter.items() 
                if count > CONFIG['max_repeated_descriptions'] and len(desc) < 100]
    
    if repeated:
        penalty = len(repeated) * CONFIG['quality_penalties']['repeated_description']
        quality_score -= penalty
        issues.append(f"Повторяющиеся формулировки: {len(repeated)}")
    
    # Анализ конкретности
    vague_count = sum(1 for d in descriptions if 5 < len(d) < CONFIG['min_description_length'])
    vague_ratio = vague_count / len(descriptions)
    
    if vague_ratio > 0.3:
        penalty = CONFIG['quality_penalties']['vague_description']
        quality_score -= penalty
        issues.append(f"Слишком краткие описания: {vague_count}")
    
    return max(quality_score, 0), issues

def analyze_employee_performance(employee_name, employee_data, working_days_list, expected_hours):
    """Анализирует эффективность одного сотрудника"""
    result = {
        'name': employee_name,
        'total_hours': employee_data['Часы'].sum(),
        'project_hours': 0,
        'admin_hours': 0,
        'training_hours': 0,
        'vacation_hours': 0,
        'sales_hours': 0,
        'gap_hours': 0,
        'gap_days': [],
        'vacation_days': 0,
        'efficiency_percent': 0,
        'quality_score': 0,
        'quality_issues': [],
        'total_score': 0
    }
    
    # Распределение по типам работ
    type_hours = employee_data.groupby('Тип_работы')['Часы'].sum()
    for work_type in type_hours.index:
        result[f'{work_type}_hours'] = type_hours[work_type]
    
    # Анализ простоев и отпусков
    daily_hours = employee_data.groupby(employee_data['Дата'].dt.date)['Часы'].sum()
    daily_types = employee_data.groupby(employee_data['Дата'].dt.date)['Тип_работы'].apply(list)
    
    for work_day in working_days_list:
        day_hours = daily_hours.get(work_day, 0)
        day_types = daily_types.get(work_day, [])
        
        # Учитываем отпуск
        if 'vacation' in day_types:
            result['vacation_days'] += 1
            continue
        
        # Проверяем простой (только для рабочих дней)
        if day_hours < CONFIG['expected_hours_per_day']:
            gap = CONFIG['expected_hours_per_day'] - day_hours
            result['gap_hours'] += gap
            result['gap_days'].append(work_day)
    
    # Расчет эффективности с учетом российского календаря
    adjusted_expected = expected_hours - (result['vacation_days'] * CONFIG['expected_hours_per_day'])
    if adjusted_expected > 0:
        result['efficiency_percent'] = (result['project_hours'] / adjusted_expected) * 100

def analyze_suspicious_tasks(employee_data):
    """Анализ подозрительно долгих задач"""
    suspicious_tasks = []
    
    for _, task in employee_data.iterrows():
        hours = task['Часы']
        description = str(task['Описание']).lower()

            suspicious_tasks.append({
                'date': task['Дата'],
                'hours': hours,
                'description': task['Описание'],
                'project': task['Проект'],
                'reason': 'Задача превышает 8 часов'
            })
        
        # Проверяем слишком общие описания с большими трудозатратами

        if any(pattern in description for pattern in vague_patterns) and hours > 4:
            suspicious_tasks.append({
                'date': task['Дата'],

                'description': task['Описание'],
                'project': task['Проект'],
