Вот пример кода на Python, который реализует базовые функции, аналогичные приложению [byebyedpi](https://github.com/ValdikSS/GoodbyeDPI) для обхода DPI-блокировок:

```python
import socket
import ssl
import threading
from scapy.all import *
from scapy.layers.inet import IP, TCP

# Конфигурация
LOCAL_HOST = '127.0.0.1'
LOCAL_PORT = 8080
FAKE_SNI = "example.com"
TARGET_DOMAINS = ["blocked-site.com", "restricted.org"]

def modify_packet(pkt):
    """Модификация TCP-пакетов для обхода DPI"""
    if pkt.haslayer(TCP) and pkt.haslayer(IP):
        # Изменяем порядок флагов для обхода примитивных DPI
        if pkt[TCP].flags == 'S':
            pkt[TCP].flags = 'FSU'
        
        # Фрагментация пакетов
        pkt[IP].flags = 1  # Устанавливаем флаг MF (More Fragments)
        pkt[IP].frag = 0

        # Подмена SNI в TLS-пакетах
        if pkt[TCP].dport == 443 and pkt.haslayer(Raw):
            load = pkt[Raw].load
            if b'blocked-site.com' in load:
                pkt[Raw].load = load.replace(b'blocked-site.com', FAKE_SNI.encode())
        
        # Удаляем проверочные поля и вычисляем заново
        del pkt[IP].chksum
        del pkt[TCP].chksum
        return pkt

def handle_client(client_socket):
    """Обработка клиентского соединения"""
    try:
        request = client_socket.recv(4096)
        
        # Подменяем Host-заголовок для HTTP-запросов
        if b'Host: blocked-site.com' in request:
            request = request.replace(b'Host: blocked-site.com', f'Host: {FAKE_SNI}'.encode())

        # Перенаправляем трафик через внешний прокси
        remote_socket = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
        remote_socket.connect(('8.8.8.8', 80))  # Пример внешнего адреса
        remote_socket.send(request)

        while True:
            response = remote_socket.recv(4096)
            if not response:
                break
            client_socket.send(response)
            
    except Exception as e:
        print(f"Ошибка: {e}")
    finally:
        client_socket.close()

def start_proxy():
    """Запуск локального прокси-сервера"""
    server = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
    server.bind((LOCAL_HOST, LOCAL_PORT))
    server.listen(5)
    print(f"Прокси запущен на {LOCAL_HOST}:{LOCAL_PORT}")

    while True:
        client_socket, addr = server.accept()
        print(f"Подключение от {addr}")
        client_thread = threading.Thread(target=handle_client, args=(client_socket,))
        client_thread.start()

def packet_sniffer():
    """Сниффинг и модификация сетевого трафика"""
    print("Запуск сниффера...")
    sniff(filter="tcp", prn=modify_packet, store=0)

if __name__ == "__main__":
    # Запуск прокси в отдельном потоке
    proxy_thread = threading.Thread(target=start_proxy)
    proxy_thread.daemon = True
    proxy_thread.start()

    # Запуск сниффера (требует прав администратора)
    try:
        packet_sniffer()
    except PermissionError:
        print("Требуются права администратора для сниффинга пакетов")
```

Этот код реализует два основных метода обхода DPI:

1. **Локальный прокси-сервер**:
   - Подменяет Host-заголовки в HTTP-запросах
   - Перенаправляет трафик через внешний сервер
   - Работает на порту 8080

2. **Модификация сетевых пакетов** (требует прав root/admin):
   - Изменяет TCP-флаги
   - Добавляет фрагментацию пакетов
   - Подменяет SNI в TLS-соединениях

Для использования:
1. Установите зависимости: `pip install scapy`
2. Настройте браузер на использование прокси 127.0.0.1:8080
3. Запустите скрипт с правами администратора

**Важные замечания**:
1. Это упрощенная демонстрационная версия
2. Для полной функциональности требуется доработка обработки HTTPS
3. Некоторые методы могут нарушать работу сетевого стека
4. Актуальные реализации DPI могут использовать более сложные методы обнаружения

Рекомендуется использовать оригинальное приложение [GoodbyeDPI](https://github.com/ValdikSS/GoodbyeDPI) для реального применения, так как оно содержит более продвинутые методы обхода и регулярно обновляется.
