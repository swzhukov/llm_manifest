    """Извлекает текст из различных форматов ответа"""

    if isinstance(result_data, str):
        return result_data
    
    # JSON с разными структурами
    text = result_data.get('text') or result_data.get('result')

@contextlib.contextmanager
def temporary_audio_file(content, filename):
    """Контекстный менеджер для временных файлов"""
    temp_dir = tempfile.mkdtemp()
    temp_path = os.path.join(temp_dir, filename)

            f.write(content)
        yield temp_path

### 💡 Решение 2: Использование виртуального окружения (Более надежное)
