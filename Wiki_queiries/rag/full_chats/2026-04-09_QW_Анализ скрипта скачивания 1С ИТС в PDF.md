class ITS1CDownloader:
    def __init__(self, login, password, delay_min=1.5, delay_max=3.5):
        self.login, self.password = login, password
        self.delay_min, self.delay_max = delay_min, delay_max
        self._req_count = 0
        self.session = requests.Session()
        self.session.headers.update({
            "User-Agent": random.choice(USER_AGENTS),
            "Accept": "text/html,application/xhtml+xml,application/xml;q=0.9,*/*;q=0.8",
            "Accept-Language": "ru-RU,ru;q=0.9,en;q=0.8",
        })

    def _sleep(self):
        time.sleep(random.uniform(self.delay_min, self.delay_max))

    def _get(self, url, **kwargs):
        self._req_count += 1
        if self._req_count % 20 == 0:
            self.session.headers["User-Agent"] = random.choice(USER_AGENTS)
        self._sleep()
        for attempt in range(3):
            try:
                r = self.session.get(url, timeout=30, **kwargs)
                r.raise_for_status()
                return r
            except requests.RequestException as e:
                log.warning("Попытка %d/3 [%s]: %s", attempt+1, url, e)
                if attempt < 2:
                    time.sleep(5 * (attempt + 1))
        raise RuntimeError("Не удалось загрузить: " + url)

    def authorize(self):
        log.info("Авторизация: %s", self.login)
        log.info("Получение cookies its.1c.ru...")
        self.session.get(BASE_URL, timeout=30)
        time.sleep(random.uniform(1.0, 2.0))
        cas_url = CAS_LOGIN_URL + "?service=" + quote(LOGIN_SERVICE_URL)
        r1 = self.session.get(cas_url, timeout=30)
        soup = BeautifulSoup(r1.text, "html.parser")
        exec_input = soup.find("input", {"name": "execution"})
        if not exec_input:
            log.error("Execution token не найден")
            return False
        execution = exec_input.get("value", "")
        time.sleep(random.uniform(1.0, 2.0))
        r2 = self.session.post(cas_url,
            data={"username": self.login, "password": self.password,
                  "execution": execution, "_eventId": "submit", "geolocation": ""},
            allow_redirects=False, timeout=30)
        if r2.status_code not in (301, 302):
            log.error("Ошибка входа HTTP %d", r2.status_code)
            return False
        ticket_url = r2.headers.get("location", "")
        if not ticket_url:
            log.error("Тикет не получен")
            return False
        time.sleep(1)
        self.session.get(ticket_url, allow_redirects=False, timeout=30)
        time.sleep(1)
        self.session.get(LOGIN_SERVICE_URL, allow_redirects=True, timeout=30)
        bitrix_login = next((c.value for c in self.session.cookies if c.name == "BITRIX_SM_LOGIN"), None)
        if bitrix_login:
            log.info("Авторизован: %s", bitrix_login)
            return True
        log.error("Авторизация не выполнена")
        return False

    def get_toc(self, book_code):
        log.info("Загрузка оглавления: %s", book_code)
        r = self._get(BASE_URL + "/db/metadata/" + book_code + "/toc/1_")
        html = r.content.decode("cp1251", errors="replace")
        soup = BeautifulSoup(html, "html.parser")
        pages, seen_ids = [], set()
        def process_ul(ul_el, level=0):
            for li in ul_el.find_all("li", recursive=False):
                link = li.find("a", href=re.compile(r"/db/" + re.escape(book_code) + r"/content/\d+/hdoc"))
                if link:
                    href = link.get("href", "")
                    m = re.search(r"/content/(\d+)/hdoc", href)
                    if m:
                        pid = int(m.group(1))
                        if pid not in seen_ids:
                            seen_ids.add(pid)
                            pages.append({"id": pid, "title": link.get_text(strip=True), "level": level})
                for sub_ul in li.find_all("ul", recursive=False):
                    process_ul(sub_ul, level + 1)
        root_ul = soup.find("ul")
        if root_ul:
            process_ul(root_ul)
        log.info("Найдено разделов: %d", len(pages))
        return pages

    def get_page_content(self, book_code, page_id):
        r = self._get(BASE_URL + "/db/" + book_code + "/content/" + str(page_id) + "/hdoc")
        soup = BeautifulSoup(r.content.decode("cp1251", errors="replace"), "html.parser")
        iframe = soup.find("iframe", id="w_metadata_doc_frame")
        if iframe:
            src = iframe.get("src", "")
        else:
            base = soup.find("base")
            src = base.get("href", "") if base else ""
        if not src:
            return ""
        r2 = self._get(urljoin(BASE_URL, src))
        return r2.content.decode("cp1251", errors="replace")

    def download_image(self, book_code, img_src):
        img_name = img_src.split("?")[0]
        img_url = BASE_URL + "/db/content/" + book_code + "/src/" + img_name
        try:
            r = self.session.get(img_url, timeout=20)
            if r.status_code == 200 and r.content:
                return r.content
        except Exception as e:
            log.warning("Ошибка картинки %s: %s", img_url, e)
        return None

    def download_book(self, book_code):
        toc = self.get_toc(book_code)
        if not toc:
            return [], {}
        pages_content, total = {}, len(toc)
        for i, page in enumerate(toc, 1):
            pid = page["id"]
            log.info("[%d/%d] %s", i, total, page["title"][:65])
            try:
                pages_content[pid] = self.get_page_content(book_code, pid)
            except Exception as e:
                log.warning("Ошибка стр. %d: %s", pid, e)
                pages_content[pid] = ""
        downloaded = sum(1 for v in pages_content.values() if v)
        log.info("Загружено: %d/%d", downloaded, total)
        return toc, pages_content


def register_fonts():
    candidates = [
        ("/usr/share/fonts/truetype/dejavu/DejaVuSans.ttf",      "DejaVuSans"),
        ("/usr/share/fonts/truetype/dejavu/DejaVuSans-Bold.ttf",  "DejaVuSans-Bold"),
        ("/usr/share/fonts/truetype/dejavu/DejaVuSansMono.ttf",   "DejaVuSansMono"),
        ("/usr/share/fonts/truetype/liberation/LiberationSans-Regular.ttf", "LiberationSans"),
        ("/usr/share/fonts/truetype/liberation/LiberationSans-Bold.ttf",    "LiberationSans-Bold"),
    ]
    for path, name in candidates:
        if os.path.exists(path) and name not in pdfmetrics.getRegisteredFontNames():
            try:
                pdfmetrics.registerFont(TTFont(name, path))
            except Exception:
                pass
    reg = pdfmetrics.getRegisteredFontNames()
    if "DejaVuSans" in reg:
        fn = "DejaVuSans"
        fb = "DejaVuSans-Bold" if "DejaVuSans-Bold" in reg else fn
        fm = "DejaVuSansMono" if "DejaVuSansMono" in reg else fn
        return fn, fb, fm
    if "LiberationSans" in reg:
        fn = "LiberationSans"
        fb = "LiberationSans-Bold" if "LiberationSans-Bold" in reg else fn
        return fn, fb, fn
    return "Helvetica", "Helvetica-Bold", "Courier"


def clean_for_pdf(text, max_len=4000):
    text = re.sub(r"<[^>]+>", "", str(text))
    text = re.sub(r"\s+", " ", text).strip()
    return text.replace("&", "&amp;").replace("<", "&lt;").replace(">", "&gt;")[:max_len]


def prepare_image(img_bytes, max_w_cm=15.0):
    try:
        pil_img = PILImage.open(io.BytesIO(img_bytes))
        if pil_img.mode in ("RGBA", "LA", "P"):
            bg = PILImage.new("RGB", pil_img.size, (255, 255, 255))
            if pil_img.mode == "P":
                pil_img = pil_img.convert("RGBA")
            bg.paste(pil_img, mask=pil_img.split()[-1] if pil_img.mode in ("RGBA","LA") else None)
            pil_img = bg
        elif pil_img.mode != "RGB":
            pil_img = pil_img.convert("RGB")
        w, h = pil_img.size
        w_pt = w * 72.0 / 96
        h_pt = h * 72.0 / 96
        max_w_pt = max_w_cm * cm
        if w_pt > max_w_pt:
            h_pt = h_pt * (max_w_pt / w_pt)
            w_pt = max_w_pt
        buf = io.BytesIO()
        pil_img.save(buf, format="PNG", optimize=True)
        buf.seek(0)
        return RLImage(buf, width=w_pt, height=h_pt)
    except Exception as e:
        log.warning("Ошибка обработки картинки: %s", e)
        return None


def make_link(text, href, toc_ids, book_code):
    """
    Создать текст ссылки:
    - Если ведёт на страницу из книги, которая в TOC -> внутренний anchor (#anchor_N)
    - Если ведёт на страницу книги вне TOC -> внешняя ссылка на its.1c.ru
    - Если внешняя ссылка -> внешняя как есть
    """
    c = clean_for_pdf(text)
    if not c:
        return c
    m = re.search(r"/content/(\d+)/hdoc", href)
    if m:
        pid = int(m.group(1))
        if pid in toc_ids:
            return '<link href="#anchor_' + str(pid) + '"><font color="#1a5ab5"><u>' + c + '</u></font></link>'
        else:
            ext_url = BASE_URL + "/db/" + book_code + "/content/" + str(pid) + "/hdoc"
            return '<link href="' + ext_url + '"><font color="#1a5ab5"><u>' + c + '</u></font></link>'
    if href.startswith("http"):
        safe = href.replace("&", "&amp;")
        return '<link href="' + safe + '"><font color="#1a5ab5"><u>' + c + '</u></font></link>'
    return c


def html_to_blocks(html, book_code, downloader, toc_ids):
    if not html:
        return []
    soup = BeautifulSoup(html, "html.parser")
    for tag in soup.find_all(["script", "style", "meta", "link", "noscript"]):
        tag.decompose()
    blocks = []

    def walk(el):
        if not hasattr(el, "name") or el.name is None:
            t = str(el).strip()
            if t and len(t) > 1:
                blocks.append(("body", clean_for_pdf(t)))
            return
        tag = el.name

        if tag in ("h1","h2","h3","h4","h5","h6"):
            t = el.get_text(" ", strip=True)
            if t:
                blocks.append((tag, clean_for_pdf(t)))

        elif tag == "p":
            parts = []
            for ch in el.children:
                if not hasattr(ch, "name") or ch.name is None:
                    t = str(ch).strip()
                    if t:
                        parts.append(clean_for_pdf(t))
                elif ch.name == "img":
                    src = ch.get("src", "")
                    if src:
                        if parts:
                            blocks.append(("body", " ".join(parts)))
                            parts = []
                        ib = downloader.download_image(book_code, src)
                        if ib:
                            ri = prepare_image(ib)
                            if ri:
                                blocks.append(("__img__", ri))
                elif ch.name == "a":
                    href = ch.get("href", "")
                    txt = ch.get_text(strip=True)
                    if txt:
                        parts.append(make_link(txt, href, toc_ids, book_code))
                elif ch.name in ("b", "strong"):
                    t = ch.get_text(strip=True)
                    if t:
                        parts.append("<b>" + clean_for_pdf(t) + "</b>")
                elif ch.name in ("i", "em"):
                    t = ch.get_text(strip=True)
                    if t:
                        parts.append("<i>" + clean_for_pdf(t) + "</i>")
                else:
                    t = ch.get_text(strip=True)
                    if t:
                        parts.append(clean_for_pdf(t))
            if parts:
                blocks.append(("body", " ".join(parts)))

        elif tag == "img":
            src = el.get("src", "")
            if src:
                ib = downloader.download_image(book_code, src)
                if ib:
                    ri = prepare_image(ib)
                    if ri:
                        blocks.append(("__img__", ri))

        elif tag in ("ul", "ol"):
            for li in el.find_all("li", recursive=False):
                parts = []
                for ch in li.children:
                    if not hasattr(ch, "name") or ch.name is None:
                        t = str(ch).strip()
                        if t:
                            parts.append(clean_for_pdf(t))
                    elif ch.name == "a":
                        href = ch.get("href", "")
                        txt = ch.get_text(strip=True)
                        if txt:
                            parts.append(make_link(txt, href, toc_ids, book_code))
                    else:
                        t = ch.get_text(strip=True)
                        if t:
                            parts.append(clean_for_pdf(t))
                if parts:
                    blocks.append(("li", "- " + " ".join(parts)))

        elif tag in ("pre", "code"):
            t = el.get_text(strip=True)
            if t:
                blocks.append(("pre", t))

        elif tag == "table":
            for row in el.find_all("tr"):
                rt = " | ".join(c.get_text(strip=True) for c in row.find_all(["td","th"]))
                if rt.strip():
                    blocks.append(("tbl", clean_for_pdf(rt)))

        elif tag in ("div","section","article","main","body","html","span"):
            for ch in el.children:
                walk(ch)
        else:
            t = el.get_text(" ", strip=True)
            if t and len(t) > 2:
                blocks.append(("body", clean_for_pdf(t)))

    body = soup.find("body")
    for ch in (body if body else soup).children:
        walk(ch)
    return blocks

    for page in toc:
        lvl = min(page["level"], 3)
        prefix = "" if lvl == 0 else "- "
        title_c = clean_for_pdf(page["title"])
        toc_line = '<link href="#anchor_' + str(page["id"]) + '"><font color="#1a5ab5">' + prefix + title_c + "</font></link>"

    total = len(toc)
    for i, page in enumerate(toc):
        pid, level = page["id"], page["level"]
        html = pages_content.get(pid, "")
        if level == 0 and i > 0:
            story.append(PageBreak())
        story.append(AnchorFlowable("anchor_" + str(pid)))
        story.append(Paragraph(clean_for_pdf(page["title"]), ch_sm.get(min(level,3), S["ch3"])))
        if level == 0:
            story.append(HRFlowable(width="100%", thickness=1.5, color=colors.HexColor("#2d4a8a")))
        if html:
            blocks = html_to_blocks(html, book_code, downloader, toc_ids)
            for bt, bd in blocks:
                if bt == "__img__":
                    story.append(Spacer(1, 0.2*cm))
                    story.append(bd)
                    story.append(Spacer(1, 0.2*cm))
                else:
                    if not str(bd).strip():
                        continue
                    st = S.get(bt, S["body"])
                    if bt == "pre":
                        for line in str(bd)[:1500].split("\n")[:40]:
                            if line.strip():
                                story.append(Paragraph(clean_for_pdf(line[:150]), S["pre"]))
                    else:
                        try:
                            story.append(Paragraph(str(bd), st))
                        except Exception:
                            story.append(Paragraph(clean_for_pdf(str(bd)), st))
        story.append(Spacer(1, 0.15*cm))

    doc.build(story)
    size_mb = os.path.getsize(output_path) / 1024 / 1024
    log.info("PDF готов: %s (%.1f MB)", output_path, size_mb)
    return True

## AI

Предоставленный скрипт для Google Colab предназначен для скачивания документации с портала 1С:ИТС (конкретно книга `pubnewui85`) и создания PDF-файла с сохранением структуры, картинок, внутренних и внешних ссылок. Ниже — подробный разбор его работы, возможные проблемы и рекомендации по использованию.

---

## 🔍 Общий принцип работы

1. **Авторизация**  
   Используется CAS-аутентификация 1С (логин/пароль) с получением тикета и последующим установлением сессии.

2. **Получение оглавления**  
   Парсинг страницы `/db/metadata/{book_code}/toc/1_`, извлечение всех разделов с их идентификаторами и уровнями вложенности.

3. **Загрузка содержимого каждой страницы**  
   Для каждого ID страницы загружается HTML-фрейм с основным текстом (через `iframe`). Декодирование происходит в `cp1251`.

4. **Скачивание изображений**  
   Картинки загружаются отдельно по ссылкам внутри страниц, конвертируются в PNG и встраиваются в PDF.

5. **Формирование PDF**

## ⚠️ Потенциальные проблемы и ограничения

| Проблема | Пояснение | Рекомендация |
|----------|-----------|--------------|
| **Жёсткая привязка к кодировке cp1251** | Страницы ИТС реально отдаются в Windows-1251, но если в будущем кодировка изменится — скрипт сломается. | Использовать `r.apparent_encoding` или `chardet` для автоматического определения. |
| **Отсутствие обработки капчи или дополнительных шагов авторизации** | Если вход потребует капчу или двухфакторную аутентификацию, скрипт не справится. | Для автоматизации в таких случаях потребуется более сложная логика или ручное вмешательство. |
| **Большая задержка при скачивании всей книги** | Установлены паузы 1.5–3.5 секунды между запросами, что при большом количестве страниц может занять часы. | Можно уменьшить задержку до 0.5–1 с, но есть риск временной блокировки. Лучше оставить как есть. |
| **Отсутствие сохранения прогресса** | При обрыве соединения придётся начинать заново. | Добавить сохранение уже скачанных страниц в кэш (например, в файлы или переменные). |
| **Логин и пароль в открытом виде** | В текущей версии они жёстко прописаны в начале кода. | В Colab лучше использовать `getpass` или переменные окружения. |
| **Большой расход памяти** | Все страницы и картинки хранятся в оперативной памяти до генерации PDF. Для очень больших книг это может привести к переполнению. | Можно записывать страницы во временные файлы или генерировать PDF по частям (но это усложнит логику оглавления). |
| **Обработка ошибок при скачивании картинок** | Если картинка не загрузилась, она просто пропускается без явного уведомления в итоговом PDF. | Добавить placeholder или текстовое сообщение об отсутствии изображения. |
| **Вёрстка PDF не идеальна** | Используются фиксированные стили, возможны наложения или разрывы страниц в неудобных местах. | Для более точного воспроизведения оригинального форматирования потребуется доработка парсинга CSS/стилей. |
