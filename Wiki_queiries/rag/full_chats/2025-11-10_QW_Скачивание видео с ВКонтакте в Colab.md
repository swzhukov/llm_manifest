Вот пример кода для Google Colab, который реализует описанный алгоритм скачивания видео из ВКонтакте через мобильную версию сайта:

```python
import requests
import re
from urllib.parse import urljoin
import IPython.display as display

def download_vk_video(video_url, quality=720):
    # Добавляем m. после https:// для мобильной версии
    mobile_url = video_url.replace('https://', 'https://m.', 1)
    
    # Заголовки для имитации мобильного браузера
    headers = {
        'User-Agent': 'Mozilla/5.0 (iPhone; CPU iPhone OS 14_0 like Mac OS X) AppleWebKit/605.1.15 (KHTML, like Gecko) Version/14.0 Mobile/15E148 Safari/604.1'
    }
    
    # Получаем HTML-код страницы
    response = requests.get(mobile_url, headers=headers)
    if response.status_code != 200:
        raise Exception("Ошибка при загрузке страницы")
    
    # Ищем ссылку на видео с указанным качеством
    pattern = f'https:[^"\'\\s]*mp4_{quality}[^"\'\\s]*\\.mp4'
    matches = re.findall(pattern, response.text)
    
    if not matches:
        # Пробуем найти альтернативные варианты ссылок
        alt_pattern = f'https:[^"\'\\s]*{quality}p[^"\'\\s]*\\.mp4'
        matches = re.findall(alt_pattern, response.text)
    
    if not matches:
        available_qualities = ['144', '240', '360', '480', '720', '1080']
        for q in available_qualities:
            q_pattern = f'https:[^"\'\\s]*mp4_{q}[^"\'\\s]*\\.mp4'
            if re.findall(q_pattern, response.text):
                print(f"Качество {quality}p не найдено. Доступные качества: {available_qualities}")
                return None
    
    if matches:
        video_link = matches[0]
        # Заменяем Unicode-escape последовательности
        video_link = video_link.encode('utf-8').decode('unicode_escape')
        
        # Скачиваем видео
        video_response = requests.get(video_link, headers=headers, stream=True)
        if video_response.status_code == 200:
            filename = f"vk_video_{quality}p.mp4"
            with open(filename, 'wb') as f:
                for chunk in video_response.iter_content(chunk_size=1024):
                    if chunk:
                        f.write(chunk)
            print(f"Видео успешно скачано: {filename}")
            return filename
        else:
            raise Exception("Ошибка при загрузке видео")
    else:
        raise Exception("Ссылка на видео не найдена")

# Использование
try:
    video_url = input("Введите URL видео ВКонтакте: ")
    quality = input("Введите качество (144, 240, 360, 480, 720, 1080): ") or "720"
    
    file_path = download_vk_video(video_url, int(quality))
    
    if file_path:
        # Показываем видео в колабе
        display.display(display.Video(file_path, embed=True))
        
except Exception as e:
    print(f"Произошла ошибка: {str(e)}")
```

Как использовать:
1. Запустите код в Google Colab
2. Когда появится поле ввода - вставьте ссылку на видео ВК
3. Укажите желаемое качество (или нажмите Enter для качества 720p)
4. Видео будет скачано и показано в интерфейсе Colab

```python
    cleaned = re.sub(r'[<>:"/\\|?*]', '_', filename)
    # Убираем лишние пробелы

    # Ограничиваем длину названия (например, 100 символов)
```

```python
# Установка yt-dlp
!pip install -q yt-dlp

import yt_dlp
from google.colab import files
import re
import os

def clean_filename(filename):
    """Очищает название от недопустимых символов для файлов"""
    cleaned = re.sub(r'[<>:"/\\|?*]', '_', filename)
    cleaned = re.sub(r'\s+', ' ', cleaned).strip()
    return cleaned[:100]

def download_multiple_vk_audios(urls):
    downloaded_files = []
    
    for i, url in enumerate(urls, 1):
        print(f"📥 Обрабатываю ссылку {i}/{len(urls)}...")
        
        try:
            # Сначала получаем информацию о видео
            ydl_info = yt_dlp.YoutubeDL({'quiet': True})
            info = ydl_info.extract_info(url, download=False)
            video_title = info.get('title', f'vk_audio_{i}')
            clean_title = clean_filename(video_title)
            
            print(f"   🎬 Название: {video_title}")
            
            # Настройки для скачивания аудио
            ydl_opts = {
                'format': 'bestaudio/best',
                'outtmpl': f'{clean_title}.%(ext)s',
                'postprocessors': [{
                    'key': 'FFmpegExtractAudio',
                    'preferredcodec': 'mp3',
                    'preferredquality': '192',
                }],
                'quiet': True,
            }
            
            # Скачиваем аудио
            with yt_dlp.YoutubeDL(ydl_opts) as ydl:
                ydl.download([url])
            
            # Проверяем, создался ли файл
            mp3_filename = f"{clean_title}.mp3"
            if os.path.exists(mp3_filename):
                downloaded_files.append(mp3_filename)
                print(f"   ✅ Успешно: {mp3_filename}")
            else:
                print(f"   ❌ Файл не создался: {mp3_filename}")
                
        except Exception as e:
            print(f"   ❌ Ошибка при обработке ссылки {i}: {e}")
            continue
    
    return downloaded_files

# ВВЕДИ СВОИ ССЫЛКИ ЗДЕСЬ (через запятую)
video_urls = [
    "https://vkvideo.ru/video865929390_456240405?list=ln-Fx68FahbYkguRMC9xn&ref_domain=mind-universe.getcourse.ru",
    "https://vkvideo.ru/video123456789_123456789",  # пример второй ссылки
    # добавь столько ссылок, сколько нужно
]

# Запускаем обработку всех ссылок
if video_urls:
    print(f"🎯 Начинаю обработку {len(video_urls)} ссылок...")
    downloaded_files = download_multiple_vk_audios(video_urls)
    
    if downloaded_files:
        print(f"\n📦 Скачано файлов: {len(downloaded_files)}")
        
        # Скачиваем каждый файл на компьютер
        for file in downloaded_files:
            try:
                files.download(file)
                print(f"⬇️  Загружаю на компьютер: {file}")
            except Exception as e:
                print(f"❌ Ошибка при загрузке {file}: {e}")
        
        print("\n🎉 Все файлы обработаны!")
    else:
        print("❌ Не удалось скачать ни одного файла")
else:
    print("⚠️  Список ссылок пуст")
```
