    # Удаляем строки с пустыми датами или сотрудниками
    df = df.dropna(subset=['date', 'employee'])
    
    # Добавляем дополнительные фичи
    df['day_of_week'] = df['date'].dt.day_name()
    df['is_weekend'] = df['day_of_week'].isin(['Saturday', 'Sunday'])
    df['description_length'] = df['description'].str.len()
    df['has_detailed_description'] = df['description_length'] > 20
    df['is_administrative'] = df['project'].str.contains('административн', case=False, na=False)
    
    return df

# Анализ качества описаний
def analyze_description_quality(description):
    if pd.isna(description) or description == 'nan':
        return 0
    
    description = str(description)
    
    score = 0
    # Баллы за конкретные результаты
    if any(word in description.lower() for word in ['сделал', 'завершил', 'реализовал', 'настроил', 'исправил']):
        score += 2
    if any(word in description.lower() for word in ['отчет', 'документ', 'код', 'настройк', 'тест']):
        score += 1
    if len(description) > 50:
        score += 1
    if any(char in description for char in ['№', '#', 'http']):
        score += 1
    if any(word in description.lower() for word in ['ошибк', 'проблем', 'баг']):
        score += 1
    
    return min(score, 5)

# Анализ паттернов работы
def analyze_work_patterns(df):
    patterns = {}
    
    for employee in df['employee'].unique():
        emp_data = df[df['employee'] == employee].copy()
        patterns[employee] = {
            'avg_daily_hours': emp_data.groupby('date')['actual_hours'].sum().mean(),
            'consistency_score': 1 - (emp_data.groupby('date')['actual_hours'].sum().std() / 8),
            'meeting_intensity': len(emp_data[emp_data['description'].str.contains('встреч', case=False, na=False)]) / len(emp_data),
            'documentation_focus': len(emp_data[emp_data['description'].str.contains('документ|инструкц|отчет', case=False, na=False)]) / len(emp_data),
            'unique_descriptions_ratio': emp_data['description'].nunique() / len(emp_data)
        }
    
    return patterns

def detect_anomalies(df):
    anomalies = []
    
    daily_hours = df.groupby(['employee', 'date'])['actual_hours'].sum().reset_index()
    
    for employee in daily_hours['employee'].unique():
        emp_hours = daily_hours[daily_hours['employee'] == employee]['actual_hours']
        
        # Аномально низкая загрузка
        low_hours = daily_hours[(daily_hours['employee'] == employee) & 
                               (daily_hours['actual_hours'] < 4) &
                               (~daily_hours['date'].dt.day_name().isin(['Saturday', 'Sunday']))]
        
        # Аномально высокая загрузка
        high_hours = daily_hours[(daily_hours['employee'] == employee) & 
                                (daily_hours['actual_hours'] > 10)]
        
        for _, row in low_hours.iterrows():
            anomalies.append({
                'employee': employee,
                'date': row['date'],
                'type': 'Низкая загрузка',
                'hours': row['actual_hours'],
                'severity': 'medium'
            })
            
        for _, row in high_hours.iterrows():
            anomalies.append({
                'employee': employee,
                'date': row['date'],
                'type': 'Высокая загрузка',
                'hours': row['actual_hours'],
                'severity': 'high'
            })
    
    return pd.DataFrame(anomalies)
