### 🔍 Анализ проблемы и решение

Ваш код ожидает, что ID задачи будет в поле `id` ответа, но сервер BIT.ASR возвращает его в поле `task_id`[citation:1]. Из-за этого условие `if not task_id` оценивается как `True`, и программа считает, что ID не получен.

Нужно исправить часть кода, которая извлекает `task_id` из ответа API.

    def analyze_diarization(self, text):
        """Анализирует качество диаризации"""
        if not text:
            return {
                'total_segments': 0,
                'unique_speakers': 0,
                'speaker_list': [],
                'speaker_changes': 0,
                'segments_per_minute': 0
            }
            
        lines = text.split('\n')
        speaker_segments = []
        
        # Ищем сегменты с метками спикеров
        for line in lines:
            if any(pattern in line for pattern in ['SPEAKER', 'СПИКЕР', 'Диктор']):
                speaker_segments.append(line)
        
        # Анализ уникальных спикеров
        unique_speakers = set()
        speaker_changes = 0
        last_speaker = None
        
        for segment in speaker_segments:
            current_speaker = None
            patterns = [
                r'\[SPEAKER_(\d+)\]',
                r'\[СПИКЕР_(\d+)\]', 
                r'\[Диктор_(\d+)\]',
                r'\[SPEAKER\]',
                r'\[СПИКЕР\]',
                r'\[Диктор\]'
            ]
            
            for pattern in patterns:
                match = re.search(pattern, segment)
                if match:
                    if len(match.groups()) > 0:
                        current_speaker = f"SPEAKER_{match.group(1)}"
                    else:
                        current_speaker = "SPEAKER"
                    break
            
            if current_speaker:
                unique_speakers.add(current_speaker)
                if current_speaker != last_speaker:
                    speaker_changes += 1
                    last_speaker = current_speaker
        
        return {
            'total_segments': len(speaker_segments),
            'unique_speakers': len(unique_speakers),
            'speaker_list': list(unique_speakers),
            'speaker_changes': max(0, speaker_changes - 1),  # Первый сегмент не считается сменой
            'segments_per_minute': len(speaker_segments) / 11.3  # Примерная длительность
        }

        print(f"[{self.get_time()}] ✅ ZIP-архив создан")
        return zip_buffer

    def analyze_diarization(self, text):
        """Анализирует качество диаризации"""
        if not text:
            return {
                'total_segments': 0,
                'unique_speakers': 0,
                'speaker_list': [],
                'speaker_changes': 0,
                'segments_per_minute': 0
            }
            
        lines = text.split('\n')
        speaker_segments = []
        
        # Ищем сегменты с метками спикеров
        for line in lines:
            if any(pattern in line for pattern in ['SPEAKER', 'СПИКЕР', 'Диктор']):
                speaker_segments.append(line)
        
        # Анализ уникальных спикеров
        unique_speakers = set()
        speaker_changes = 0
        last_speaker = None
        
        for segment in speaker_segments:
            current_speaker = None
            patterns = [
                r'\[SPEAKER_(\d+)\]',
                r'\[СПИКЕР_(\d+)\]', 
                r'\[Диктор_(\d+)\]',
                r'\[SPEAKER\]',
                r'\[СПИКЕР\]',
                r'\[Диктор\]'
            ]
            
            for pattern in patterns:
                match = re.search(pattern, segment)
                if match:
                    if len(match.groups()) > 0:
                        current_speaker = f"SPEAKER_{match.group(1)}"
                    else:
                        current_speaker = "SPEAKER"
                    break
            
            if current_speaker:
                unique_speakers.add(current_speaker)
                if current_speaker != last_speaker:
                    speaker_changes += 1
                    last_speaker = current_speaker
        
        return {
            'total_segments': len(speaker_segments),
            'unique_speakers': len(unique_speakers),
            'speaker_list': list(unique_speakers),
            'speaker_changes': max(0, speaker_changes - 1),  # Первый сегмент не считается сменой
            'segments_per_minute': len(speaker_segments) / 11.3  # Примерная длительность
        }
